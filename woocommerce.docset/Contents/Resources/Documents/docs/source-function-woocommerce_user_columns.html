 36 /**
 37  * Define values for custom columns.
 38  *
 39  * @access public
 40  * @param mixed $value The value of the column being displayed
 41  * @param mixed $column_name The name of the column being displayed
 42  * @param mixed $user_id The ID of the user being displayed
 43  * @return string Value for the column
 44  */
 45 function woocommerce_user_column_values( $value, $column_name, $user_id ) {
 46     global $woocommerce, $wpdb;
 47     switch ( $column_name ) :
 48         case "woocommerce_order_count" :
 49 
 50             if ( ! $count = get_user_meta( $user_id, '_order_count', true ) ) {
 51 
 52                 $count = $wpdb->get_var( "SELECT COUNT(*)
 53                     FROM $wpdb->posts as posts
 54 
 55                     LEFT JOIN {$wpdb->postmeta} AS meta ON posts.ID = meta.post_id
 56                     LEFT JOIN {$wpdb->term_relationships} AS rel ON posts.ID=rel.object_ID
 57                     LEFT JOIN {$wpdb->term_taxonomy} AS tax USING( term_taxonomy_id )
 58                     LEFT JOIN {$wpdb->terms} AS term USING( term_id )
 59 
 60                     WHERE   meta.meta_key       = '_customer_user'
 61                     AND     posts.post_type     = 'shop_order'
 62                     AND     posts.post_status   = 'publish'
 63                     AND     tax.taxonomy        = 'shop_order_status'
 64                     AND     term.slug           IN ( 'completed' )
 65                     AND     meta_value          = $user_id
 66                 " );
 67 
 68                 update_user_meta( $user_id, '_order_count', $count );
 69             }
 70 
 71             $value = '<a href="' . admin_url( 'edit.php?post_status=all&post_type=shop_order&shop_order_status=completed&_customer_user=' . absint( $user_id ) . '' ) . '">' . absint( $count ) . '</a>';
 72 
 73         break;
 74         case "woocommerce_billing_address" :
 75             $address = apply_filters( 'woocommerce_user_column_billing_address', array(
 76                 'first_name'    => get_user_meta( $user_id, 'billing_first_name', true ),
 77                 'last_name'     => get_user_meta( $user_id, 'billing_last_name', true ),
 78                 'company'       => get_user_meta( $user_id, 'billing_company', true ),
 79                 'address_1'     => get_user_meta( $user_id, 'billing_address_1', true ),
 80                 'address_2'     => get_user_meta( $user_id, 'billing_address_2', true ),
 81                 'city'          => get_user_meta( $user_id, 'billing_city', true ),
 82                 'state'         => get_user_meta( $user_id, 'billing_state', true ),
 83                 'postcode'      => get_user_meta( $user_id, 'billing_postcode', true ),
 84                 'country'       => get_user_meta( $user_id, 'billing_country', true )
 85             ), $user_id );
 86 
 87             $formatted_address = $woocommerce->countries->get_formatted_address( $address );
 88 
 89             if ( ! $formatted_address )
 90                 $value = __( 'N/A', 'woocommerce' );
 91             else
 92                 $value = $formatted_address;
 93 
 94             $value = wpautop( $value );
 95         break;
 96         case "woocommerce_shipping_address" :
 97             $address = apply_filters( 'woocommerce_user_column_shipping_address', array(
 98                 'first_name'    => get_user_meta( $user_id, 'shipping_first_name', true ),
 99                 'last_name'     => get_user_meta( $user_id, 'shipping_last_name', true ),
100                 'company'       => get_user_meta( $user_id, 'shipping_company', true ),
101                 'address_1'     => get_user_meta( $user_id, 'shipping_address_1', true ),
102                 'address_2'     => get_user_meta( $user_id, 'shipping_address_2', true ),
103                 'city'          => get_user_meta( $user_id, 'shipping_city', true ),
104                 'state'         => get_user_meta( $user_id, 'shipping_state', true ),
105                 'postcode'      => get_user_meta( $user_id, 'shipping_postcode', true ),
106                 'country'       => get_user_meta( $user_id, 'shipping_country', true )
107             ), $user_id );
108 
109             $formatted_address = $woocommerce->countries->get_formatted_address( $address );
110 
111             if ( ! $formatted_address )
112                 $value = __( 'N/A', 'woocommerce' );
113             else
114                 $value = $formatted_address;
115 
116             $value = wpautop( $value );
117         break;
118         case "woocommerce_paying_customer" :
119 
120             $paying_customer = get_user_meta( $user_id, 'paying_customer', true );
121 
122             if ( $paying_customer )
123                 $value = '<img src="' . $woocommerce->plugin_url() . '/assets/images/success@2x.png" alt="yes" width="16px" />';
124             else
125                 $value = '<img src="' . $woocommerce->plugin_url() . '/assets/images/success-off@2x.png" alt="no" width="16px" />';
126 
127         break;
128     endswitch;
129     return $value;
130 }

284 function woocommerce_save_customer_meta_fields( $user_id ) {
285     if ( ! current_user_can( 'manage_woocommerce' ) )
286         return $columns;
287 
288     $save_fields = woocommerce_get_customer_meta_fields();
289 
290     foreach( $save_fields as $fieldset )
291         foreach( $fieldset['fields'] as $key => $field )
292             if ( isset( $_POST[ $key ] ) )
293                 update_user_meta( $user_id, $key, woocommerce_clean( $_POST[ $key ] ) );
294 }
