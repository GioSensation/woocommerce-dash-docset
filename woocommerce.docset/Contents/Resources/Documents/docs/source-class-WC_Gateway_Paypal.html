 20     /**
 21      * Constructor for the gateway.
 22      *
 23      * @access public
 24      * @return void
 25      */
 26     public function __construct() {
 27         global $woocommerce;
 28 
 29         $this->id           = 'paypal';
 30         $this->icon         = apply_filters( 'woocommerce_paypal_icon', $woocommerce->plugin_url() . '/assets/images/icons/paypal.png' );
 31         $this->has_fields   = false;
 32         $this->liveurl      = 'https://www.paypal.com/cgi-bin/webscr';
 33         $this->testurl      = 'https://www.sandbox.paypal.com/cgi-bin/webscr';
 34         $this->method_title = __( 'PayPal', 'woocommerce' );
 35         $this->notify_url   = str_replace( 'https:', 'http:', add_query_arg( 'wc-api', 'WC_Gateway_Paypal', home_url( '/' ) ) );
 36 
 37         // Load the settings.
 38         $this->init_form_fields();
 39         $this->init_settings();
 40 
 41         // Define user set variables
 42         $this->title            = $this->get_option( 'title' );
 43         $this->description      = $this->get_option( 'description' );
 44         $this->email            = $this->get_option( 'email' );
 45         $this->receiver_email   = $this->get_option( 'receiver_email', $this->email );
 46         $this->testmode         = $this->get_option( 'testmode' );
 47         $this->send_shipping    = $this->get_option( 'send_shipping' );
 48         $this->address_override = $this->get_option( 'address_override' );
 49         $this->debug            = $this->get_option( 'debug' );
 50         $this->form_submission_method = $this->get_option( 'form_submission_method' ) == 'yes' ? true : false;
 51         $this->page_style       = $this->get_option( 'page_style' );
 52         $this->invoice_prefix   = $this->get_option( 'invoice_prefix', 'WC-' );
 53 
 54         // Logs
 55         if ( 'yes' == $this->debug )
 56             $this->log = $woocommerce->logger();
 57 
 58         // Actions
 59         add_action( 'valid-paypal-standard-ipn-request', array( $this, 'successful_request' ) );
 60         add_action( 'woocommerce_receipt_paypal', array( $this, 'receipt_page' ) );
 61         add_action( 'woocommerce_update_options_payment_gateways_' . $this->id, array( $this, 'process_admin_options' ) );
 62 
 63         // Payment listener/API hook
 64         add_action( 'woocommerce_api_wc_gateway_paypal', array( $this, 'check_ipn_response' ) );
 65 
 66         if ( !$this->is_valid_for_use() ) $this->enabled = false;
 67     }

 78 
 79         return true;
 80     }

220     /**
221      * Get PayPal Args for passing to PP
222      *
223      * @access public
224      * @param mixed $order
225      * @return array
226      */
227     function get_paypal_args( $order ) {
228         global $woocommerce;
229 
230         $order_id = $order->id;
231 
232         if ( 'yes' == $this->debug )
233             $this->log->add( 'paypal', 'Generating payment form for order ' . $order->get_order_number() . '. Notify URL: ' . $this->notify_url );
234 
235         if ( in_array( $order->billing_country, array( 'US','CA' ) ) ) {
236             $order->billing_phone = str_replace( array( '( ', '-', ' ', ' )', '.' ), '', $order->billing_phone );
237             $phone_args = array(
238                 'night_phone_a' => substr( $order->billing_phone, 0, 3 ),
239                 'night_phone_b' => substr( $order->billing_phone, 3, 3 ),
240                 'night_phone_c' => substr( $order->billing_phone, 6, 4 ),
241                 'day_phone_a'   => substr( $order->billing_phone, 0, 3 ),
242                 'day_phone_b'   => substr( $order->billing_phone, 3, 3 ),
243                 'day_phone_c'   => substr( $order->billing_phone, 6, 4 )
244             );
245         } else {
246             $phone_args = array(
247                 'night_phone_b' => $order->billing_phone,
248                 'day_phone_b'   => $order->billing_phone
249             );
250         }
251 
252         // PayPal Args
253         $paypal_args = array_merge(
254             array(
255                 'cmd'                   => '_cart',
256                 'business'              => $this->email,
257                 'no_note'               => 1,
258                 'currency_code'         => get_woocommerce_currency(),
259                 'charset'               => 'UTF-8',
260                 'rm'                    => is_ssl() ? 2 : 1,
261                 'upload'                => 1,
262                 'return'                => add_query_arg( 'utm_nooverride', '1', $this->get_return_url( $order ) ),
263                 'cancel_return'         => $order->get_cancel_order_url(),
264                 'page_style'            => $this->page_style,
265 
266                 // Order key + ID
267                 'invoice'               => $this->invoice_prefix . ltrim( $order->get_order_number(), '#' ),
268                 'custom'                => serialize( array( $order_id, $order->order_key ) ),
269 
270                 // IPN
271                 'notify_url'            => $this->notify_url,
272 
273                 // Billing Address info
274                 'first_name'            => $order->billing_first_name,
275                 'last_name'             => $order->billing_last_name,
276                 'company'               => $order->billing_company,
277                 'address1'              => $order->billing_address_1,
278                 'address2'              => $order->billing_address_2,
279                 'city'                  => $order->billing_city,
280                 'state'                 => $order->billing_state,
281                 'zip'                   => $order->billing_postcode,
282                 'country'               => $order->billing_country,
283                 'email'                 => $order->billing_email
284             ),
285             $phone_args
286         );
287 
288         // Shipping
289         if ( $this->send_shipping=='yes' ) {
290             $paypal_args['address_override'] = ( $this->address_override == 'yes' ) ? 1 : 0;
291 
292             $paypal_args['no_shipping'] = 0;
293 
294             // If we are sending shipping, send shipping address instead of billing
295             $paypal_args['first_name']      = $order->shipping_first_name;
296             $paypal_args['last_name']       = $order->shipping_last_name;
297             $paypal_args['company']         = $order->shipping_company;
298             $paypal_args['address1']        = $order->shipping_address_1;
299             $paypal_args['address2']        = $order->shipping_address_2;
300             $paypal_args['city']            = $order->shipping_city;
301             $paypal_args['state']           = $order->shipping_state;
302             $paypal_args['country']         = $order->shipping_country;
303             $paypal_args['zip']             = $order->shipping_postcode;
304         } else {
305             $paypal_args['no_shipping'] = 1;
306         }
307 
308         // If prices include tax or have order discounts, send the whole order as a single item
309         if ( get_option( 'woocommerce_prices_include_tax' ) == 'yes' || $order->get_order_discount() > 0 ) {
310 
311             // Discount
312             $paypal_args['discount_amount_cart'] = $order->get_order_discount();
313 
314             // Don't pass items - paypal borks tax due to prices including tax. PayPal has no option for tax inclusive pricing sadly. Pass 1 item for the order items overall
315             $item_names = array();
316 
317             if ( sizeof( $order->get_items() ) > 0 )
318                 foreach ( $order->get_items() as $item )
319                     if ( $item['qty'] )
320                         $item_names[] = $item['name'] . ' x ' . $item['qty'];
321 
322             $paypal_args['item_name_1']     = sprintf( __( 'Order %s' , 'woocommerce'), $order->get_order_number() ) . " - " . implode( ', ', $item_names );
323             $paypal_args['quantity_1']      = 1;
324             $paypal_args['amount_1']        = number_format( $order->get_total() - $order->get_shipping() - $order->get_shipping_tax() + $order->get_order_discount(), 2, '.', '' );
325 
326             // Shipping Cost
327             // No longer using shipping_1 because
328             //      a) paypal ignore it if *any* shipping rules are within paypal
329             //      b) paypal ignore anything over 5 digits, so 999.99 is the max
330             // $paypal_args['shipping_1']       = number_format( $order->get_shipping() + $order->get_shipping_tax() , 2, '.', '' );
331             if ( ( $order->get_shipping() + $order->get_shipping_tax() ) > 0 ) {
332                 $paypal_args['item_name_2'] = __( 'Shipping via', 'woocommerce' ) . ' ' . ucwords( $order->shipping_method_title );
333                 $paypal_args['quantity_2']  = '1';
334                 $paypal_args['amount_2']    = number_format( $order->get_shipping() + $order->get_shipping_tax() , 2, '.', '' );
335             }
336 
337         } else {
338 
339             // Tax
340             $paypal_args['tax_cart'] = $order->get_total_tax();
341 
342             // Cart Contents
343             $item_loop = 0;
344             if ( sizeof( $order->get_items() ) > 0 ) {
345                 foreach ( $order->get_items() as $item ) {
346                     if ( $item['qty'] ) {
347 
348                         $item_loop++;
349 
350                         $product = $order->get_product_from_item( $item );
351 
352                         $item_name  = $item['name'];
353 
354                         $item_meta = new WC_Order_Item_Meta( $item['item_meta'] );
355                         if ( $meta = $item_meta->display( true, true ) )
356                             $item_name .= ' ( ' . $meta . ' )';
357 
358                         $paypal_args[ 'item_name_' . $item_loop ]   = $item_name;
359                         $paypal_args[ 'quantity_' . $item_loop ]    = $item['qty'];
360                         $paypal_args[ 'amount_' . $item_loop ]      = $order->get_item_subtotal( $item, false );
361 
362                         if ( $product->get_sku() )
363                             $paypal_args[ 'item_number_' . $item_loop ] = $product->get_sku();
364                     }
365                 }
366             }
367 
368             // Discount
369             if ( $order->get_cart_discount() > 0 )
370                 $paypal_args['discount_amount_cart'] = round( $order->get_cart_discount(), 2 );
371 
372             // Fees
373             if ( sizeof( $order->get_fees() ) > 0 ) {
374                 foreach ( $order->get_fees() as $item ) {
375                     $item_loop++;
376 
377                     $paypal_args[ 'item_name_' . $item_loop ]   = $item['name'];
378                     $paypal_args[ 'quantity_' . $item_loop ]    = 1;
379                     $paypal_args[ 'amount_' . $item_loop ]      = $item['line_total'];
380                 }
381             }
382 
383             // Shipping Cost item - paypal only allows shipping per item, we want to send shipping for the order
384             if ( $order->get_shipping() > 0 ) {
385                 $item_loop++;
386                 $paypal_args[ 'item_name_' . $item_loop ]   = __( 'Shipping via', 'woocommerce' ) . ' ' . ucwords( $order->shipping_method_title );
387                 $paypal_args[ 'quantity_' . $item_loop ]    = '1';
388                 $paypal_args[ 'amount_' . $item_loop ]      = number_format( $order->get_shipping(), 2, '.', '' );
389             }
390 
391         }
392 
393         $paypal_args = apply_filters( 'woocommerce_paypal_args', $paypal_args );
394 
395         return $paypal_args;
396     }

454     /**
455      * Process the payment and return the result
456      *
457      * @access public
458      * @param int $order_id
459      * @return array
460      */
461     function process_payment( $order_id ) {
462 
463         $order = new WC_Order( $order_id );
464 
465         if ( ! $this->form_submission_method ) {
466 
467             $paypal_args = $this->get_paypal_args( $order );
468 
469             $paypal_args = http_build_query( $paypal_args, '', '&' );
470 
471             if ( $this->testmode == 'yes' ):
472                 $paypal_adr = $this->testurl . '?test_ipn=1&';
473             else :
474                 $paypal_adr = $this->liveurl . '?';
475             endif;
476 
477             return array(
478                 'result'    => 'success',
479                 'redirect'  => $paypal_adr . $paypal_args
480             );
481 
482         } else {
483 
484             return array(
485                 'result'    => 'success',
486                 'redirect'  => add_query_arg('order', $order->id, add_query_arg('key', $order->order_key, get_permalink(woocommerce_get_page_id('pay' ))))
487             );
488 
489         }
490 
491     }

508     /**
509      * Check PayPal IPN validity
510      **/
511     function check_ipn_request_is_valid() {
512         global $woocommerce;
513 
514         if ( 'yes' == $this->debug )
515             $this->log->add( 'paypal', 'Checking IPN response is valid...' );
516 
517         // Get recieved values from post data
518         $received_values = array( 'cmd' => '_notify-validate' );
519         $received_values += stripslashes_deep( $_POST );
520 
521         // Send back post vars to paypal
522         $params = array(
523             'body'          => $received_values,
524             'sslverify'     => false,
525             'timeout'       => 60,
526             'user-agent'    => 'WooCommerce/' . $woocommerce->version
527         );
528 
529         if ( 'yes' == $this->debug )

735      * get_paypal_order function.
736      *
737      * @access public
738      * @param mixed $posted
739      * @return void
740      */
741     function get_paypal_order( $posted ) {
742         $custom = maybe_unserialize( $posted['custom'] );
743 
744         // Backwards comp for IPN requests
745         if ( is_numeric( $custom ) ) {
746             $order_id = (int) $custom;
747             $order_key = $posted['invoice'];
748         } elseif( is_string( $custom ) ) {
749             $order_id = (int) str_replace( $this->invoice_prefix, '', $custom );
750             $order_key = $custom;
751         } else {
752             list( $order_id, $order_key ) = $custom;
753         }
754 
755         $order = new WC_Order( $order_id );
756 
757         if ( ! isset( $order->id ) ) {
758             // We have an invalid $order_id, probably because invoice_prefix has changed
759             $order_id   = woocommerce_get_order_id_by_order_key( $order_key );
760             $order      = new WC_Order( $order_id );
761         }
762 
763         // Validate key
764         if ( $order->order_key !== $order_key ) {
765             if ( $this->debug=='yes' )
766                 $this->log->add( 'paypal', 'Error: Order Key does not match invoice.' );
767             exit;
768         }
