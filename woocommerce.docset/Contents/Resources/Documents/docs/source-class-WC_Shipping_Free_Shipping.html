156     /**
157      * is_available function.
158      *
159      * @access public
160      * @param mixed $package
161      * @return bool
162      */
163     function is_available( $package ) {
164         global $woocommerce;
165 
166         if ( $this->enabled == "no" ) return false;
167 
168         $ship_to_countries = '';
169 
170         if ( $this->availability == 'specific' ) {
171             $ship_to_countries = $this->countries;
172         } else {
173             if ( get_option('woocommerce_allowed_countries') == 'specific' )
174                 $ship_to_countries = get_option('woocommerce_specific_allowed_countries');
175         }
176 
177         if ( is_array( $ship_to_countries ) )
178             if ( ! in_array( $package['destination']['country'], $ship_to_countries ) )
179                 return false;
180 
181         // Enabled logic
182         $is_available       = false;
183         $has_coupon         = false;
184         $has_met_min_amount = false;
185 
186         if ( in_array( $this->requires, array( 'coupon', 'either', 'both' ) ) ) {
187 
188             if ( $woocommerce->cart->applied_coupons ) {
189                 foreach ($woocommerce->cart->applied_coupons as $code) {
190                     $coupon = new WC_Coupon( $code );
191 
192                     if ( $coupon->is_valid() && $coupon->enable_free_shipping() )
193                         $has_coupon = true;
194                 }
195             }
196         }
197 
198         if ( in_array( $this->requires, array( 'min_amount', 'either', 'both' ) ) ) {
199 
200             if ( isset( $woocommerce->cart->cart_contents_total ) ) {
201 
202                 if ( $woocommerce->cart->prices_include_tax )
203                     $total = $woocommerce->cart->tax_total + $woocommerce->cart->cart_contents_total;
204                 else
205                     $total = $woocommerce->cart->cart_contents_total;
206 
207                 if ( $total >= $this->min_amount )
208                     $has_met_min_amount = true;
209             }
210         }
211 
212         switch ( $this->requires ) {
213             case 'min_amount' :
214                 if ( $has_met_min_amount ) $is_available = true;
215             break;
216             case 'coupon' :
217                 if ( $has_coupon ) $is_available = true;
218             break;
219             case 'both' :
220                 if ( $has_met_min_amount && $has_coupon ) $is_available = true;
221             break;
222             case 'either' :
223                 if ( $has_met_min_amount || $has_coupon ) $is_available = true;
224             break;
225             default :
226                 $is_available = true;
227             break;
228         }
229 
230         return apply_filters( 'woocommerce_shipping_' . $this->id . '_is_available', $is_available );
231     }
