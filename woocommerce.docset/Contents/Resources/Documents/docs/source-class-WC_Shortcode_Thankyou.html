 2 /**
 3  * Thankyou Shortcode
 4  *
 5  * The thankyou page displays after successful checkout and can be hooked into by payment gateways.

13 class WC_Shortcode_Thankyou {
14 
15     /**
16      * Get the shortcode content.
17      *
18      * @access public
19      * @param array $atts
20      * @return string
21      */
22     public static function get( $atts ) {
23         global $woocommerce;
24         return $woocommerce->shortcode_wrapper( array( __CLASS__, 'output' ), $atts );
25     }
26 
27     /**
28      * Output the shortcode.
29      *
30      * @access public
31      * @param array $atts
32      * @return void
33      */
34     public static function output( $atts ) {
35         global $woocommerce;
36 
37         $woocommerce->show_messages();
38 
39         $order = false;
40 
41         // Get the order
42         $order_id  = apply_filters( 'woocommerce_thankyou_order_id', empty( $_GET['order'] ) ? 0 : absint( $_GET['order'] ) );
43         $order_key = apply_filters( 'woocommerce_thankyou_order_key', empty( $_GET['key'] ) ? '' : woocommerce_clean( $_GET['key'] ) );
44 
45         if ( $order_id > 0 ) {
46             $order = new WC_Order( $order_id );
47             if ( $order->order_key != $order_key )
48                 unset( $order );
49         }
50 
51         // Empty awaiting payment session
52         unset( $woocommerce->session->order_awaiting_payment );
53 
54         woocommerce_get_template( 'checkout/thankyou.php', array( 'order' => $order ) );
55     }
56 }
