 165     /**
 166      * List products in a category shortcode
 167      *
 168      * @access public
 169      * @param array $atts
 170      * @return string
 171      */
 172     public function product_category( $atts ){
 173         global $woocommerce, $woocommerce_loop;
 174 
 175         if ( empty( $atts ) ) return;
 176 
 177         extract( shortcode_atts( array(
 178             'per_page'      => '12',
 179             'columns'       => '4',
 180             'orderby'       => 'title',
 181             'order'         => 'desc',
 182             'category'      => ''
 183             ), $atts ) );
 184 
 185         if ( ! $category ) return;
 186 
 187         // Default ordering args
 188         $ordering_args = $woocommerce->query->get_catalog_ordering_args( $orderby, $order );
 189 
 190         $args = array(
 191             'post_type'             => 'product',
 192             'post_status'           => 'publish',
 193             'ignore_sticky_posts'   => 1,
 194             'orderby'               => $ordering_args['orderby'],
 195             'order'                 => $ordering_args['order'],
 196             'posts_per_page'        => $per_page,
 197             'meta_query'            => array(
 198                 array(
 199                     'key'           => '_visibility',
 200                     'value'         => array('catalog', 'visible'),
 201                     'compare'       => 'IN'
 202                 )
 203             ),
 204             'tax_query'             => array(
 205                 array(
 206                     'taxonomy'      => 'product_cat',
 207                     'terms'         => array( esc_attr($category) ),
 208                     'field'         => 'slug',
 209                     'operator'      => 'IN'
 210                 )
 211             )
 212         );
 213 
 214         if ( isset( $ordering_args['meta_key'] ) ) {
 215             $args['meta_key'] = $ordering_args['meta_key'];
 216         }
 217 
 218         ob_start();
 219 
 220         $products = new WP_Query( $args );
 221 
 222         $woocommerce_loop['columns'] = $columns;
 223 
 224         if ( $products->have_posts() ) : ?>
 225 
 226             <?php woocommerce_product_loop_start(); ?>
 227 
 228                 <?php while ( $products->have_posts() ) : $products->the_post(); ?>
 229 
 230                     <?php woocommerce_get_template_part( 'content', 'product' ); ?>
 231 
 232                 <?php endwhile; // end of the loop. ?>
 233 
 234             <?php woocommerce_product_loop_end(); ?>
 235 
 236         <?php endif;
 237 
 238         wp_reset_postdata();
 239 
 240         return '<div class="woocommerce">' . ob_get_clean() . '</div>';
 241     }

 244     /**
 245      * List all (or limited) product categories
 246      *
 247      * @access public
 248      * @param array $atts
 249      * @return string
 250      */
 251     public function product_categories( $atts ) {
 252         global $woocommerce_loop;
 253 
 254         extract( shortcode_atts( array (
 255             'number'     => null,
 256             'orderby'    => 'name',
 257             'order'      => 'ASC',
 258             'columns'    => '4',
 259             'hide_empty' => 1,
 260             'parent'     => ''
 261             ), $atts ) );
 262 
 263         if ( isset( $atts[ 'ids' ] ) ) {
 264             $ids = explode( ',', $atts[ 'ids' ] );
 265             $ids = array_map( 'trim', $ids );
 266         } else {
 267             $ids = array();
 268         }
 269 
 270         $hide_empty = ( $hide_empty == true || $hide_empty == 1 ) ? 1 : 0;
 271 
 272         // get terms and workaround WP bug with parents/pad counts
 273         $args = array(
 274             'orderby'    => $orderby,
 275             'order'      => $order,
 276             'hide_empty' => $hide_empty,
 277             'include'    => $ids,
 278             'pad_counts' => true,
 279             'child_of'   => $parent
 280         );
 281 
 282         $product_categories = get_terms( 'product_cat', $args );
 283 
 284         if ( $parent !== "" )
 285             $product_categories = wp_list_filter( $product_categories, array( 'parent' => $parent ) );
 286 
 287         if ( $number )
 288             $product_categories = array_slice( $product_categories, 0, $number );
 289 
 290         $woocommerce_loop['columns'] = $columns;
 291 
 292         ob_start();
 293 
 294         // Reset loop/columns globals when starting a new loop
 295         $woocommerce_loop['loop'] = $woocommerce_loop['column'] = '';
 296 
 297         if ( $product_categories ) {
 298 
 299             woocommerce_product_loop_start();
 300 
 301             foreach ( $product_categories as $category ) {
 302 
 303                 woocommerce_get_template( 'content-product_cat.php', array(
 304                     'category' => $category
 305                 ) );
 306 
 307             }
 308 
 309             woocommerce_product_loop_end();
 310 
 311         }
 312 
 313         woocommerce_reset_loop();
 314 
 315         return '<div class="woocommerce">' . ob_get_clean() . '</div>';
 316     }

 513     /**
 514      * Display a single product price + cart button
 515      *
 516      * @access public
 517      * @param array $atts
 518      * @return string
 519      */
 520     public function product_add_to_cart( $atts ) {
 521         global $wpdb, $woocommerce;
 522 
 523         if ( empty( $atts ) ) return;
 524 
 525         if ( ! isset( $atts['style'] ) ) $atts['style'] = 'border:4px solid #ccc; padding: 12px;';
 526 
 527         if ( isset( $atts['id'] ) ) {
 528             $product_data = get_post( $atts['id'] );
 529         } elseif ( isset( $atts['sku'] ) ) {
 530             $product_id = $wpdb->get_var( $wpdb->prepare( "SELECT post_id FROM $wpdb->postmeta WHERE meta_key='_sku' AND meta_value='%s' LIMIT 1", $atts['sku'] ) );
 531             $product_data = get_post( $product_id );
 532         } else {
 533             return;
 534         }
 535 
 536         if ( 'product' == $product_data->post_type ) {
 537 
 538             $product = $woocommerce->setup_product_data( $product_data );
 539 
 540             ob_start();
 541             ?>
 542             <p class="product woocommerce" style="<?php echo $atts['style']; ?>">
 543 
 544                 <?php echo $product->get_price_html(); ?>
 545 
 546                 <?php woocommerce_template_loop_add_to_cart(); ?>
 547 
 548             </p><?php
 549 
 550             wp_reset_postdata();
 551 
 552             return ob_get_clean();
 553 
 554         } elseif ( 'product_variation' == $product_data->post_type ) {
 555 
 556             $product = get_product( $product_data->post_parent );
 557 
 558             $GLOBALS['product'] = $product;
 559 
 560             $variation = get_product( $product_data );
 561 
 562             ob_start();
 563             ?>
 564             <p class="product product-variation" style="<?php echo $atts['style']; ?>">
 565 
 566                 <?php echo $product->get_price_html(); ?>
 567 
 568                 <?php
 569 
 570                 $link   = $product->add_to_cart_url();
 571 
 572                 $label  = apply_filters('add_to_cart_text', __( 'Add to cart', 'woocommerce' ));
 573 
 574                 $link = add_query_arg( 'variation_id', $variation->variation_id, $link );
 575 
 576                 foreach ($variation->variation_data as $key => $data) {
 577                     if ($data) $link = add_query_arg( $key, $data, $link );
 578                 }
 579 
 580                 printf('<a href="%s" rel="nofollow" data-product_id="%s" class="button add_to_cart_button product_type_%s">%s</a>', esc_url( $link ), $product->id, $product->product_type, $label);
 581 
 582                 ?>
 583 
 584             </p><?php
 585 
 586             wp_reset_postdata();
 587 
 588             return ob_get_clean();
 589 
 590         }
 591     }
