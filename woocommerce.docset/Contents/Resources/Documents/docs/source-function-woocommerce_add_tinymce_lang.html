 15 /**
 16  * Checks which method we're using to serve downloads
 17  *
 18  * If using force or x-sendfile, this ensures the .htaccess is in place
 19  *
 20  * @access public
 21  * @return void
 22  */
 23 function woocommerce_check_download_folder_protection() {
 24     $upload_dir         = wp_upload_dir();
 25     $downloads_url      = $upload_dir['basedir'] . '/woocommerce_uploads';
 26     $download_method    = get_option('woocommerce_file_download_method');
 27 
 28     if ( $download_method == 'redirect' ) {
 29 
 30         // Redirect method - don't protect
 31         if ( file_exists( $downloads_url . '/.htaccess' ) )
 32             unlink( $downloads_url . '/.htaccess' );
 33 
 34     } else {
 35 
 36         // Force method - protect, add rules to the htaccess file
 37         if ( ! file_exists( $downloads_url . '/.htaccess' ) ) {
 38             if ( $file_handle = @fopen( $downloads_url . '/.htaccess', 'w' ) ) {
 39                 fwrite( $file_handle, 'deny from all' );
 40                 fclose( $file_handle );
 41             }
 42         }
 43     }
 44 
 45     flush_rewrite_rules();
 46 }

 49 /**
 50  * Protect downlodas from ms-files.php in multisite
 51  *
 52  * @access public
 53  * @param mixed $rewrite
 54  * @return string
 55  */
 56 function woocommerce_ms_protect_download_rewite_rules( $rewrite ) {
 57     global $wp_rewrite;
 58 
 59     $download_method    = get_option('woocommerce_file_download_method');
 60 
 61     if (!is_multisite() || $download_method=='redirect') return $rewrite;
 62 
 63     $rule  = "\n# WooCommerce Rules - Protect Files from ms-files.php\n\n";
 64     $rule .= "<IfModule mod_rewrite.c>\n";
 65     $rule .= "RewriteEngine On\n";
 66     $rule .= "RewriteCond %{QUERY_STRING} file=woocommerce_uploads/ [NC]\n";
 67     $rule .= "RewriteRule /ms-files.php$ - [F]\n";
 68     $rule .= "</IfModule>\n\n";
 69 
 70     return $rule . $rewrite;
 71 }

 75  * Removes variations etc belonging to a deleted post, and clears transients
 76  *
 77  * @access public
 78  * @param mixed $id ID of post being deleted
 79  * @return void
 80  */
 81 function woocommerce_delete_post( $id ) {
 82     global $woocommerce, $wpdb;
 83 
 84     if ( ! current_user_can( 'delete_posts' ) )
 85         return;
 86 
 87     if ( $id > 0 ) {
 88 
 89         $post_type = get_post_type( $id );
 90 
 91         switch( $post_type ) {
 92             case 'product' :
 93 
 94                 if ( $child_product_variations =& get_children( 'post_parent=' . $id . '&post_type=product_variation' ) )
 95                     if ( $child_product_variations )
 96                         foreach ( $child_product_variations as $child )
 97                             wp_delete_post( $child->ID, true );
 98 
 99                 if ( $child_products =& get_children( 'post_parent=' . $id . '&post_type=product' ) )
100                     if ( $child_products )
101                         foreach ( $child_products as $child ) {
102                             $child_post = array();
103                             $child_post['ID'] = $child->ID;
104                             $child_post['post_parent'] = 0;
105                             wp_update_post( $child_post );
106                         }
107 
108                 $woocommerce->clear_product_transients();
109 
110             break;
111             case 'product_variation' :
112 
113                 $woocommerce->clear_product_transients();
114 
115             break;
116         }
117     }
118 }

409  *
410  * @access public
411  * @return void
412  */
413 function woocommerce_compile_less_styles() {
414     global $woocommerce;
415 
416     $colors         = array_map( 'esc_attr', (array) get_option( 'woocommerce_frontend_css_colors' ) );
417     $base_file      = $woocommerce->plugin_path() . '/assets/css/woocommerce-base.less';
418     $less_file      = $woocommerce->plugin_path() . '/assets/css/woocommerce.less';
419     $css_file       = $woocommerce->plugin_path() . '/assets/css/woocommerce.css';
420 
421     // Write less file
422     if ( is_writable( $base_file ) && is_writable( $css_file ) ) {
423 
424         // Colours changed - recompile less
425         if ( ! class_exists( 'lessc' ) )
426             include_once('includes/class-lessc.php');
427         if ( ! class_exists( 'cssmin' ) )
428             include_once('includes/class-cssmin.php');
429 
430         try {
431             // Set default if colours not set
432             if ( ! $colors['primary'] ) $colors['primary'] = '#ad74a2';
433             if ( ! $colors['secondary'] ) $colors['secondary'] = '#f7f6f7';
434             if ( ! $colors['highlight'] ) $colors['highlight'] = '#85ad74';
435             if ( ! $colors['content_bg'] ) $colors['content_bg'] = '#ffffff';
436             if ( ! $colors['subtext'] ) $colors['subtext'] = '#777777';
437 
438             // Write new color to base file
439             $color_rules = "
440 @primary:       " . $colors['primary'] . ";
441 @primarytext:   " . woocommerce_light_or_dark( $colors['primary'], 'desaturate(darken(@primary,50%),18%)', 'desaturate(lighten(@primary,50%),18%)' ) . ";
442 
443 @secondary:     " . $colors['secondary'] . ";
444 @secondarytext: " . woocommerce_light_or_dark( $colors['secondary'], 'desaturate(darken(@secondary,60%),18%)', 'desaturate(lighten(@secondary,60%),18%)' ) . ";
445 
446 @highlight:     " . $colors['highlight'] . ";
447 @highlightext:  " . woocommerce_light_or_dark( $colors['highlight'], 'desaturate(darken(@highlight,60%),18%)', 'desaturate(lighten(@highlight,60%),18%)' ) . ";
448 
449 @contentbg:     " . $colors['content_bg'] . ";
450 
451 @subtext:       " . $colors['subtext'] . ";
452             ";
453 
454             file_put_contents( $base_file, $color_rules );
455 
456             $less           = new lessc( $less_file );
457             $compiled_css   = $less->parse();
458 
459             $compiled_css = CssMin::minify( $compiled_css );
460 
461             if ( $compiled_css )
462                 file_put_contents( $css_file, $compiled_css );
463 
464         } catch ( exception $ex ) {
465             wp_die( __( 'Could not compile woocommerce.less:', 'woocommerce' ) . ' ' . $ex->getMessage() );
466         }
467     }
468 }

498 /**
499  * Process the new bulk actions for changing order status
500  *
501  * @access public
502  * @return void
503  */
504 function woocommerce_order_bulk_action() {
505     $wp_list_table = _get_list_table( 'WP_Posts_List_Table' );
506     $action = $wp_list_table->current_action();
507 
508     switch ( $action ) {
509         case 'mark_completed':
510             $new_status = 'completed';
511             $report_action = 'marked_completed';
512             break;
513         case 'mark_processing':
514             $new_status = 'processing';
515             $report_action = 'marked_processing';
516             break;
517         default:
518             return;
519     }
520 
521     $changed = 0;
522 
523     $post_ids = array_map( 'absint', (array) $_REQUEST['post'] );
524 
525     foreach( $post_ids as $post_id ) {
526         $order = new WC_Order( $post_id );
527         $order->update_status( $new_status, __( 'Order status changed by bulk edit:', 'woocommerce' ) );
528         $changed++;
529     }
530 
531     $sendback = add_query_arg( array( 'post_type' => 'shop_order', $report_action => $changed, 'ids' => join( ',', $post_ids ) ), '' );
532     wp_redirect( $sendback );
533     exit();
534 }
