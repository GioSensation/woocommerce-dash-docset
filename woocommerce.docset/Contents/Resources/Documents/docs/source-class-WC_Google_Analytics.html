 98     /**
 99      * Google Analytics standard tracking
100      *
101      * @access public
102      * @return void
103      */
104     function google_tracking_code() {
105         global $woocommerce;
106 
107         if ( is_admin() || current_user_can('manage_options') || $this->ga_standard_tracking_enabled == "no" ) return;
108 
109         $tracking_id = $this->ga_id;
110 
111         if ( ! $tracking_id ) return;
112 
113         $loggedin   = ( is_user_logged_in() ) ? 'yes' : 'no';
114         if ( is_user_logged_in() ) {
115             $user_id        = get_current_user_id();
116             $current_user   = get_user_by('id', $user_id);
117             $username       = $current_user->user_login;
118         } else {
119             $user_id        = '';
120             $username       = __( 'Guest', 'woocommerce' );
121         }
122 
123         if ( ! empty( $this->ga_set_domain_name ) )
124             $set_domain_name = "['_setDomainName', '" . esc_js( $this->ga_set_domain_name ) . "'],\n";
125         else
126             $set_domain_name = '';
127 
128         echo "<script type='text/javascript'>
129 
130             var _gaq = _gaq || [];
131             _gaq.push(
132                 ['_setAccount', '" . esc_js( $tracking_id ) . "'], " . $set_domain_name . "
133                 ['_setCustomVar', 1, 'logged-in', '" . $loggedin . "', 1],
134                 ['_setCustomVar', 2, 'user-id', '" . $user_id . "', 1],
135                 ['_setCustomVar', 3, 'username', '" . $username . "', 1],
136                 ['_trackPageview']
137             );
138 
139             (function() {
140                 var ga = document.createElement('script'); ga.type = 'text/javascript'; ga.async = true;
141                 ga.src = ('https:' == document.location.protocol ? 'https://ssl' : 'http://www') + '.google-analytics.com/ga.js';
142                 var s = document.getElementsByTagName('script')[0]; s.parentNode.insertBefore(ga, s);
143             })();
144 
145         </script>";
146     }

149     /**
150      * Google Analytics eCommerce tracking
151      *
152      * @access public
153      * @param mixed $order_id
154      * @return void
155      */
156     function ecommerce_tracking_code( $order_id ) {
157         global $woocommerce;
158 
159         if ( $this->ga_ecommerce_tracking_enabled == "no" || current_user_can('manage_options') || get_post_meta( $order_id, '_ga_tracked', true ) == 1 )
160             return;
161 
162         $tracking_id = $this->ga_id;
163 
164         if ( ! $tracking_id ) return;
165 
166         // Doing eCommerce tracking so unhook standard tracking from the footer
167         remove_action( 'wp_footer', array( $this, 'google_tracking_code' ) );
168 
169         // Get the order and output tracking code
170         $order = new WC_Order( $order_id );
171 
172         $loggedin = is_user_logged_in() ? 'yes' : 'no';
173 
174         if ( is_user_logged_in() ) {
175             $user_id        = get_current_user_id();
176             $current_user   = get_user_by('id', $user_id);
177             $username       = $current_user->user_login;
178         } else {
179             $user_id        = '';
180             $username       = __( 'Guest', 'woocommerce' );
181         }
182 
183         if ( ! empty( $this->ga_set_domain_name ) )
184             $set_domain_name = "['_setDomainName', '" . esc_js( $this->ga_set_domain_name ) . "'],";
185         else
186             $set_domain_name = '';
187 
188         $code = "
189             var _gaq = _gaq || [];
190 
191             _gaq.push(
192                 ['_setAccount', '" . esc_js( $tracking_id ) . "'], " . $set_domain_name . "
193                 ['_setCustomVar', 1, 'logged-in', '" . esc_js( $loggedin ) . "', 1],
194                 ['_setCustomVar', 2, 'user-id', '" . esc_js( $user_id ) . "', 1],
195                 ['_setCustomVar', 3, 'username', '" . esc_js( $username ) . "', 1],
196                 ['_trackPageview']
197             );
198 
199             _gaq.push(['_addTrans',
200                 '" . esc_js( $order_id ) . "',                  // order ID - required
201                 '" . esc_js( get_bloginfo( 'name' ) ) . "',     // affiliation or store name
202                 '" . esc_js( $order->get_total() ) . "',    // total - required
203                 '" . esc_js( $order->get_total_tax() ) . "',  // tax
204                 '" . esc_js( $order->get_shipping() ) . "', // shipping
205                 '" . esc_js( $order->billing_city ) . "',     // city
206                 '" . esc_js( $order->billing_state ) . "',    // state or province
207                 '" . esc_js( $order->billing_country ) . "'   // country
208             ]);
209         ";
210 
211         // Order items
212         if ( $order->get_items() ) {
213             foreach ( $order->get_items() as $item ) {
214                 $_product = $order->get_product_from_item( $item );
215 
216                 $code .= "_gaq.push(['_addItem',";
217                 $code .= "'" . esc_js( $order_id ) . "',";
218                 $code .= "'" . esc_js( $_product->get_sku() ? __( 'SKU:', 'woocommerce' ) . ' ' . $_product->get_sku() : $_product->id ) . "',";
219                 $code .= "'" . esc_js( $item['name'] ) . "',";
220 
221                 if ( isset( $_product->variation_data ) ) {
222 
223                     $code .= "'" . esc_js( woocommerce_get_formatted_variation( $_product->variation_data, true ) ) . "',";
224 
225                 } else {
226                     $out = array();
227                     $categories = get_the_terms($_product->id, 'product_cat');
228                     if ( $categories ) {
229                         foreach ( $categories as $category ){
230                             $out[] = $category->name;
231                         }
232                     }
233                     $code .= "'" . esc_js( join( "/", $out) ) . "',";
234                 }
235 
236                 $code .= "'" . esc_js( $order->get_item_total( $item ) ) . "',";
237                 $code .= "'" . esc_js( $item['qty'] ) . "'";
238                 $code .= "]);";
239             }
240         }
241 
242         $code .= "
243             _gaq.push(['_trackTrans']);                     // submits transaction to the Analytics servers
244 
245             (function() {
246                 var ga = document.createElement('script'); ga.type = 'text/javascript'; ga.async = true;
247                 ga.src = ('https:' == document.location.protocol ? 'https://ssl' : 'http://www') + '.google-analytics.com/ga.js';
248                 var s = document.getElementsByTagName('script')[0]; s.parentNode.insertBefore(ga, s);
249             })();
250         ";
251 
252         echo '<script type="text/javascript">' . $code . '</script>';
253 
254         update_post_meta( $order_id, '_ga_tracked', 1 );
255     }

260      *
261      * @access public
262      * @return void
263      */
264     function add_to_cart() {
265 
266         if ( $this->disable_tracking( $this->ga_event_tracking_enabled ) ) return;
267         if ( ! is_single() ) return;
268 
269         global $product;
270 
271         $parameters = array();
272         // Add single quotes to allow jQuery to be substituted into _trackEvent parameters
273         $parameters['category'] = "'" . __( 'Products', 'woocommerce' ) . "'";
274         $parameters['action'] = "'" . __( 'Add to cart', 'woocommerce' ) . "'";
275         $parameters['label'] = "'" . esc_js( $product->get_sku() ? __('SKU:', 'woocommerce') . ' ' . $product->get_sku() : "#" . $product->id ) . "'";
276 
277         $this->event_tracking_code( $parameters, '.single_add_to_cart_button' );
278     }
