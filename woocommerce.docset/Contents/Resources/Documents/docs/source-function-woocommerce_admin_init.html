105 /**
106  * Highlights the correct top level admin menu item for post type add screens.
107  *
108  * @access public
109  * @return void
110  */
111 function woocommerce_admin_menu_highlight() {
112     global $menu, $submenu, $parent_file, $submenu_file, $self, $post_type, $taxonomy;
113 
114     $to_highlight_types = array( 'shop_order', 'shop_coupon' );
115 
116     if ( isset( $post_type ) ) {
117         if ( in_array( $post_type, $to_highlight_types ) ) {
118             $submenu_file = 'edit.php?post_type=' . esc_attr( $post_type );
119             $parent_file  = 'woocommerce';
120         }
121 
122         if ( 'product' == $post_type ) {
123             $screen = get_current_screen();
124 
125             if ( $screen->base == 'edit-tags' && 'pa_' == substr( $taxonomy, 0, 3 ) ) {
126                 $submenu_file = 'woocommerce_attributes';
127                 $parent_file  = 'edit.php?post_type=' . esc_attr( $post_type );
128             }
129         }
130     }
131 
132     if ( isset( $submenu['woocommerce'] ) && isset( $submenu['woocommerce'][2] ) ) {
133         $submenu['woocommerce'][0] = $submenu['woocommerce'][2];
134         unset( $submenu['woocommerce'][2] );
135     }
136 
137     // Sort out Orders menu when on the top level
138     if ( ! current_user_can( 'manage_woocommerce' ) ) {
139         foreach ( $menu as $key => $menu_item ) {
140             if ( strpos( $menu_item[0], _x('Orders', 'Admin menu name', 'woocommerce') ) === 0 ) {
141 
142                 $menu_name = _x('Orders', 'Admin menu name', 'woocommerce');
143                 $menu_name_count = '';
144                 if ( $order_count = woocommerce_processing_order_count() ) {
145                     $menu_name_count = " <span class='awaiting-mod update-plugins count-$order_count'><span class='processing-count'>" . number_format_i18n( $order_count ) . "</span></span>" ;
146                 }
147 
148                 $menu[$key][0] = $menu_name . $menu_name_count;
149                 $submenu['edit.php?post_type=shop_order'][5][0] = $menu_name;
150                 break;
151             }
152         }
153     }
154 }

165 function woocommerce_admin_notices_styles() {
166 
167     if ( get_option( '_wc_needs_update' ) == 1 || get_option( '_wc_needs_pages' ) == 1 ) {
168         wp_enqueue_style( 'woocommerce-activation', plugins_url(  '/assets/css/activation.css', dirname( __FILE__ ) ) );
169         add_action( 'admin_notices', 'woocommerce_admin_install_notices' );
170     }
171 
172     $template = get_option( 'template' );
173 
174     if ( ! current_theme_supports( 'woocommerce' ) && ! in_array( $template, array( 'twentythirteen', 'twentyeleven', 'twentytwelve', 'twentyten' ) ) ) {
175 
176         if ( ! empty( $_GET['hide_woocommerce_theme_support_check'] ) ) {
177             update_option( 'woocommerce_theme_support_check', $template );
178             return;
179         }
180 
181         if ( get_option( 'woocommerce_theme_support_check' ) !== $template ) {
182             wp_enqueue_style( 'woocommerce-activation', plugins_url(  '/assets/css/activation.css', dirname( __FILE__ ) ) );
183             add_action( 'admin_notices', 'woocommerce_theme_check_notice' );
184         }
185 
186     }
187 
188 }

224 /**
225  * Include some admin files conditonally.
226  *
227  * @access public
228  * @return void
229  */
230 function woocommerce_admin_init() {
231     global $pagenow, $typenow;
232 
233     ob_start();
234 
235     // Install - Add pages button
236     if ( ! empty( $_GET['install_woocommerce_pages'] ) ) {
237 
238         require_once( 'woocommerce-admin-install.php' );
239         woocommerce_create_pages();
240 
241         // We no longer need to install pages
242         delete_option( '_wc_needs_pages' );
243         delete_transient( '_wc_activation_redirect' );
244 
245         // What's new redirect
246         wp_safe_redirect( admin_url( 'index.php?page=wc-about&wc-installed=true' ) );
247         exit;
248 
249     // Skip button
250     } elseif ( ! empty( $_GET['skip_install_woocommerce_pages'] ) ) {
251 
252         // We no longer need to install pages
253         delete_option( '_wc_needs_pages' );
254         delete_transient( '_wc_activation_redirect' );
255 
256         // Flush rules after install
257         flush_rewrite_rules();
258 
259         // What's new redirect
260         wp_safe_redirect( admin_url( 'index.php?page=wc-about' ) );
261         exit;
262 
263     // Update button
264     } elseif ( ! empty( $_GET['do_update_woocommerce'] ) ) {
265 
266         include_once( 'woocommerce-admin-update.php' );
267         do_update_woocommerce();
268 
269         // Update complete
270         delete_option( '_wc_needs_pages' );
271         delete_option( '_wc_needs_update' );
272         delete_transient( '_wc_activation_redirect' );
273 
274         // What's new redirect
275         wp_safe_redirect( admin_url( 'index.php?page=wc-about&wc-updated=true' ) );
276         exit;
277     }
278 
279     // Includes
280     if ( $typenow == 'post' && isset( $_GET['post'] ) && ! empty( $_GET['post'] ) ) {
281         $typenow = $post->post_type;
282     } elseif ( empty( $typenow ) && ! empty( $_GET['post'] ) ) {
283         $post = get_post( $_GET['post'] );
284         $typenow = $post->post_type;
285     }
286 
287     if ( $pagenow == 'index.php' ) {
288 
289         include_once( 'woocommerce-admin-dashboard.php' );
290 
291     } elseif ( $pagenow == 'admin.php' && isset( $_GET['import'] ) ) {
292 
293         include_once( 'woocommerce-admin-import.php' );
294 
295     } elseif ( $pagenow == 'post-new.php' || $pagenow == 'post.php' || $pagenow == 'edit.php' ) {
296 
297         include_once( 'post-types/writepanels/writepanels-init.php' );
298 
299         if ( in_array( $typenow, array( 'product', 'shop_coupon', 'shop_order' ) ) )
300             add_action('admin_print_styles', 'woocommerce_admin_help_tab');
301 
302     } elseif ( $pagenow == 'users.php' || $pagenow == 'user-edit.php' || $pagenow == 'profile.php' ) {
303 
304         include_once( 'woocommerce-admin-users.php' );
305 
306     }
307 
308     // Register importers
309     if ( defined( 'WP_LOAD_IMPORTERS' ) ) {
310         include_once( 'importers/importers-init.php' );
311     }
312 }

554 /**

556  *
557  * @access public
558  * @param mixed $menu_order
559  * @return void
560  */
561 function woocommerce_admin_menu_order( $menu_order ) {
562 
563     // Initialize our custom order array
564     $woocommerce_menu_order = array();
565 
566     // Get the index of our custom separator
567     $woocommerce_separator = array_search( 'separator-woocommerce', $menu_order );
568 
569     // Get index of product menu
570     $woocommerce_product = array_search( 'edit.php?post_type=product', $menu_order );
571 
572     // Loop through menu order and do some rearranging
573     foreach ( $menu_order as $index => $item ) :
574 
575         if ( ( ( 'woocommerce' ) == $item ) ) :
576             $woocommerce_menu_order[] = 'separator-woocommerce';
577             $woocommerce_menu_order[] = $item;
578             $woocommerce_menu_order[] = 'edit.php?post_type=product';
579             unset( $menu_order[$woocommerce_separator] );
580             unset( $menu_order[$woocommerce_product] );
581         elseif ( !in_array( $item, array( 'separator-woocommerce' ) ) ) :
582             $woocommerce_menu_order[] = $item;
583         endif;
584 
585     endforeach;
586 
587     // Return order
588     return $woocommerce_menu_order;
589 }

834  * @return void

836 function woocommerce_permalink_settings_save() {
837     if ( ! is_admin() )
