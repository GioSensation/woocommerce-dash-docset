 39 /**
 40  * Values for Columns on the Coupons admin page.
 41  *
 42  * @access public
 43  * @param mixed $column
 44  * @return void
 45  */
 46 function woocommerce_custom_coupon_columns( $column ) {
 47     global $post, $woocommerce;
 48 
 49     switch ( $column ) {
 50         case "coupon_code" :
 51             $edit_link = get_edit_post_link( $post->ID );
 52             $title = _draft_or_post_title();
 53             $post_type_object = get_post_type_object( $post->post_type );
 54             $can_edit_post = current_user_can( $post_type_object->cap->edit_post, $post->ID );
 55 
 56             echo '<div class="code tips" data-tip="' . __( 'Edit coupon', 'woocommerce' ) . '"><a href="' . esc_attr( $edit_link ) . '"><span>' . esc_html( $title ). '</span></a></div>';
 57 
 58             _post_states( $post );
 59 
 60             // Get actions
 61             $actions = array();
 62 
 63             if ( current_user_can( $post_type_object->cap->delete_post, $post->ID ) ) {
 64                 if ( 'trash' == $post->post_status )
 65                     $actions['untrash'] = "<a title='" . esc_attr( __( 'Restore this item from the Trash' ) ) . "' href='" . wp_nonce_url( admin_url( sprintf( $post_type_object->_edit_link . '&amp;action=untrash', $post->ID ) ), 'untrash-post_' . $post->ID ) . "'>" . __( 'Restore' ) . "</a>";
 66                 elseif ( EMPTY_TRASH_DAYS )
 67                     $actions['trash'] = "<a class='submitdelete' title='" . esc_attr( __( 'Move this item to the Trash' ) ) . "' href='" . get_delete_post_link( $post->ID ) . "'>" . __( 'Trash' ) . "</a>";
 68                 if ( 'trash' == $post->post_status || !EMPTY_TRASH_DAYS )
 69                     $actions['delete'] = "<a class='submitdelete' title='" . esc_attr( __( 'Delete this item permanently' ) ) . "' href='" . get_delete_post_link( $post->ID, '', true ) . "'>" . __( 'Delete Permanently' ) . "</a>";
 70             }
 71 
 72             $actions = apply_filters( 'post_row_actions', $actions, $post );
 73 
 74             echo '<div class="row-actions">';
 75 
 76             $i = 0;
 77             $action_count = sizeof($actions);
 78 
 79             foreach ( $actions as $action => $link ) {
 80                 ++$i;
 81                 ( $i == $action_count ) ? $sep = '' : $sep = ' | ';
 82                 echo "<span class='$action'>$link$sep</span>";
 83             }
 84             echo '</div>';
 85 
 86         break;
 87         case "type" :
 88             echo esc_html( $woocommerce->get_coupon_discount_type( get_post_meta( $post->ID, 'discount_type', true ) ) );
 89         break;
 90         case "amount" :
 91             echo esc_html( get_post_meta( $post->ID, 'coupon_amount', true ) );
 92         break;
 93         case "products" :
 94             $product_ids = get_post_meta( $post->ID, 'product_ids', true );
 95             $product_ids = $product_ids ? array_map( 'absint', explode( ',', $product_ids ) ) : array();
 96             if ( sizeof( $product_ids ) > 0 )
 97                 echo esc_html( implode( ', ', $product_ids ) );
 98             else
 99                 echo '&ndash;';
100         break;
101         case "usage_limit" :
102             $usage_limit = get_post_meta( $post->ID, 'usage_limit', true );
103 
104             if ( $usage_limit )
105                 echo esc_html( $usage_limit );
106             else
107                 echo '&ndash;';
108         break;
109         case "usage" :
110             $usage_count = absint( get_post_meta( $post->ID, 'usage_count', true ) );
111             $usage_limit = esc_html( get_post_meta($post->ID, 'usage_limit', true) );
112 
113             if ( $usage_limit )
114                 printf( __( '%s / %s', 'woocommerce' ), $usage_count, $usage_limit );
115             else
116                 printf( __( '%s / &infin;', 'woocommerce' ), $usage_count );
117         break;
118         case "expiry_date" :
119             $expiry_date = get_post_meta($post->ID, 'expiry_date', true);
120 
121             if ( $expiry_date )
122                 echo esc_html( date_i18n( 'F j, Y', strtotime( $expiry_date ) ) );
123             else
124                 echo '&ndash;';
125         break;
126         case "description" :
127             echo wp_kses_post( $post->post_excerpt );
128         break;
129     }
130 }

140 function woocommerce_restrict_manage_coupons() {

142

145 
146     // Type

149         <option value=""><?php _e( 'Show all types', 'woocommerce' ); ?></option>
150         <?php
151             $types = $woocommerce->get_coupon_discount_types();
152 
153             foreach ( $types as $name => $type ) {
154                 echo '<option value="' . esc_attr( $name ) . '"';
155 
156                 if ( isset( $_GET['coupon_type'] ) )

158 
159                 echo '>' . esc_html__( $type, 'woocommerce' ) . '</option>';
160             }
161         ?>
162         </select>

164

166         jQuery('select#dropdown_shop_coupon_type, select[name=m]').css('width', '150px').chosen();
