 15 /**
 16  * Runs the installer.
 17  *
 18  * @access public
 19  * @return void
 20  */
 21 function do_install_woocommerce() {
 22     global $woocommerce_settings, $woocommerce;
 23 
 24     // Do install
 25     woocommerce_default_options();
 26     woocommerce_tables_install();
 27     woocommerce_init_roles();
 28 
 29     // Register post types
 30     $woocommerce->init_taxonomy();
 31 
 32     // Add default taxonomies
 33     woocommerce_default_taxonomies();
 34 
 35     // Cron jobs
 36     wp_clear_scheduled_hook( 'woocommerce_scheduled_sales' );
 37     wp_clear_scheduled_hook( 'woocommerce_cancel_unpaid_orders' );
 38     wp_clear_scheduled_hook( 'woocommerce_cleanup_sessions' );
 39 
 40     $ve = get_option('gmt_offset') > 0 ? '+' : '-';
 41 
 42     wp_schedule_event( strtotime( 'tomorrow ' . $ve . get_option('gmt_offset') . ' HOURS' ), 'daily', 'woocommerce_scheduled_sales');
 43 
 44     $held_duration = get_option( 'woocommerce_hold_stock_minutes', null );
 45 
 46     if ( is_null( $held_duration ) )
 47         $held_duration = '60';
 48 
 49     if ( $held_duration != '' )
 50         wp_schedule_single_event( time() + ( absint( $held_duration ) * 60 ), 'woocommerce_cancel_unpaid_orders' );
 51 
 52     wp_schedule_event( time(), 'twicedaily', 'woocommerce_cleanup_sessions' );
 53 
 54     // Install files and folders for uploading files and prevent hotlinking
 55     $upload_dir =  wp_upload_dir();
 56 
 57     $files = array(
 58         array(
 59             'base'      => $upload_dir['basedir'] . '/woocommerce_uploads',
 60             'file'      => '.htaccess',
 61             'content'   => 'deny from all'
 62         ),
 63         array(
 64             'base'      => $upload_dir['basedir'] . '/woocommerce_uploads',
 65             'file'      => 'index.html',
 66             'content'   => ''
 67         ),
 68         array(
 69             'base'      => WP_PLUGIN_DIR . "/" . plugin_basename( dirname( dirname( __FILE__ ) ) ) . '/logs',
 70             'file'      => '.htaccess',
 71             'content'   => 'deny from all'
 72         ),
 73         array(
 74             'base'      => WP_PLUGIN_DIR . "/" . plugin_basename( dirname( dirname( __FILE__ ) ) ) . '/logs',
 75             'file'      => 'index.html',
 76             'content'   => ''
 77         )
 78     );
 79 
 80     foreach ( $files as $file ) {
 81         if ( wp_mkdir_p( $file['base'] ) && ! file_exists( trailingslashit( $file['base'] ) . $file['file'] ) ) {
 82             if ( $file_handle = @fopen( trailingslashit( $file['base'] ) . $file['file'], 'w' ) ) {
 83                 fwrite( $file_handle, $file['content'] );
 84                 fclose( $file_handle );
 85             }
 86         }
 87     }
 88 
 89     // Clear transient cache
 90     $woocommerce->clear_product_transients();
 91 
 92     // Recompile LESS styles if they are custom
 93     if ( get_option( 'woocommerce_frontend_css' ) == 'yes' ) {
 94 
 95         $colors = get_option( 'woocommerce_frontend_css_colors' );
 96 
 97         if ( ( ! empty( $colors['primary'] ) && ! empty( $colors['secondary'] ) && ! empty( $colors['highlight'] ) && ! empty( $colors['content_bg'] ) && ! empty( $colors['subtext'] ) ) && ( $colors['primary'] != '#ad74a2' || $colors['secondary'] != '#f7f6f7' || $colors['highlight'] != '#85ad74' || $colors['content_bg'] != '#ffffff' || $colors['subtext'] != '#777777' ) )
 98             woocommerce_compile_less_styles();
 99 
100     }
101 
102     // Queue upgrades
103     $current_version = get_option( 'woocommerce_version', null );
104     $current_db_version = get_option( 'woocommerce_db_version', null );
105 
106     if ( version_compare( $current_db_version, '2.0', '<' ) && null !== $current_db_version ) {
107         update_option( '_wc_needs_update', 1 );
108     } else {
109         update_option( 'woocommerce_db_version', $woocommerce->version );
110     }
111 
112     // Update version
113     update_option( 'woocommerce_version', $woocommerce->version );
114 
115     // Flush rewrite rules
116     flush_rewrite_rules();
117 }

144 /**
145  * Create a page
146  *
147  * @access public
148  * @param mixed $slug Slug for the new page
149  * @param mixed $option Option name to store the page's ID
150  * @param string $page_title (default: '') Title for the new page
151  * @param string $page_content (default: '') Content for the new page
152  * @param int $post_parent (default: 0) Parent for the new page
153  * @return void
154  */
155 function woocommerce_create_page( $slug, $option, $page_title = '', $page_content = '', $post_parent = 0 ) {
156     global $wpdb;
157 
158     $option_value = get_option( $option );
159 
160     if ( $option_value > 0 && get_post( $option_value ) )
161         return;
162 
163     $page_found = $wpdb->get_var( $wpdb->prepare( "SELECT ID FROM " . $wpdb->posts . " WHERE post_name = %s LIMIT 1;", $slug ) );
164     if ( $page_found ) {
165         if ( ! $option_value )
166             update_option( $option, $page_found );
167         return;
168     }
169 
170     $page_data = array(
171         'post_status'       => 'publish',
172         'post_type'         => 'page',
173         'post_author'       => 1,
174         'post_name'         => $slug,
175         'post_title'        => $page_title,
176         'post_content'      => $page_content,
177         'post_parent'       => $post_parent,
178         'comment_status'    => 'closed'
179     );
180     $page_id = wp_insert_post( $page_data );
181 
182     update_option( $option, $page_id );
183 }

229 /**
230  * Set up the database tables which the plugin needs to function.
231  *
232  * Tables:
233  *      woocommerce_attribute_taxonomies - Table for storing attribute taxonomies - these are user defined
234  *      woocommerce_termmeta - Term meta table - sadly WordPress does not have termmeta so we need our own
235  *      woocommerce_downloadable_product_permissions - Table for storing user and guest download permissions.
236  *          KEY(order_id, product_id, download_id) used for organizing downloads on the My Account page
237  *      woocommerce_order_items - Order line items are stored in a table to make them easily queryable for reports
238  *      woocommerce_order_itemmeta - Order line item meta is stored in a table for storing extra data.
239  *      woocommerce_tax_rates - Tax Rates are stored inside 2 tables making tax queries simple and efficient.
240  *      woocommerce_tax_rate_locations - Each rate can be applied to more than one postcode/city hence the second table.
241  *
242  * @access public
243  * @return void
244  */
245 function woocommerce_tables_install() {
246     global $wpdb, $woocommerce;
247 
248     $wpdb->hide_errors();
249 
250     $collate = '';
251 
252     if ( $wpdb->has_cap( 'collation' ) ) {
253         if( ! empty($wpdb->charset ) )
254             $collate .= "DEFAULT CHARACTER SET $wpdb->charset";
255         if( ! empty($wpdb->collate ) )
256             $collate .= " COLLATE $wpdb->collate";
257     }
258 
259     require_once( ABSPATH . 'wp-admin/includes/upgrade.php' );
260 
261     // WooCommerce Tables
262     $woocommerce_tables = "
263 CREATE TABLE {$wpdb->prefix}woocommerce_attribute_taxonomies (
264   attribute_id bigint(20) NOT NULL auto_increment,
265   attribute_name varchar(200) NOT NULL,
266   attribute_label longtext NULL,
267   attribute_type varchar(200) NOT NULL,
268   attribute_orderby varchar(200) NOT NULL,
269   PRIMARY KEY  (attribute_id),
270   KEY attribute_name (attribute_name)
271 ) $collate;
272 CREATE TABLE {$wpdb->prefix}woocommerce_termmeta (
273   meta_id bigint(20) NOT NULL auto_increment,
274   woocommerce_term_id bigint(20) NOT NULL,
275   meta_key varchar(255) NULL,
276   meta_value longtext NULL,
277   PRIMARY KEY  (meta_id),
278   KEY woocommerce_term_id (woocommerce_term_id),
279   KEY meta_key (meta_key)
280 ) $collate;
281 CREATE TABLE {$wpdb->prefix}woocommerce_downloadable_product_permissions (
282   download_id varchar(32) NOT NULL,
283   product_id bigint(20) NOT NULL,
284   order_id bigint(20) NOT NULL DEFAULT 0,
285   order_key varchar(200) NOT NULL,
286   user_email varchar(200) NOT NULL,
287   user_id bigint(20) NULL,
288   downloads_remaining varchar(9) NULL,
289   access_granted datetime NOT NULL default '0000-00-00 00:00:00',
290   access_expires datetime NULL default null,
291   download_count bigint(20) NOT NULL DEFAULT 0,
292   PRIMARY KEY  (product_id,order_id,order_key,download_id),
293   KEY download_order_product (download_id,order_id,product_id)
294 ) $collate;
295 CREATE TABLE {$wpdb->prefix}woocommerce_order_items (
296   order_item_id bigint(20) NOT NULL auto_increment,
297   order_item_name longtext NOT NULL,
298   order_item_type varchar(200) NOT NULL DEFAULT '',
299   order_id bigint(20) NOT NULL,
300   PRIMARY KEY  (order_item_id),
301   KEY order_id (order_id)
302 ) $collate;
303 CREATE TABLE {$wpdb->prefix}woocommerce_order_itemmeta (
304   meta_id bigint(20) NOT NULL auto_increment,
305   order_item_id bigint(20) NOT NULL,
306   meta_key varchar(255) NULL,
307   meta_value longtext NULL,
308   PRIMARY KEY  (meta_id),
309   KEY order_item_id (order_item_id),
310   KEY meta_key (meta_key)
311 ) $collate;
312 CREATE TABLE {$wpdb->prefix}woocommerce_tax_rates (
313   tax_rate_id bigint(20) NOT NULL auto_increment,
314   tax_rate_country varchar(200) NOT NULL DEFAULT '',
315   tax_rate_state varchar(200) NOT NULL DEFAULT '',
316   tax_rate varchar(200) NOT NULL DEFAULT '',
317   tax_rate_name varchar(200) NOT NULL DEFAULT '',
318   tax_rate_priority bigint(20) NOT NULL,
319   tax_rate_compound int(1) NOT NULL DEFAULT 0,
320   tax_rate_shipping int(1) NOT NULL DEFAULT 1,
321   tax_rate_order bigint(20) NOT NULL,
322   tax_rate_class varchar(200) NOT NULL DEFAULT '',
323   PRIMARY KEY  (tax_rate_id),
324   KEY tax_rate_country (tax_rate_country),
325   KEY tax_rate_state (tax_rate_state),
326   KEY tax_rate_class (tax_rate_class),
327   KEY tax_rate_priority (tax_rate_priority)
328 ) $collate;
329 CREATE TABLE {$wpdb->prefix}woocommerce_tax_rate_locations (
330   location_id bigint(20) NOT NULL auto_increment,
331   location_code varchar(255) NOT NULL,
332   tax_rate_id bigint(20) NOT NULL,
333   location_type varchar(40) NOT NULL,
334   PRIMARY KEY  (location_id),
335   KEY location_type (location_type),
336   KEY location_type_code (location_type,location_code)
337 ) $collate;
338 ";
339     dbDelta( $woocommerce_tables );
340 }

343 /**
344  * Add the default terms for WC taxonomies - product types and order statuses. Modify this at your own risk.
345  *
346  * @access public
347  * @return void
348  */
349 function woocommerce_default_taxonomies() {
350 
351     $taxonomies = array(
352         'product_type' => array(
353             'simple',
354             'grouped',
355             'variable',
356             'external'
357         ),
358         'shop_order_status' => array(
359             'pending',
360             'failed',
361             'on-hold',
362             'processing',
363             'completed',
364             'refunded',
365             'cancelled'
366         )
367     );
368 
369     foreach ( $taxonomies as $taxonomy => $terms ) {
370         foreach ( $terms as $term ) {
371             if ( ! get_term_by( 'slug', sanitize_title( $term ), $taxonomy ) ) {
372                 wp_insert_term( $term, $taxonomy );
373             }
374         }
375     }
376 }
