 307  *

 309  *
 310  * @access public
 311  * @param mixed $vars
 312  * @return array
 313  */
 314 function woocommerce_custom_product_orderby( $vars ) {
 315     if (isset( $vars['orderby'] )) :
 316         if ( 'price' == $vars['orderby'] ) :
 317             $vars = array_merge( $vars, array(
 318                 'meta_key'  => '_price',
 319                 'orderby'   => 'meta_value_num'
 320             ) );
 321         endif;
 322         if ( 'featured' == $vars['orderby'] ) :
 323             $vars = array_merge( $vars, array(
 324                 'meta_key'  => '_featured',
 325                 'orderby'   => 'meta_value'

 327         endif;
 328         if ( 'sku' == $vars['orderby'] ) :
 329             $vars = array_merge( $vars, array(
 330                 'meta_key'  => '_sku',
 331                 'orderby'   => 'meta_value'
 332             ) );
 333         endif;
 334     endif;
 335 
 336     return $vars;

 425 function woocommerce_admin_product_filter_query( $query ) {

 448                 'operator' => 'NOT IN'

 460 /**
 461  * Search by SKU or ID for products. Adapted from code by BenIrvin (Admin Search by ID)
 462  *
 463  * @access public
 464  * @param mixed $wp
 465  * @return void
 466  */
 467 function woocommerce_admin_product_search( $wp ) {
 468     global $pagenow, $wpdb;
 469 
 470     if( 'edit.php' != $pagenow ) return;
 471     if( !isset( $wp->query_vars['s'] ) ) return;
 472     if ($wp->query_vars['post_type']!='product') return;
 473 
 474     if( '#' == substr( $wp->query_vars['s'], 0, 1 ) ) :
 475 
 476         $id = absint( substr( $wp->query_vars['s'], 1 ) );
 477 
 478         if( !$id ) return;
 479 
 480         unset( $wp->query_vars['s'] );
 481         $wp->query_vars['p'] = $id;
 482 
 483     elseif( 'SKU:' == substr( $wp->query_vars['s'], 0, 4 ) ) :
 484 
 485         $sku = trim( substr( $wp->query_vars['s'], 4 ) );
 486 
 487         if( !$sku ) return;
 488 
 489         $ids = $wpdb->get_col( 'SELECT post_id FROM ' . $wpdb->postmeta . ' WHERE meta_key="_sku" AND meta_value LIKE "%' . $sku . '%";' );
 490 
 491         if ( ! $ids ) return;
 492 
 493         unset( $wp->query_vars['s'] );
 494         $wp->query_vars['post__in'] = $ids;
 495         $wp->query_vars['sku'] = $sku;
 496 
 497     endif;
 498 }

 701  * @param mixed $post
 702  * @return void
 703  */
 704 function woocommerce_admin_product_quick_edit_save( $post_id, $post ) {
 705 
 706     if ( ! $_POST || is_int( wp_is_post_revision( $post_id ) ) || is_int( wp_is_post_autosave( $post_id ) ) ) return $post_id;
 707     if ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) return $post_id;
 708     if ( ! isset( $_POST['woocommerce_quick_edit_nonce'] ) || ! wp_verify_nonce( $_POST['woocommerce_quick_edit_nonce'], 'woocommerce_quick_edit_nonce' ) ) return $post_id;
 709     if ( ! current_user_can( 'edit_post', $post_id ) ) return $post_id;
 710     if ( $post->post_type != 'product' ) return $post_id;
 711 
 712     global $woocommerce, $wpdb;
 713 
 714     $product           = get_product( $post );
 715     $old_regular_price = $product->regular_price;
 716     $old_sale_price    = $product->sale_price;
 717 
 718     // Save fields
 719     if ( isset( $_POST['_sku'] ) ) update_post_meta( $post_id, '_sku', woocommerce_clean( $_POST['_sku'] ) );
 720     if ( isset( $_POST['_weight'] ) ) update_post_meta( $post_id, '_weight', woocommerce_clean( $_POST['_weight'] ) );
 721     if ( isset( $_POST['_length'] ) ) update_post_meta( $post_id, '_length', woocommerce_clean( $_POST['_length'] ) );
 722     if ( isset( $_POST['_width'] ) ) update_post_meta( $post_id, '_width', woocommerce_clean( $_POST['_width'] ) );
 723     if ( isset( $_POST['_height'] ) ) update_post_meta( $post_id, '_height', woocommerce_clean( $_POST['_height'] ) );
 724     if ( isset( $_POST['_stock_status'] ) ) update_post_meta( $post_id, '_stock_status', woocommerce_clean( $_POST['_stock_status'] ) );
 725     if ( isset( $_POST['_visibility'] ) ) update_post_meta( $post_id, '_visibility', woocommerce_clean( $_POST['_visibility'] ) );
 726     if ( isset( $_POST['_featured'] ) ) update_post_meta( $post_id, '_featured', 'yes' ); else update_post_meta( $post_id, '_featured', 'no' );
 727 
 728     if ( $product->is_type('simple') || $product->is_type('external') ) {
 729 
 730         if ( isset( $_POST['_regular_price'] ) ) update_post_meta( $post_id, '_regular_price', woocommerce_clean( $_POST['_regular_price'] ) );
 731         if ( isset( $_POST['_sale_price'] ) ) update_post_meta( $post_id, '_sale_price', woocommerce_clean( $_POST['_sale_price'] ) );
 732 
 733         // Handle price - remove dates and set to lowest
 734         $price_changed = false;
 735 
 736         if ( isset( $_POST['_regular_price'] ) && woocommerce_clean( $_POST['_regular_price'] ) != $old_regular_price ) $price_changed = true;
 737         if ( isset( $_POST['_sale_price'] ) && woocommerce_clean( $_POST['_sale_price'] ) != $old_sale_price ) $price_changed = true;
 738 
 739         if ( $price_changed ) {
 740             update_post_meta( $post_id, '_sale_price_dates_from', '' );
 741             update_post_meta( $post_id, '_sale_price_dates_to', '' );
 742 
 743             if ( isset( $_POST['_sale_price'] ) && $_POST['_sale_price'] != '' ) {
 744                 update_post_meta( $post_id, '_price', woocommerce_clean( $_POST['_sale_price'] ) );
 745             } else {
 746                 update_post_meta( $post_id, '_price', woocommerce_clean( $_POST['_regular_price'] ) );
 747             }
 748         }
 749     }
 750 
 751     // Handle stock
 752     if ( ! $product->is_type('grouped') ) {
 753         if ( isset( $_POST['_manage_stock'] ) ) {
 754             update_post_meta( $post_id, '_manage_stock', 'yes' );
 755             update_post_meta( $post_id, '_stock', (int) $_POST['_stock'] );
 756         } else {
 757             update_post_meta( $post_id, '_manage_stock', 'no' );
 758             update_post_meta( $post_id, '_stock', '0' );
 759         }
 760     }
 761 
 762     // Clear transient
 763     $woocommerce->clear_product_transients( $post_id );
 764 }
