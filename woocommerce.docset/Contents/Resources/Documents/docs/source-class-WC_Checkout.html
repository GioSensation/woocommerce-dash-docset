134     /**
135      * create_order function.
136      *
137      * @access public
138      * @return void
139      */
140     public function create_order() {
141         global $woocommerce, $wpdb;
142 
143         // Give plugins the opportunity to create an order themselves
144         $order_id = apply_filters( 'woocommerce_create_order', null, $this );
145 
146         if ( is_numeric( $order_id ) )
147             return $order_id;
148 
149         // Create Order (send cart variable so we can record items and reduce inventory). Only create if this is a new order, not if the payment was rejected.
150         $order_data = apply_filters( 'woocommerce_new_order_data', array(
151             'post_type'     => 'shop_order',
152             'post_title'    => sprintf( __( 'Order &ndash; %s', 'woocommerce' ), strftime( _x( '%b %d, %Y @ %I:%M %p', 'Order date parsed by strftime', 'woocommerce' ) ) ),
153             'post_status'   => 'publish',
154             'ping_status'   => 'closed',
155             'post_excerpt'  => isset( $this->posted['order_comments'] ) ? $this->posted['order_comments'] : '',
156             'post_author'   => 1,
157             'post_password' => uniqid( 'order_' )   // Protects the post just in case
158         ) );
159 
160         // Insert or update the post data
161         $create_new_order = true;
162 
163         if ( $woocommerce->session->order_awaiting_payment > 0 ) {
164 
165             $order_id = absint( $woocommerce->session->order_awaiting_payment );
166 
167             /* Check order is unpaid by getting its status */
168             $terms = wp_get_object_terms( $order_id, 'shop_order_status', array( 'fields' => 'slugs' ) );
169             $order_status = isset( $terms[0] ) ? $terms[0] : 'pending';
170 
171             // Resume the unpaid order if its pending
172             if ( $order_status == 'pending' || $order_status == 'failed' ) {
173 
174                 // Update the existing order as we are resuming it
175                 $create_new_order = false;
176                 $order_data['ID'] = $order_id;
177                 wp_update_post( $order_data );
178 
179                 // Clear the old line items - we'll add these again in case they changed
180                 $wpdb->query( $wpdb->prepare( "DELETE FROM {$wpdb->prefix}woocommerce_order_itemmeta WHERE order_item_id IN ( SELECT order_item_id FROM {$wpdb->prefix}woocommerce_order_items WHERE order_id = %d )", $order_id ) );
181 
182                 $wpdb->query( $wpdb->prepare( "DELETE FROM {$wpdb->prefix}woocommerce_order_items WHERE order_id = %d", $order_id ) );
183 
184                 // Trigger an action for the resumed order
185                 do_action( 'woocommerce_resume_order', $order_id );
186             }
187         }
188 
189         if ( $create_new_order ) {
190             $order_id = wp_insert_post( $order_data );
191 
192             if ( is_wp_error( $order_id ) )
193                 throw new MyException( 'Error: Unable to create order. Please try again.' );
194             else
195                 do_action( 'woocommerce_new_order', $order_id );
196         }
197 
198         // Store user data
199         if ( $this->checkout_fields['billing'] ) {
200             foreach ( $this->checkout_fields['billing'] as $key => $field ) {
201 
202                 update_post_meta( $order_id, '_' . $key, $this->posted[ $key ] );
203 
204                 // User
205                 if ( $this->customer_id && ! empty( $this->posted[ $key ] ) ) {
206                     update_user_meta( $this->customer_id, $key, $this->posted[ $key ] );
207 
208                     // Special fields
209                     switch ( $key ) {
210                         case "billing_email" :
211                             if ( ! email_exists( $this->posted[ $key ] ) )
212                                 wp_update_user( array ( 'ID' => $this->customer_id, 'user_email' => $this->posted[ $key ] ) ) ;
213                         break;
214                         case "billing_first_name" :
215                             wp_update_user( array ( 'ID' => $this->customer_id, 'first_name' => $this->posted[ $key ] ) ) ;
216                         break;
217                         case "billing_last_name" :
218                             wp_update_user( array ( 'ID' => $this->customer_id, 'last_name' => $this->posted[ $key ] ) ) ;
219                         break;
220                     }
221                 }
222             }
223         }
224 
225         if ( $this->checkout_fields['shipping'] && ( $woocommerce->cart->needs_shipping() || get_option('woocommerce_require_shipping_address') == 'yes' ) ) {
226             foreach ( $this->checkout_fields['shipping'] as $key => $field ) {
227                 $postvalue = false;
228 
229                 if ( $this->posted['shiptobilling'] ) {
230                     if ( isset( $this->posted[ str_replace( 'shipping_', 'billing_', $key ) ] ) ) {
231                         $postvalue = $this->posted[ str_replace( 'shipping_', 'billing_', $key ) ];
232                         update_post_meta( $order_id, '_' . $key, $postvalue );
233                     }
234                 } else {
235                     $postvalue = $this->posted[ $key ];
236                     update_post_meta( $order_id, '_' . $key, $postvalue );
237                 }
238 
239                 // User
240                 if ( $postvalue && $this->customer_id )
241                     update_user_meta( $this->customer_id, $key, $postvalue );
242             }
243         }
244 
245         // Save any other user meta
246         if ( $this->customer_id )
247             do_action( 'woocommerce_checkout_update_user_meta', $this->customer_id, $this->posted );
248 
249         // Store the line items to the new/resumed order
250         foreach ( $woocommerce->cart->get_cart() as $cart_item_key => $values ) {
251 
252             $_product = $values['data'];
253 
254             // Add line item
255             $item_id = woocommerce_add_order_item( $order_id, array(
256                 'order_item_name'       => $_product->get_title(),
257                 'order_item_type'       => 'line_item'
258             ) );
259 
260             // Add line item meta
261             if ( $item_id ) {
262                 woocommerce_add_order_item_meta( $item_id, '_qty', apply_filters( 'woocommerce_stock_amount', $values['quantity'] ) );
263                 woocommerce_add_order_item_meta( $item_id, '_tax_class', $_product->get_tax_class() );
264                 woocommerce_add_order_item_meta( $item_id, '_product_id', $values['product_id'] );
265                 woocommerce_add_order_item_meta( $item_id, '_variation_id', $values['variation_id'] );
266                 woocommerce_add_order_item_meta( $item_id, '_line_subtotal', woocommerce_format_decimal( $values['line_subtotal'], 4 ) );
267                 woocommerce_add_order_item_meta( $item_id, '_line_total', woocommerce_format_decimal( $values['line_total'], 4 ) );
268                 woocommerce_add_order_item_meta( $item_id, '_line_tax', woocommerce_format_decimal( $values['line_tax'], 4 ) );
269                 woocommerce_add_order_item_meta( $item_id, '_line_subtotal_tax', woocommerce_format_decimal( $values['line_subtotal_tax'], 4 ) );
270 
271                 // Store variation data in meta so admin can view it
272                 if ( $values['variation'] && is_array( $values['variation'] ) )
273                     foreach ( $values['variation'] as $key => $value )
274                         woocommerce_add_order_item_meta( $item_id, esc_attr( str_replace( 'attribute_', '', $key ) ), $value );
275 
276                 // Add line item meta for backorder status
277                 if ( $_product->backorders_require_notification() && $_product->is_on_backorder( $values['quantity'] ) )
278                     woocommerce_add_order_item_meta( $item_id, apply_filters( 'woocommerce_backordered_item_meta_name', __( 'Backordered', 'woocommerce' ), $cart_item_key, $order_id ), $values['quantity'] - max( 0, $_product->get_total_stock() ) );
279 
280                 //allow plugins to add order item meta
281                 do_action( 'woocommerce_add_order_item_meta', $item_id, $values );
282             }
283         }
284 
285         // Store fees
286         foreach ( $woocommerce->cart->get_fees() as $fee ) {
287             $item_id = woocommerce_add_order_item( $order_id, array(
288                 'order_item_name'       => $fee->name,
289                 'order_item_type'       => 'fee'
290             ) );
291 
292             if ( $fee->taxable )
293                 woocommerce_add_order_item_meta( $item_id, '_tax_class', $fee->tax_class );
294             else
295                 woocommerce_add_order_item_meta( $item_id, '_tax_class', '0' );
296 
297             woocommerce_add_order_item_meta( $item_id, '_line_total', woocommerce_format_decimal( $fee->amount ) );
298             woocommerce_add_order_item_meta( $item_id, '_line_tax', woocommerce_format_decimal( $fee->tax ) );
299         }
300 
301         // Store tax rows
302         foreach ( array_keys( $woocommerce->cart->taxes + $woocommerce->cart->shipping_taxes ) as $key ) {
303 
304             $item_id = woocommerce_add_order_item( $order_id, array(
305                 'order_item_name'       => $woocommerce->cart->tax->get_rate_code( $key ),
306                 'order_item_type'       => 'tax'
307             ) );
308 
309             // Add line item meta
310             if ( $item_id ) {
311                 woocommerce_add_order_item_meta( $item_id, 'rate_id', $key );
312                 woocommerce_add_order_item_meta( $item_id, 'label', $woocommerce->cart->tax->get_rate_label( $key ) );
313                 woocommerce_add_order_item_meta( $item_id, 'compound', absint( $woocommerce->cart->tax->is_compound( $key ) ? 1 : 0 ) );
314                 woocommerce_add_order_item_meta( $item_id, 'tax_amount', woocommerce_clean( isset( $woocommerce->cart->taxes[ $key ] ) ? $woocommerce->cart->taxes[ $key ] : 0 ) );
315                 woocommerce_add_order_item_meta( $item_id, 'shipping_tax_amount', woocommerce_clean( isset( $woocommerce->cart->shipping_taxes[ $key ] ) ? $woocommerce->cart->shipping_taxes[ $key ] : 0 ) );
316             }
317         }
318 
319         // Store coupons
320         if ( $applied_coupons = $woocommerce->cart->get_applied_coupons() ) {
321             foreach ( $applied_coupons as $code ) {
322 
323                 $item_id = woocommerce_add_order_item( $order_id, array(
324                     'order_item_name'       => $code,
325                     'order_item_type'       => 'coupon'
326                 ) );
327 
328                 // Add line item meta
329                 if ( $item_id ) {
330                     woocommerce_add_order_item_meta( $item_id, 'discount_amount', isset( $woocommerce->cart->coupon_discount_amounts[ $code ] ) ? $woocommerce->cart->coupon_discount_amounts[ $code ] : 0 );
331                 }
332             }
333         }
334 
335         // Store meta
336         if ( $this->shipping_method ) {
337             update_post_meta( $order_id, '_shipping_method',        $this->shipping_method->id );
338             update_post_meta( $order_id, '_shipping_method_title',  $this->shipping_method->label );
339         }
340 
341         if ( $this->payment_method ) {
342             update_post_meta( $order_id, '_payment_method',         $this->payment_method->id );
343             update_post_meta( $order_id, '_payment_method_title',   $this->payment_method->get_title() );
344         }
345 
346         update_post_meta( $order_id, '_order_shipping',         woocommerce_format_total( $woocommerce->cart->shipping_total ) );
347         update_post_meta( $order_id, '_order_discount',         woocommerce_format_total( $woocommerce->cart->get_order_discount_total() ) );
348         update_post_meta( $order_id, '_cart_discount',          woocommerce_format_total( $woocommerce->cart->get_cart_discount_total() ) );
349         update_post_meta( $order_id, '_order_tax',              woocommerce_clean( $woocommerce->cart->tax_total ) );
350         update_post_meta( $order_id, '_order_shipping_tax',     woocommerce_clean( $woocommerce->cart->shipping_tax_total ) );
351         update_post_meta( $order_id, '_order_total',            woocommerce_format_total( $woocommerce->cart->total ) );
352         update_post_meta( $order_id, '_order_key',              apply_filters('woocommerce_generate_order_key', uniqid('order_') ) );
353         update_post_meta( $order_id, '_customer_user',          absint( $this->customer_id ) );
354         update_post_meta( $order_id, '_order_currency',         get_woocommerce_currency() );
355         update_post_meta( $order_id, '_prices_include_tax',     get_option( 'woocommerce_prices_include_tax' ) );
356         update_post_meta( $order_id, '_customer_ip_address',    isset( $_SERVER['HTTP_X_FORWARDED_FOR'] ) ? $_SERVER['HTTP_X_FORWARDED_FOR'] : $_SERVER['REMOTE_ADDR'] );
357         update_post_meta( $order_id, '_customer_user_agent',    isset( $_SERVER['HTTP_USER_AGENT'] ) ? $_SERVER['HTTP_USER_AGENT'] : '' );
358 
359         // Let plugins add meta
360         do_action( 'woocommerce_checkout_update_order_meta', $order_id, $this->posted );
361 
362         // Order status
363         wp_set_object_terms( $order_id, 'pending', 'shop_order_status' );
364 
365         return $order_id;
366     }

368     /**
369      * Process the checkout after the confirm order button is pressed
370      *
371      * @access public
372      * @return void
373      */
374     public function process_checkout() {
375         global $wpdb, $woocommerce, $current_user;
376 
377         $woocommerce->verify_nonce( 'process_checkout' );
378 
379         if ( ! defined( 'WOOCOMMERCE_CHECKOUT' ) )
380             define( 'WOOCOMMERCE_CHECKOUT', true );
381 
382         // Prevent timeout
383         @set_time_limit(0);
384 
385         do_action( 'woocommerce_before_checkout_process' );
386 
387         if ( sizeof( $woocommerce->cart->get_cart() ) == 0 )
388             $woocommerce->add_error( sprintf( __( 'Sorry, your session has expired. <a href="%s">Return to homepage &rarr;</a>', 'woocommerce' ), home_url() ) );
389 
390         do_action( 'woocommerce_checkout_process' );
391 
392         // Checkout fields (not defined in checkout_fields)
393         $this->posted['shiptobilling']      = isset( $_POST['shiptobilling'] ) ? 1 : 0;
394         $this->posted['terms']              = isset( $_POST['terms'] ) ? 1 : 0;
395         $this->posted['createaccount']      = isset( $_POST['createaccount'] ) ? 1 : 0;
396         $this->posted['payment_method']     = isset( $_POST['payment_method'] ) ? woocommerce_clean( $_POST['payment_method'] ) : '';
397         $this->posted['shipping_method']    = isset( $_POST['shipping_method'] ) ? woocommerce_clean( $_POST['shipping_method'] ) : '';
398 
399         // Ship to billing only option
400         if ( $woocommerce->cart->ship_to_billing_address_only() )
401             $this->posted['shiptobilling'] = 1;
402 
403         // Update customer shipping and payment method to posted method
404         $woocommerce->session->chosen_shipping_method   = $this->posted['shipping_method'];
405         $woocommerce->session->chosen_payment_method    = $this->posted['payment_method'];
406 
407         // Note if we skip shipping
408         $skipped_shipping = false;
409 
410         // Get validation class
411         $validation = $woocommerce->validation();
412 
413         // Get posted checkout_fields and do validation
414         foreach ( $this->checkout_fields as $fieldset_key => $fieldset ) {
415 
416             // Skip shipping if its not needed
417             if ( $fieldset_key == 'shipping' && ( $woocommerce->cart->ship_to_billing_address_only() || $this->posted['shiptobilling'] || ( ! $woocommerce->cart->needs_shipping() && get_option('woocommerce_require_shipping_address') == 'no' ) ) ) {
418                 $skipped_shipping = true;
419                 continue;
420             }
421 
422             foreach ( $fieldset as $key => $field ) {
423 
424                 if ( ! isset( $field['type'] ) ) $field['type'] = 'text';
425 
426                 // Get Value
427                 switch ( $field['type'] ) {
428                     case "checkbox" :
429                         $this->posted[ $key ] = isset( $_POST[$key] ) ? 1 : 0;
430                     break;
431                     default :
432                         $this->posted[ $key ] = isset( $_POST[$key] ) ? woocommerce_clean( $_POST[ $key ] ) : '';
433                     break;
434                 }
435 
436                 // Hook to allow modification of value
437                 $this->posted[ $key ] = apply_filters( 'woocommerce_process_checkout_field_' . $key, $this->posted[$key] );
438 
439                 // Validation: Required fields
440                 if ( isset( $field['required'] ) && $field['required'] && empty( $this->posted[$key] ) ) $woocommerce->add_error( '<strong>' . $field['label'] . '</strong> ' . __( 'is a required field.', 'woocommerce' ) );
441 
442                 if ( ! empty( $this->posted[ $key ] ) ) {
443 
444                     // Special handling for validation and formatting
445                     switch ( $key ) {
446                         case "billing_postcode" :
447                         case "shipping_postcode" :
448 
449                             $validate_against = $key == 'billing_postcode' ? 'billing_country' : 'shipping_country';
450                             $this->posted[ $key ] = strtoupper( str_replace( ' ', '', $this->posted[ $key ] ) );
451 
452                             if ( ! $validation->is_postcode( $this->posted[ $key ], $_POST[ $validate_against ] ) )
453                                 $woocommerce->add_error( '<strong>' . $field['label'] . '</strong> ' . sprintf( __( '(%s) is not a valid postcode/ZIP.', 'woocommerce' ), $this->posted[ $key ] ) );
454                             else
455                                 $this->posted[ $key ] = $validation->format_postcode( $this->posted[$key], $_POST[ $validate_against ] );
456 
457                         break;
458                         case "billing_state" :
459                         case "shipping_state" :
460 
461                             // Get valid states
462                             $validate_against = $key == 'billing_state' ? 'billing_country' : 'shipping_country';
463                             $valid_states = $woocommerce->countries->get_states( $_POST[ $validate_against ] );
464                             if ( $valid_states )
465                                 $valid_state_values = array_flip( array_map( 'strtolower', $valid_states ) );
466 
467                             // Convert value to key if set
468                             if ( isset( $valid_state_values[ strtolower( $this->posted[ $key ] ) ] ) )
469                                  $this->posted[ $key ] = $valid_state_values[ strtolower( $this->posted[ $key ] ) ];
470 
471                             // Only validate if the country has specific state options
