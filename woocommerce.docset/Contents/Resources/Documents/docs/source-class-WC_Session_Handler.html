  5 /**
  6  * Handle data for the current customers session.
  7  * Implements the WC_Session abstract class
  8  *
  9  * Long term plan will be, if https://github.com/ericmann/wp-session-manager/ gains traction
 10  * in WP core, this will be switched out to use it and maintain backwards compatibility :)
 11  *
 12  * Partly based on WP SESSION by Eric Mann.
 13  *
 14  * @class       WC_Session_Handler
 15  * @version     2.0.0
 16  * @package     WooCommerce/Classes
 17  * @category    Class
 18  * @author      WooThemes
 19  */
 20 class WC_Session_Handler extends WC_Session {
 21 
 22     /** cookie name */
 23     private $_cookie;
 24 
 25     /** session due to expire timestamp */
 26     private $_session_expiring;
 27 
 28     /** session expiration timestamp */
 29     private $_session_expiration;
 30 
 31     /**
 32      * Constructor for the session class.
 33      *
 34      * @access public
 35      * @return void
 36      */
 37     public function __construct() {
 38         $this->_cookie = 'wc_session_cookie_' . COOKIEHASH;
 39 
 40         if ( $cookie = $this->get_session_cookie() ) {
 41             $this->_customer_id        = $cookie[0];
 42             $this->_session_expiration = $cookie[1];
 43             $this->_session_expiring   = $cookie[2];
 44 
 45             // Update session if its close to expiring
 46             if ( time() > $this->_session_expiring ) {
 47                 $this->set_session_expiration();
 48                 update_option( '_wc_session_expires_' . $this->_customer_id, $this->_session_expiration );
 49             }
 50 
 51         } else {
 52             $this->set_session_expiration();
 53             $this->_customer_id = $this->generate_customer_id();
 54         }
 55 
 56         $this->_data = $this->get_session_data();
 57 
 58         // Set/renew our cookie
 59         $to_hash      = $this->_customer_id . $this->_session_expiration;
 60         $cookie_hash  = hash_hmac( 'md5', $to_hash, wp_hash( $to_hash ) );
 61         $cookie_value = $this->_customer_id . '||' . $this->_session_expiration . '||' . $this->_session_expiring . '||' . $cookie_hash;
 62 
 63         setcookie( $this->_cookie, $cookie_value, $this->_session_expiration, COOKIEPATH, COOKIE_DOMAIN, false, true );
 64 
 65         // Actions
 66         add_action( 'woocommerce_cleanup_sessions', array( $this, 'cleanup_sessions' ), 10 );
 67         add_action( 'shutdown', array( $this, 'save_data' ), 20 );
 68     }

 70     /**
 71      * set_session_expiration function.
 72      *
 73      * @access private
 74      * @return void
 75      */
 76     private function set_session_expiration() {
 77         $this->_session_expiring    = time() + intval( apply_filters( 'wc_session_expiring', 60 * 60 * 47 ) ); // 47 Hours
 78         $this->_session_expiration  = time() + intval( apply_filters( 'wc_session_expiration', 60 * 60 * 48 ) ); // 48 Hours
 79     }
 80 
 81     /**
 82      * generate_customer_id function.
 83      *
 84      * @access private
 85      * @return mixed
 86      */
 87     private function generate_customer_id() {
 88         if ( is_user_logged_in() )
 89             return get_current_user_id();
 90         else
 91             return wp_generate_password( 32, false );
 92     }
 93 
 94     /**
 95      * get_session_cookie function.
 96      *
 97      * @access private
 98      * @return mixed
 99      */
100     private function get_session_cookie() {
101         if ( empty( $_COOKIE[ $this->_cookie ] ) )
102             return false;
103 
104         list( $customer_id, $session_expiration, $session_expiring, $cookie_hash ) = explode( '||', $_COOKIE[ $this->_cookie ] );
105 
106         // Validate hash
107         $to_hash = $customer_id . $session_expiration;
108         $hash    = hash_hmac( 'md5', $to_hash, wp_hash( $to_hash ) );
109 
110         if ( $hash != $cookie_hash )
111             return false;
112 
113         return array( $customer_id, $session_expiration, $session_expiring, $cookie_hash );
114     }

122     private function get_session_data() {
123         return get_option( '_wc_session_' . $this->_customer_id, array() );
124     }

126     /**
127      * save_data function.
128      *
129      * @access public
130      * @return void
131      */
132     public function save_data() {
133         // Dirty if something changed - prevents saving nothing new
134         if ( $this->_dirty ) {
135 
136             $session_option = '_wc_session_' . $this->_customer_id;
137             $session_expiry_option = '_wc_session_expires_' . $this->_customer_id;
138 
139             if ( false === get_option( $session_option ) ) {
140                 add_option( $session_option, $this->_data, '', 'no' );
141                 add_option( $session_expiry_option, $this->_session_expiration, '', 'no' );
142             } else {
143                 update_option( $session_option, $this->_data );
144             }
145         }
146     }

148     /**
149      * cleanup_sessions function.
150      *
151      * @access public
152      * @return void
153      */
154     public function cleanup_sessions() {
155         global $wpdb;
156 
157         $now = time();
158 
159         $session_names = $wpdb->get_col( $wpdb->prepare( "
160             SELECT
161                 a.option_name
162             FROM
163                 {$wpdb->options} a
164             WHERE
165                 a.option_name LIKE '_wc_session_expires_%%'
166                 AND a.option_value < %s
167         ", $now ) );
168 
169         // Clear cache
170         foreach ( $session_names as $session_name )
171             wp_cache_delete( substr( $session_name, 10 ), 'options' );
172 
173         // Delete rows
174         $wpdb->query( $wpdb->prepare( "
175             DELETE
176                 a, b
177             FROM
178                 {$wpdb->options} a, {$wpdb->options} b
179             WHERE
180                 a.option_name LIKE '_wc_session_%%' AND
181                 b.option_name = CONCAT(
182                     '_wc_session_expires_',
183                     SUBSTRING(
184                         a.option_name,
185                         CHAR_LENGTH('_wc_session_') + 1
186                     )
187                 )
188                 AND b.option_value < %s
189         ", $now ) );
190     }
