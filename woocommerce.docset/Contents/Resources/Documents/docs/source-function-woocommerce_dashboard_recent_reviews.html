333 /**
334  * Sales widget javascript
335  *
336  * @access public
337  * @return void
338  */
339 function woocommerce_dashboard_sales_js() {
340 
341     global $woocommerce, $wp_locale;
342 
343     $screen = get_current_screen();
344 
345     if (!$screen || $screen->id!=='dashboard') return;
346 
347     global $current_month_offset, $the_month_num, $the_year;
348 
349     // Get orders to display in widget
350     add_filter( 'posts_where', 'orders_this_month' );
351 
352     $args = array(
353         'numberposts'     => -1,
354         'orderby'         => 'post_date',
355         'order'           => 'DESC',
356         'post_type'       => 'shop_order',
357         'post_status'     => 'publish' ,
358         'suppress_filters' => false,
359         'tax_query' => array(
360             array(
361                 'taxonomy' => 'shop_order_status',
362                 'terms' => apply_filters( 'woocommerce_reports_order_statuses', array( 'completed', 'processing', 'on-hold' ) ),
363                 'field' => 'slug',
364                 'operator' => 'IN'
365             )
366         )
367     );
368     $orders = get_posts( $args );
369 
370     $order_counts = array();
371     $order_amounts = array();
372 
373     // Blank date ranges to begin
374     $month = $the_month_num;
375     $year = (int) $the_year;
376 
377     $first_day = strtotime("{$year}-{$month}-01");
378     $last_day = strtotime('-1 second', strtotime('+1 month', $first_day));
379 
380     if ((date('m') - $the_month_num)==0) :
381         $up_to = date('d', strtotime('NOW'));
382     else :
383         $up_to = date('d', $last_day);
384     endif;
385     $count = 0;
386 
387     while ($count < $up_to) :
388 
389         $time = strtotime(date('Ymd', strtotime('+ '.$count.' DAY', $first_day))).'000';
390 
391         $order_counts[$time] = 0;
392         $order_amounts[$time] = 0;
393 
394         $count++;
395     endwhile;
396 
397     if ($orders) :
398         foreach ($orders as $order) :
399 
400             $order_data = new WC_Order($order->ID);
401 
402             if ($order_data->status=='cancelled' || $order_data->status=='refunded') continue;
403 
404             $time = strtotime(date('Ymd', strtotime($order->post_date))).'000';
405 
406             if (isset($order_counts[$time])) :
407                 $order_counts[$time]++;
408             else :
409                 $order_counts[$time] = 1;
410             endif;
411 
412             if (isset($order_amounts[$time])) :
413                 $order_amounts[$time] = $order_amounts[$time] + $order_data->order_total;
414             else :
415                 $order_amounts[$time] = (float) $order_data->order_total;
416             endif;
417 
418         endforeach;
419     endif;
420 
421     remove_filter( 'posts_where', 'orders_this_month' );
422 
423     /* Script variables */
424     $params = array(
425         'currency_symbol'   => get_woocommerce_currency_symbol(),
426         'number_of_sales'   => absint( array_sum( $order_counts ) ),
427         'sales_amount'      => woocommerce_price( array_sum( $order_amounts ) ),
428         'sold'              => __( 'Sold', 'woocommerce' ),
429         'earned'            => __( 'Earned', 'woocommerce' ),
430         'month_names'       => array_values( $wp_locale->month_abbrev ),
431     );
432 
433     $order_counts_array = array();
434     foreach ($order_counts as $key => $count) :
435         $order_counts_array[] = array($key, $count);
436     endforeach;
437 
438     $order_amounts_array = array();
439     foreach ($order_amounts as $key => $amount) :
440         $order_amounts_array[] = array($key, $amount);
441     endforeach;
442 
443     $order_data = array( 'order_counts' => $order_counts_array, 'order_amounts' => $order_amounts_array );
444 
445     $params['order_data'] = json_encode($order_data);
446 
447     // Queue scripts
448     $suffix = defined('SCRIPT_DEBUG') && SCRIPT_DEBUG ? '' : '.min';
449 
450     wp_register_script( 'woocommerce_dashboard_sales', $woocommerce->plugin_url() . '/assets/js/admin/dashboard_sales' . $suffix . '.js', array( 'jquery', 'flot', 'flot-resize' ), '1.0' );
451     wp_register_script( 'flot', $woocommerce->plugin_url() . '/assets/js/admin/jquery.flot'.$suffix.'.js', 'jquery', '1.0' );
452     wp_register_script( 'flot-resize', $woocommerce->plugin_url() . '/assets/js/admin/jquery.flot.resize'.$suffix.'.js', 'jquery', '1.0' );
453 
454     wp_localize_script( 'woocommerce_dashboard_sales', 'params', $params );
455 
456     wp_print_scripts('woocommerce_dashboard_sales');
457 }
