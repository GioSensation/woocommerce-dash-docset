117      * @access public
118      * @param WC_Order $order
119      * @param bool $sent_to_admin

122     function email_instructions( $order, $sent_to_admin ) {
123         if ( $sent_to_admin ) return;
124 
125         if ( $order->status !== 'on-hold') return;
126 
127         if ( $order->payment_method !== 'cheque') return;
128 
129         if ( $description = $this->get_description() )
130             echo wpautop( wptexturize( $description ) );
131     }

134     /**
135      * Process the payment and return the result
136      *
137      * @access public
138      * @param int $order_id
139      * @return array
140      */
141     function process_payment( $order_id ) {
142         global $woocommerce;
143 
144         $order = new WC_Order( $order_id );
145 
146         // Mark as on-hold (we're awaiting the cheque)
147         $order->update_status('on-hold', __( 'Awaiting cheque payment', 'woocommerce' ));
148 
149         // Reduce stock levels
150         $order->reduce_order_stock();
151 
152         // Remove cart
153         $woocommerce->cart->empty_cart();
154 
155         // Return thankyou redirect
156         return array(
157             'result'    => 'success',
158             'redirect'  => add_query_arg('key', $order->order_key, add_query_arg('order', $order_id, get_permalink(woocommerce_get_page_id('thanks'))))
159         );
160 
161     }
