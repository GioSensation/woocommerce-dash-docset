169 /**
170  * Save the coupon data meta box.
171  *
172  * @access public
173  * @param mixed $post_id
174  * @param mixed $post
175  * @return void
176  */
177 function woocommerce_process_shop_coupon_meta( $post_id, $post ) {
178     global $wpdb, $woocommerce_errors;
179 
180     // Ensure coupon code is correctly formatted
181     $post->post_title = apply_filters( 'woocommerce_coupon_code', $post->post_title );
182     $wpdb->update( $wpdb->posts, array( 'post_title' => $post->post_title ), array( 'ID' => $post_id ) );
183 
184     // Check for dupe coupons
185     $coupon_found = $wpdb->get_var( $wpdb->prepare( "
186         SELECT $wpdb->posts.ID
187         FROM $wpdb->posts
188         WHERE $wpdb->posts.post_type = 'shop_coupon'
189         AND $wpdb->posts.post_status = 'publish'
190         AND $wpdb->posts.post_title = '%s'
191         AND $wpdb->posts.ID != %s
192      ", $post->post_title, $post_id ) );
193 
194     if ( $coupon_found )
195         $woocommerce_errors[] = __( 'Coupon code already exists - customers will use the latest coupon with this code.', 'woocommerce' );
196 
197     // Add/Replace data to array
198     $type               = woocommerce_clean( $_POST['discount_type'] );
199     $amount             = woocommerce_clean( $_POST['coupon_amount'] );
200     $usage_limit        = empty( $_POST['usage_limit'] ) ? '' : absint( $_POST['usage_limit'] );
201     $individual_use     = isset( $_POST['individual_use'] ) ? 'yes' : 'no';
202     $expiry_date        = woocommerce_clean( $_POST['expiry_date'] );
203     $apply_before_tax   = isset( $_POST['apply_before_tax'] ) ? 'yes' : 'no';
204     $free_shipping      = isset( $_POST['free_shipping'] ) ? 'yes' : 'no';
205     $exclude_sale_items = isset( $_POST['exclude_sale_items'] ) ? 'yes' : 'no';
206     $minimum_amount     = woocommerce_clean( $_POST['minimum_amount'] );
207     $customer_email     = array_filter( array_map( 'trim', explode( ',', woocommerce_clean( $_POST['customer_email'] ) ) ) );
208 
209     if ( isset( $_POST['product_ids'] ) ) {
210         $product_ids            = implode( ',', array_filter( array_map( 'intval', (array) $_POST['product_ids'] ) ) );
211     } else {
212         $product_ids = '';
213     }
214 
215     if ( isset( $_POST['exclude_product_ids'] ) ) {
216         $exclude_product_ids    = implode( ',', array_filter( array_map( 'intval', (array) $_POST['exclude_product_ids'] ) ) );
217     } else {
218         $exclude_product_ids = '';
219     }
220 
221     $product_categories         = isset( $_POST['product_categories'] ) ? array_map( 'intval', $_POST['product_categories'] ) : array();
222     $exclude_product_categories = isset( $_POST['exclude_product_categories'] ) ? array_map( 'intval', $_POST['exclude_product_categories'] ) : array();
223 
224     // Save
225     update_post_meta( $post_id, 'discount_type', $type );
226     update_post_meta( $post_id, 'coupon_amount', $amount );
227     update_post_meta( $post_id, 'individual_use', $individual_use );
228     update_post_meta( $post_id, 'product_ids', $product_ids );
229     update_post_meta( $post_id, 'exclude_product_ids', $exclude_product_ids );
230     update_post_meta( $post_id, 'usage_limit', $usage_limit );
231     update_post_meta( $post_id, 'expiry_date', $expiry_date );
232     update_post_meta( $post_id, 'apply_before_tax', $apply_before_tax );
233     update_post_meta( $post_id, 'free_shipping', $free_shipping );
234     update_post_meta( $post_id, 'exclude_sale_items', $exclude_sale_items );
235     update_post_meta( $post_id, 'product_categories', $product_categories );
236     update_post_meta( $post_id, 'exclude_product_categories', $exclude_product_categories );
237     update_post_meta( $post_id, 'minimum_amount', $minimum_amount );
238     update_post_meta( $post_id, 'customer_email', $customer_email );
239 
240     do_action( 'woocommerce_coupon_options' );
241 }
