 642 /**
 643  * Save the product data meta box.
 644  *
 645  * @access public
 646  * @param mixed $post_id
 647  * @param mixed $post
 648  * @return void
 649  */
 650 function woocommerce_process_product_meta( $post_id, $post ) {
 651     global $wpdb, $woocommerce, $woocommerce_errors;
 652 
 653     // Add any default post meta
 654     add_post_meta( $post_id, 'total_sales', '0', true );
 655 
 656     // Get types
 657     $product_type       = empty( $_POST['product-type'] ) ? 'simple' : sanitize_title( stripslashes( $_POST['product-type'] ) );
 658     $is_downloadable    = isset( $_POST['_downloadable'] ) ? 'yes' : 'no';
 659     $is_virtual         = isset( $_POST['_virtual'] ) ? 'yes' : 'no';
 660 
 661     // Product type + Downloadable/Virtual
 662     wp_set_object_terms( $post_id, $product_type, 'product_type' );
 663     update_post_meta( $post_id, '_downloadable', $is_downloadable );
 664     update_post_meta( $post_id, '_virtual', $is_virtual );
 665 
 666     // Gallery Images
 667     $attachment_ids = array_filter( explode( ',', woocommerce_clean( $_POST['product_image_gallery'] ) ) );
 668     update_post_meta( $post_id, '_product_image_gallery', implode( ',', $attachment_ids ) );
 669 
 670     // Update post meta
 671     update_post_meta( $post_id, '_regular_price', stripslashes( $_POST['_regular_price'] ) );
 672     update_post_meta( $post_id, '_sale_price', stripslashes( $_POST['_sale_price'] ) );
 673     update_post_meta( $post_id, '_tax_status', stripslashes( $_POST['_tax_status'] ) );
 674     update_post_meta( $post_id, '_tax_class', stripslashes( $_POST['_tax_class'] ) );
 675     update_post_meta( $post_id, '_visibility', stripslashes( $_POST['_visibility'] ) );
 676     update_post_meta( $post_id, '_purchase_note', stripslashes( $_POST['_purchase_note'] ) );
 677     update_post_meta( $post_id, '_featured', isset( $_POST['_featured'] ) ? 'yes' : 'no' );
 678 
 679     // Dimensions
 680     if ( $is_virtual == 'no' ) {
 681         update_post_meta( $post_id, '_weight', stripslashes( $_POST['_weight'] ) );
 682         update_post_meta( $post_id, '_length', stripslashes( $_POST['_length'] ) );
 683         update_post_meta( $post_id, '_width', stripslashes( $_POST['_width'] ) );
 684         update_post_meta( $post_id, '_height', stripslashes( $_POST['_height'] ) );
 685     } else {
 686         update_post_meta( $post_id, '_weight', '' );
 687         update_post_meta( $post_id, '_length', '' );
 688         update_post_meta( $post_id, '_width', '' );
 689         update_post_meta( $post_id, '_height', '' );
 690     }
 691 
 692     // Save shipping class
 693     $product_shipping_class = $_POST['product_shipping_class'] > 0 && $product_type != 'external' ? absint( $_POST['product_shipping_class'] ) : '';
 694     wp_set_object_terms( $post_id, $product_shipping_class, 'product_shipping_class');
 695 
 696     // Unique SKU
 697     $sku                = get_post_meta($post_id, '_sku', true);
 698     $new_sku            = esc_html( trim( stripslashes( $_POST['_sku'] ) ) );
 699     if ( $new_sku == '' ) {
 700         update_post_meta( $post_id, '_sku', '' );
 701     } elseif ( $new_sku !== $sku ) {
 702         if ( ! empty( $new_sku ) ) {
 703             if (
 704                 $wpdb->get_var( $wpdb->prepare("
 705                     SELECT $wpdb->posts.ID
 706                     FROM $wpdb->posts
 707                     LEFT JOIN $wpdb->postmeta ON ($wpdb->posts.ID = $wpdb->postmeta.post_id)
 708                     WHERE $wpdb->posts.post_type = 'product'
 709                     AND $wpdb->posts.post_status = 'publish'
 710                     AND $wpdb->postmeta.meta_key = '_sku' AND $wpdb->postmeta.meta_value = '%s'
 711                  ", $new_sku ) )
 712                 ) {
 713                 $woocommerce_errors[] = __( 'Product SKU must be unique.', 'woocommerce' );
 714             } else {
 715                 update_post_meta( $post_id, '_sku', $new_sku );
 716             }
 717         } else {
 718             update_post_meta( $post_id, '_sku', '' );
 719         }
 720     }
 721 
 722     // Save Attributes
 723     $attributes = array();
 724 
 725     if ( isset( $_POST['attribute_names'] ) ) {
 726         $attribute_names = $_POST['attribute_names'];
 727         $attribute_values = $_POST['attribute_values'];
 728 
 729         if ( isset( $_POST['attribute_visibility'] ) )
 730             $attribute_visibility = $_POST['attribute_visibility'];
 731 
 732         if ( isset( $_POST['attribute_variation'] ) )
 733             $attribute_variation = $_POST['attribute_variation'];
 734 
 735         $attribute_is_taxonomy = $_POST['attribute_is_taxonomy'];
 736         $attribute_position = $_POST['attribute_position'];
 737 
 738         $attribute_names_count = sizeof( $attribute_names );
 739 
 740         for ( $i=0; $i < $attribute_names_count; $i++ ) {
 741             if ( ! $attribute_names[ $i ] )
 742                 continue;
 743 
 744             $is_visible     = isset( $attribute_visibility[ $i ] ) ? 1 : 0;
 745             $is_variation   = isset( $attribute_variation[ $i ] ) ? 1 : 0;
 746             $is_taxonomy    = $attribute_is_taxonomy[ $i ] ? 1 : 0;
 747 
 748             if ( $is_taxonomy ) {
 749

 754                         $values = array_map( 'sanitize_title', $attribute_values[ $i ] );
 755                     } else {
 756                         // Text based, separate by pipe
 757                         $values = array_map( 'sanitize_title', explode( '|', $attribute_values[ $i ] ) );
 758                     }
 759 
 760                     // Remove empty items in the array
 761                     $values = array_filter( $values );
 762 
 763                 } else {
 764                     $values = array();
 765                 }
 766 
 767                 // Update post terms
 768                 if ( taxonomy_exists( $attribute_names[ $i ] ) )
 769                     wp_set_object_terms( $post_id, $values, $attribute_names[ $i ] );
 770 
 771                 if ( $values ) {
 772                     // Add attribute to array, but don't set values
 773                     $attributes[ sanitize_title( $attribute_names[ $i ] ) ] = array(
 774                         'name'          => woocommerce_clean( $attribute_names[ $i ] ),
 775                         'value'         => '',
 776                         'position'      => $attribute_position[ $i ],
 777                         'is_visible'    => $is_visible,
 778                         'is_variation'  => $is_variation,
 779                         'is_taxonomy'   => $is_taxonomy
 780                     );
 781                 }
 782 
 783             } elseif ( isset( $attribute_values[ $i ] ) ) {
 784 
 785                 // Text based, separate by pipe
 786                 $values = implode( ' | ', array_map( 'woocommerce_clean', explode( '|', $attribute_values[ $i ] ) ) );
 787 
 788                 // Custom attribute - Add attribute to array and set the values
 789                 $attributes[ sanitize_title( $attribute_names[ $i ] ) ] = array(
 790                     'name'          => woocommerce_clean( $attribute_names[ $i ] ),
 791                     'value'         => $values,
 792                     'position'      => $attribute_position[ $i ],
 793                     'is_visible'    => $is_visible,
 794                     'is_variation'  => $is_variation,
 795                     'is_taxonomy'   => $is_taxonomy
 796                 );
 797             }
 798 
 799          }
 800     }
 801 
 802     if ( ! function_exists( 'attributes_cmp' ) ) {
 803         function attributes_cmp( $a, $b ) {
 804             if ( $a['position'] == $b['position'] ) return 0;
 805             return ( $a['position'] < $b['position'] ) ? -1 : 1;
 806         }
 807     }
 808     uasort( $attributes, 'attributes_cmp' );
 809 
 810     update_post_meta( $post_id, '_product_attributes', $attributes );
 811 
 812     // Sales and prices
 813     if ( in_array( $product_type, array( 'variable', 'grouped' ) ) ) {
 814 
 815         // Variable and grouped products have no prices
 816         update_post_meta( $post_id, '_regular_price', '' );
 817         update_post_meta( $post_id, '_sale_price', '' );
 818         update_post_meta( $post_id, '_sale_price_dates_from', '' );
 819         update_post_meta( $post_id, '_sale_price_dates_to', '' );
 820         update_post_meta( $post_id, '_price', '' );
 821 
 822     } else {
 823 
 824         $date_from = isset( $_POST['_sale_price_dates_from'] ) ? $_POST['_sale_price_dates_from'] : '';
 825         $date_to = isset( $_POST['_sale_price_dates_to'] ) ? $_POST['_sale_price_dates_to'] : '';
 826 
 827         // Dates
 828         if ( $date_from )
 829             update_post_meta( $post_id, '_sale_price_dates_from', strtotime( $date_from ) );
 830         else
 831             update_post_meta( $post_id, '_sale_price_dates_from', '' );
 832 
 833         if ( $date_to )
 834             update_post_meta( $post_id, '_sale_price_dates_to', strtotime( $date_to ) );
 835         else
 836             update_post_meta( $post_id, '_sale_price_dates_to', '' );
 837 
 838         if ( $date_to && ! $date_from )
 839             update_post_meta( $post_id, '_sale_price_dates_from', strtotime( 'NOW', current_time( 'timestamp' ) ) );
 840 
 841         // Update price if on sale
 842         if ( $_POST['_sale_price'] != '' && $date_to == '' && $date_from == '' )
 843             update_post_meta( $post_id, '_price', stripslashes( $_POST['_sale_price'] ) );
 844         else
 845             update_post_meta( $post_id, '_price', stripslashes( $_POST['_regular_price'] ) );
 846 
 847         if ( $_POST['_sale_price'] != '' && $date_from && strtotime( $date_from ) < strtotime( 'NOW', current_time( 'timestamp' ) ) )
 848             update_post_meta( $post_id, '_price', stripslashes($_POST['_sale_price']) );
 849 
 850         if ( $date_to && strtotime( $date_to ) < strtotime( 'NOW', current_time( 'timestamp' ) ) ) {
 851             update_post_meta( $post_id, '_price', stripslashes($_POST['_regular_price']) );
 852             update_post_meta( $post_id, '_sale_price_dates_from', '');
 853             update_post_meta( $post_id, '_sale_price_dates_to', '');
 854         }
 855     }
 856 
 857     // Update parent if grouped so price sorting works and stays in sync with the cheapest child
 858     if ( $post->post_parent > 0 || $product_type == 'grouped' || $_POST['previous_parent_id'] > 0 ) {
 859 
 860         $clear_parent_ids = array();
 861 
 862         if ( $post->post_parent > 0 )
 863             $clear_parent_ids[] = $post->post_parent;
 864
