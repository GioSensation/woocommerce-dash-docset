 54     /**
 55      * Loads all product data from custom fields
 56      *
 57      * @access public
 58      * @param int $variation_id ID of the variation to load
 59      * @param array $args Array of the arguments containing parent product data
 60      * @return void
 61      */
 62     public function __construct( $variation, $args = array() ) {
 63 
 64         $this->product_type = 'variation';
 65 
 66         if ( is_object( $variation ) ) {
 67             $this->variation_id = absint( $variation->ID );
 68         } else {
 69             $this->variation_id = absint( $variation );
 70         }
 71 
 72         /* Get main product data from parent (args) */
 73         $this->id   = ! empty( $args['parent_id'] ) ? intval( $args['parent_id'] ) : wp_get_post_parent_id( $this->variation_id );
 74 
 75         // The post doesn't have a parent id, therefore its invalid.
 76         if ( empty( $this->id ) )
 77             return false;
 78 
 79         // Get post data
 80         $this->parent = ! empty( $args['parent'] ) ? $args['parent'] : get_product( $this->id );
 81         $this->post   = ! empty( $this->parent->post ) ? $this->parent->post : array();
 82         $this->product_custom_fields = get_post_meta( $this->variation_id );
 83 
 84         // Get the variation attributes from meta
 85         foreach ( $this->product_custom_fields as $name => $value ) {
 86             if ( ! strstr( $name, 'attribute_' ) )
 87                 continue;
 88 
 89             $this->variation_data[ $name ] = sanitize_title( $value[0] );
 90         }
 91 
 92         // Now get variation meta to override the parent variable product
 93         if ( ! empty( $this->product_custom_fields['_sku'][0] ) ) {
 94             $this->variation_has_sku = true;
 95             $this->sku               = $this->product_custom_fields['_sku'][0];
 96         }
 97 
 98         if ( isset( $this->product_custom_fields['_stock'][0] ) && $this->product_custom_fields['_stock'][0] !== '' ) {
 99             $this->variation_has_stock = true;
100             $this->manage_stock        = 'yes';
101             $this->stock               = $this->product_custom_fields['_stock'][0];
102         }
103 
104         if ( isset( $this->product_custom_fields['_weight'][0] ) && $this->product_custom_fields['_weight'][0] !== '' ) {
105             $this->variation_has_weight = true;
106             $this->weight               = $this->product_custom_fields['_weight'][0];
107         }
108 
109         if ( isset( $this->product_custom_fields['_length'][0] ) && $this->product_custom_fields['_length'][0] !== '' ) {
110             $this->variation_has_length = true;
111             $this->length               = $this->product_custom_fields['_length'][0];
112         }
113 
114         if ( isset( $this->product_custom_fields['_width'][0] ) && $this->product_custom_fields['_width'][0] !== '' ) {
115             $this->variation_has_width = true;
116             $this->width               = $this->product_custom_fields['_width'][0];
117         }
118 
119         if ( isset( $this->product_custom_fields['_height'][0] ) && $this->product_custom_fields['_height'][0] !== '' ) {
120             $this->variation_has_height = true;
121             $this->height               = $this->product_custom_fields['_height'][0];
122         }
123 
124         if ( isset( $this->product_custom_fields['_downloadable'][0] ) && $this->product_custom_fields['_downloadable'][0] == 'yes' ) {
125             $this->downloadable = 'yes';
126         } else {
127             $this->downloadable = 'no';
128         }
129 
130         if ( isset( $this->product_custom_fields['_virtual'][0] ) && $this->product_custom_fields['_virtual'][0] == 'yes' ) {
131             $this->virtual = 'yes';
132         } else {
133             $this->virtual = 'no';
134         }
135 
136         if ( isset( $this->product_custom_fields['_tax_class'][0] ) ) {
137             $this->variation_has_tax_class = true;
138             $this->tax_class               = $this->product_custom_fields['_tax_class'][0];
139         }
140 
141         if ( isset( $this->product_custom_fields['_sale_price_dates_from'][0] ) )
142             $this->sale_price_dates_from = $this->product_custom_fields['_sale_price_dates_from'][0];
143 
144         if ( isset( $this->product_custom_fields['_sale_price_dates_to'][0] ) )
145             $this->sale_price_dates_from = $this->product_custom_fields['_sale_price_dates_to'][0];
146 
147         // Prices
148         $this->price         = isset( $this->product_custom_fields['_price'][0] ) ? $this->product_custom_fields['_price'][0] : '';
149         $this->regular_price = isset( $this->product_custom_fields['_regular_price'][0] ) ? $this->product_custom_fields['_regular_price'][0] : '';
150         $this->sale_price    = isset( $this->product_custom_fields['_sale_price'][0] ) ? $this->product_custom_fields['_sale_price'][0] : '';
151 
152         // Backwards compat for prices
153         if ( $this->price !== '' && $this->regular_price == '' ) {
154             update_post_meta( $this->variation_id, '_regular_price', $this->price );
155             $this->regular_price = $this->price;
156 
157             if ( $this->sale_price !== '' && $this->sale_price < $this->regular_price ) {
158                 update_post_meta( $this->variation_id, '_price', $this->sale_price );
159                 $this->price = $this->sale_price;
160             }
161         }
162 
163         $this->total_stock = $this->stock;
164     }

183 
184         $visible = true;
185 
186         // Out of stock visibility
187         if ( get_option('woocommerce_hide_out_of_stock_items') == 'yes' && ! $this->is_in_stock() )
188             $visible = false;
189 
190         // Price not set
191         elseif ( $this->price == "" )
192             $visible = false;
193 
194         return apply_filters( 'woocommerce_product_is_visible', $visible, $this->id );

226     /**
227      * Get variation price HTML. Prices are not inherited from parents.
228      *
229      * @return string containing the formatted price
230      */
231     public function get_price_html() {
232 
233         if ( $this->price !== '' ) {
234             if ( $this->price == $this->sale_price && $this->sale_price < $this->regular_price ) {
235 
236                 $price = '<del>' . woocommerce_price( $this->regular_price ) . '</del> <ins>' . woocommerce_price( $this->sale_price ) . '</ins>';
237                 $price = apply_filters( 'woocommerce_variation_sale_price_html', $price, $this );
238 
239             } elseif ( $this->price > 0 ) {
240 
241                 $price = woocommerce_price( $this->price );
242                 $price = apply_filters( 'woocommerce_variation_price_html', $price, $this );
243 
244             } else {
245 
246                 $price = __( 'Free!', 'woocommerce' );
247                 $price = apply_filters( 'woocommerce_variation_free_price_html', $price, $this );
248 
249             }
250         } else {
251             $price = apply_filters( 'woocommerce_variation_empty_price_html', '', $this );
252         }
253 
254         return $price;
255     }

284     /**
285      * Set stock level of the product.
286      *
287      * @access public
288      * @param mixed $amount (default: null)
289      * @return int Stock
290      */
291     function set_stock( $amount = null ) {
292         global $woocommerce;
293 
294         if ( $this->variation_has_stock ) {
295             if ( $this->managing_stock() && ! is_null( $amount ) ) {
296 
297                 $this->stock = intval( $amount );
298                 $this->total_stock = intval( $amount );
299                 update_post_meta( $this->variation_id, '_stock', $this->stock );
300                 $woocommerce->clear_product_transients( $this->id ); // Clear transient
301 
302                 // Check parents out of stock attribute
303                 if ( ! $this->is_in_stock() ) {
304 
305                     // Check parent
306                     $parent_product = get_product( $this->id );
307 
308                     // Only continue if the parent has backorders off
309                     if ( ! $parent_product->backorders_allowed() && $parent_product->get_total_stock() <= 0 )
310                         $this->set_stock_status( 'outofstock' );
311 
312                 } elseif ( $this->is_in_stock() ) {
313                     $this->set_stock_status( 'instock' );
314                 }
315 
316                 return apply_filters( 'woocommerce_stock_amount', $this->stock );
317             }
318         } else {
319             return parent::set_stock( $amount );
320         }
321     }

440     public function get_file_download_path( $download_id ) {
441

448         } elseif ( isset( $file_paths[ $download_id ] ) ) {
449             $file_path = $file_paths[ $download_id ];
