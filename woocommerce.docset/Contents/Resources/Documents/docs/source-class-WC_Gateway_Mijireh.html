181     /**
182      * Process the payment and return the result
183      *
184      * @access public
185      * @param int $order_id
186      * @return array
187      */
188     public function process_payment( $order_id ) {
189         global $woocommerce;
190 
191         $this->init_mijireh();
192 
193         $mj_order = new Mijireh_Order();
194         $wc_order = new WC_Order( $order_id );
195 
196         // add items to order
197         $items = $wc_order->get_items();
198 
199         foreach( $items as $item ) {
200             $product = $wc_order->get_product_from_item( $item );
201 
202             if ( get_option( 'woocommerce_prices_include_tax' ) == 'yes' ) {
203 
204                 $mj_order->add_item( $item['name'], $wc_order->get_item_subtotal( $item, true, false ), $item['qty'], $product->get_sku() );
205 
206             } else {
207 
208                 $mj_order->add_item( $item['name'], $wc_order->get_item_subtotal( $item, false, false ), $item['qty'], $product->get_sku() );
209 
210             }
211 
212 
213         }
214 
215         // Handle fees
216         $items = $wc_order->get_fees();
217 
218         foreach( $items as $item ) {
219             $mj_order->add_item( $item['name'], $item['line_total'], 1, '' );
220         }
221 
222         // add billing address to order
223         $billing                    = new Mijireh_Address();
224         $billing->first_name        = $wc_order->billing_first_name;
225         $billing->last_name         = $wc_order->billing_last_name;
226         $billing->street            = $wc_order->billing_address_1;
227         $billing->apt_suite         = $wc_order->billing_address_2;
228         $billing->city              = $wc_order->billing_city;
229         $billing->state_province    = $wc_order->billing_state;
230         $billing->zip_code          = $wc_order->billing_postcode;
231         $billing->country           = $wc_order->billing_country;
232         $billing->company           = $wc_order->billing_company;
233         $billing->phone             = $wc_order->billing_phone;
234         if ( $billing->validate() )
235             $mj_order->set_billing_address( $billing );
236 
237         // add shipping address to order
238         $shipping                   = new Mijireh_Address();
239         $shipping->first_name       = $wc_order->shipping_first_name;
240         $shipping->last_name        = $wc_order->shipping_last_name;
241         $shipping->street           = $wc_order->shipping_address_1;
242         $shipping->apt_suite        = $wc_order->shipping_address_2;
243         $shipping->city             = $wc_order->shipping_city;
244         $shipping->state_province   = $wc_order->shipping_state;
245         $shipping->zip_code         = $wc_order->shipping_postcode;
246         $shipping->country          = $wc_order->shipping_country;
247         $shipping->company          = $wc_order->shipping_company;
248         if ( $shipping->validate() )
249             $mj_order->set_shipping_address( $shipping );
250 
251         // set order name
252         $mj_order->first_name       = $wc_order->billing_first_name;
253         $mj_order->last_name        = $wc_order->billing_last_name;
254         $mj_order->email            = $wc_order->billing_email;
255 
256         // set order totals
257         $mj_order->total            = $wc_order->get_order_total();
258         $mj_order->discount         = $wc_order->get_total_discount();
259 
260         if ( get_option( 'woocommerce_prices_include_tax' ) == 'yes' ) {
261             $mj_order->shipping         = $wc_order->get_shipping() + $wc_order->get_shipping_tax();
262             $mj_order->show_tax         = false;
263         } else {
264             $mj_order->shipping         = $wc_order->get_shipping();
265             $mj_order->tax              = $wc_order->get_total_tax();
266         }
267 
268         // add meta data to identify woocommerce order
269         $mj_order->add_meta_data( 'wc_order_id', $order_id );
270 
271         // Set URL for mijireh payment notificatoin - use WC API
272         $mj_order->return_url       = str_replace( 'https:', 'http:', add_query_arg( 'wc-api', 'WC_Gateway_Mijireh', home_url( '/' ) ) );
273 
274         // Identify woocommerce
275         $mj_order->partner_id       = 'woo';
276 
277         try {
278             $mj_order->create();
279             $result = array(
280                 'result' => 'success',
281                 'redirect' => $mj_order->checkout_url
282             );
283             return $result;
284         } catch (Mijireh_Exception $e) {
285             $woocommerce->add_error( __('Mijireh error:', 'woocommerce' ) . $e->getMessage() );
286         }
287     }

295     public function init_mijireh() {
296         if ( ! class_exists( 'Mijireh' ) ) {
