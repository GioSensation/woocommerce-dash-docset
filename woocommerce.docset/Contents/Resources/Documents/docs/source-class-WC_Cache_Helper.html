27     /**
28      * Prevent caching on dynamic pages.
29      *
30      * @access public
31      * @return void
32      */
33     public function init() {
34         if ( false === ( $wc_page_uris = get_transient( 'woocommerce_cache_excluded_uris' ) ) ) {
35 
36             if ( ! woocommerce_get_page_id( 'cart' ) || ! woocommerce_get_page_id( 'checkout' ) || ! woocommerce_get_page_id( 'myaccount' ) )
37                 return;
38 
39             $wc_page_uris   = array();
40             $cart_page      = get_post( woocommerce_get_page_id( 'cart' ) );
41             $checkout_page  = get_post( woocommerce_get_page_id( 'checkout' ) );
42             $account_page   = get_post( woocommerce_get_page_id( 'myaccount' ) );
43 
44             $wc_page_uris[] = '/' . $cart_page->post_name;
45             $wc_page_uris[] = '/' . $checkout_page->post_name;
46             $wc_page_uris[] = '/' . $account_page->post_name;
47             $wc_page_uris[] = 'p=' . $cart_page->ID;
48             $wc_page_uris[] = 'p=' . $checkout_page->ID;
49             $wc_page_uris[] = 'p=' . $account_page->ID;
50 
51             set_transient( 'woocommerce_cache_excluded_uris', $wc_page_uris );
52         }
53 
54         if ( is_array( $wc_page_uris ) )
55             foreach( $wc_page_uris as $uri )
56                 if ( strstr( $_SERVER['REQUEST_URI'], $uri ) ) {
57                     $this->nocache();
58                     break;
59                 }
60     }
