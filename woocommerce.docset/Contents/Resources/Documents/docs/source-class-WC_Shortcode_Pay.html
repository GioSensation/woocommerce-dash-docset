 42 
 43             // Pay for existing order
 44             $order_key            = urldecode( $_GET[ 'order' ] );
 45             $order_id             = absint( $_GET[ 'order_id' ] );

 50 
 51                 // Set customer location to order location

 53                     $woocommerce->customer->set_country( $order->billing_country );
 54                 if ( $order->billing_state )

 56                 if ( $order->billing_postcode )
 57                     $woocommerce->customer->set_postcode( $order->billing_postcode );
 58 
 59                 // Show form
 60                 woocommerce_get_template( 'checkout/form-pay.php', array( 'order' => $order ) );
 61 
 62             } elseif ( ! in_array( $order->status, $valid_order_statuses ) ) {
