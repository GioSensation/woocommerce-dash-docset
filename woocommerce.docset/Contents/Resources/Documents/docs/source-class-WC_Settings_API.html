 93     /**
 94      * Initialise Gateway Settings
 95      *
 96      * Store all settings in a single database entry
 97      * and make sure the $settings array is either the default
 98      * or the settings stored in the database.
 99      *
100      * @since 1.0.0
101      * @uses get_option(), add_option()
102      * @access public
103      * @return void
104      */
105     public function init_settings() {
106 
107         if ( ! empty( $this->settings ) )
108             return;
109 
110         // Load form_field settings
111         $this->settings = get_option( $this->plugin_id . $this->id . '_settings', null );
112 
113         if ( ! $this->settings || ! is_array( $this->settings ) ) {
114 
115             $this->settings = array();
116 
117             // If there are no settings defined, load defaults
118             if ( $this->form_fields )
119                 foreach ( $this->form_fields as $k => $v )
120                     $this->settings[ $k ] = isset( $v['default'] ) ? $v['default'] : '';
121         }
122 
123         if ( $this->settings && is_array( $this->settings ) ) {
124            $this->settings = array_map( array( $this, 'format_settings' ), $this->settings );
125            $this->enabled  = isset( $this->settings['enabled'] ) && $this->settings['enabled'] == 'yes' ? 'yes' : 'no';
126         }
127     }

130     /**
131      * get_option function.
132      *
133      * Gets and option from the settings API, using defaults if necessary to prevent undefined notices.
134      *
135      * @access public
136      * @param mixed $key
137      * @param mixed $empty_value
138      * @return void
139      */
140     public function get_option( $key, $empty_value = null ) {
141 
142         if ( empty( $this->settings ) )
143             $this->init_settings();
144 
145         // Get option default if unset
146         if ( ! isset( $this->settings[ $key ] ) ) {
147             $this->settings[ $key ] = isset( $this->form_fields[ $key ]['default'] ) ? $this->form_fields[ $key ]['default'] : '';
148         }
149 
150         if ( ! is_null( $empty_value ) && empty( $this->settings[ $key ] ) )
151             $this->settings[ $key ] = $empty_value;
152 
153         return $this->settings[ $key ];
154     }

169     /**
170      * Generate Settings HTML.
171      *
172      * Generate the HTML for the fields on the "settings" screen.
173      *
174      * @access public
175      * @param bool $form_fields (default: false)
176      * @since 1.0.0
177      * @uses method_exists()
178      * @access public
179      * @return string the html for the settings
180      */
181     public function generate_settings_html ( $form_fields = false ) {
182 
183         if ( ! $form_fields ) {
184             $form_fields = $this->form_fields;
185         }
186 
187         $html = '';
188         foreach ( $form_fields as $k => $v ) {
189             if ( ! isset( $v['type'] ) || ( $v['type'] == '' ) ) { $v['type'] == 'text'; } // Default to "text" field type.
190 
191             if ( method_exists( $this, 'generate_' . $v['type'] . '_html' ) ) {
192                 $html .= $this->{'generate_' . $v['type'] . '_html'}( $k, $v );
193             } else {
194                 $html .= $this->{'generate_text_html'}( $k, $v );
195             }
196         }
197 
198         echo $html;
199     }
