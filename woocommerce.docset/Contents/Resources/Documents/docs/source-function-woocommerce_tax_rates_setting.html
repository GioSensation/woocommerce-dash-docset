340  * @access public

343 function woocommerce_tax_rates_setting_save() {
344     global $wpdb, $current_section;
345 
346     // Get class
347     $tax_classes   = array_filter( array_map( 'trim', explode( "\n", get_option('woocommerce_tax_classes' ) ) ) );
348     $current_class = '';
349 
350     foreach( $tax_classes as $class )
351         if ( sanitize_title( $class ) == $current_section )
352             $current_class = $class;
353 
354     // Get POST data
355     $tax_rate_country  = isset( $_POST['tax_rate_country'] ) ? $_POST['tax_rate_country'] : array();
356     $tax_rate_state    = isset( $_POST['tax_rate_state'] ) ? $_POST['tax_rate_state'] : array();
357     $tax_rate_postcode = isset( $_POST['tax_rate_postcode'] ) ? $_POST['tax_rate_postcode'] : array();
358     $tax_rate_city     = isset( $_POST['tax_rate_city'] ) ? $_POST['tax_rate_city'] : array();
359     $tax_rate          = isset( $_POST['tax_rate'] ) ? $_POST['tax_rate'] : array();
360     $tax_rate_name     = isset( $_POST['tax_rate_name'] ) ? $_POST['tax_rate_name'] : array();
361     $tax_rate_priority = isset( $_POST['tax_rate_priority'] ) ? $_POST['tax_rate_priority'] : array();
362     $tax_rate_compound = isset( $_POST['tax_rate_compound'] ) ? $_POST['tax_rate_compound'] : array();
363     $tax_rate_shipping = isset( $_POST['tax_rate_shipping'] ) ? $_POST['tax_rate_shipping'] : array();
364 
365     $i = 0;
366 
367     // Loop posted fields
368     foreach ( $tax_rate_country as $key => $value ) {
369 
370         // new keys are inserted...
371         if ( $key == 'new' ) {
372 
373             foreach ( $value as $new_key => $new_value ) {
374 
375                 // Sanitize + format
376                 $country  = strtoupper( woocommerce_clean( $tax_rate_country[ $key ][ $new_key ] ) );
377                 $state    = strtoupper( woocommerce_clean( $tax_rate_state[ $key ][ $new_key ] ) );
378                 $postcode = woocommerce_clean( $tax_rate_postcode[ $key ][ $new_key ] );
379                 $city     = woocommerce_clean( $tax_rate_city[ $key ][ $new_key ] );
380                 $rate     = number_format( woocommerce_clean( $tax_rate[ $key ][ $new_key ] ), 4, '.', '' );
381                 $name     = woocommerce_clean( $tax_rate_name[ $key ][ $new_key ] );
382                 $priority = absint( woocommerce_clean( $tax_rate_priority[ $key ][ $new_key ] ) );
383                 $compound = isset( $tax_rate_compound[ $key ][ $new_key ] ) ? 1 : 0;
384                 $shipping = isset( $tax_rate_shipping[ $key ][ $new_key ] ) ? 1 : 0;
385 
386                 if ( ! $name )
387                     $name = __( 'Tax', 'woocommerce' );
388 
389                 if ( $country == '*' )
390                     $country = '';
391 
392                 if ( $state == '*' )
393                     $state = '';
394 
395                 $wpdb->insert(
396                     $wpdb->prefix . "woocommerce_tax_rates",
397                     array(
398                         'tax_rate_country'  => $country,
399                         'tax_rate_state'    => $state,
400                         'tax_rate'          => $rate,
401                         'tax_rate_name'     => $name,
402                         'tax_rate_priority' => $priority,
403                         'tax_rate_compound' => $compound,
404                         'tax_rate_shipping' => $shipping,
405                         'tax_rate_order'    => $i,
406                         'tax_rate_class'    => sanitize_title( $current_class )
407                     )
408                 );
409 
410                 $tax_rate_id = $wpdb->insert_id;
411 
412                 if ( ! empty( $postcode ) ) {
413                     $postcodes = explode( ';', $postcode );
414                     $postcodes = array_map( 'strtoupper', array_map( 'woocommerce_clean', $postcodes ) );
415 
416                     $postcode_query = array();
417

419                         if ( strstr( $postcode, '-' ) ) {
420                             $postcode_parts = explode( '-', $postcode );
421 
422                             if ( is_numeric( $postcode_parts[0] ) && is_numeric( $postcode_parts[1] ) && $postcode_parts[1] > $postcode_parts[0] ) {
423                                 for ( $i = $postcode_parts[0]; $i <= $postcode_parts[1]; $i ++ ) {
424                                     if ( $i )
425                                         $postcode_query[] = "( '$i', $tax_rate_id, 'postcode' )";
426                                 }
427                             }
428                         } else {
429                             if ( $postcode )
430                                 $postcode_query[] = "( '$postcode', $tax_rate_id, 'postcode' )";
431                         }
432 
433                     $wpdb->query( "INSERT INTO {$wpdb->prefix}woocommerce_tax_rate_locations ( location_code, tax_rate_id, location_type ) VALUES " . implode( ',', $postcode_query ) );
434                 }
435 
436                 if ( ! empty( $city ) ) {
437                     $cities = explode( ';', $city );
438                     $cities = array_map( 'strtoupper', array_map( 'woocommerce_clean', $cities ) );
439                     foreach( $cities as $city ) {
440                         $wpdb->insert(
441                         $wpdb->prefix . "woocommerce_tax_rate_locations",
442                             array(
443                                 'location_code' => $city,
444                                 'tax_rate_id'   => $tax_rate_id,
445                                 'location_type' => 'city',
446                             )
447                         );
448                     }
449                 }

455         } else {
456

460                 $wpdb->query( $wpdb->prepare( "DELETE FROM {$wpdb->prefix}woocommerce_tax_rate_locations WHERE tax_rate_id = %d;", $tax_rate_id ) );
461                 $wpdb->query( $wpdb->prepare( "DELETE FROM {$wpdb->prefix}woocommerce_tax_rates WHERE tax_rate_id = %d;", $tax_rate_id ) );
462                 continue;

464

466             $country  = strtoupper( woocommerce_clean( $tax_rate_country[ $key ] ) );
467             $state    = strtoupper( woocommerce_clean( $tax_rate_state[ $key ] ) );
468             $rate     = number_format( woocommerce_clean( $tax_rate[ $key ] ), 4, '.', '' );
469             $name     = woocommerce_clean( $tax_rate_name[ $key ] );
470             $priority = absint( woocommerce_clean( $tax_rate_priority[ $key ] ) );
471             $compound = isset( $tax_rate_compound[ $key ] ) ? 1 : 0;
472             $shipping = isset( $tax_rate_shipping[ $key ] ) ? 1 : 0;
473 
474             if ( ! $name )
475                 $name = __( 'Tax', 'woocommerce' );
476 
477             if ( $country == '*' )
478                 $country = '';
479 
480             if ( $state == '*' )
481                 $state = '';
