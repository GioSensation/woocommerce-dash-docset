143     function is_available( $package ) {
144         global $woocommerce;
145 
146         if ($this->enabled=="no") return false;
147 
148         if ($this->availability=='including') :
149 
150             if (is_array($this->countries)) :
151                 if ( ! in_array( $package['destination']['country'], $this->countries) ) return false;
152             endif;
153 
154         else :
155 
156             if (is_array($this->countries)) :
157                 if ( in_array( $package['destination']['country'], $this->countries) ) return false;
158             endif;
159 
160         endif;
161 
162         return apply_filters( 'woocommerce_shipping_' . $this->id . '_is_available', true );
