119     /**
120      * Check If The Gateway Is Available For Use
121      *
122      * @access public
123      * @return bool
124      */
125     function is_available() {
126         global $woocommerce;
127 
128         if ( ! empty( $this->enable_for_methods ) ) {
129 
130             if ( is_page( woocommerce_get_page_id( 'pay' ) ) ) {
131 
132                 $order_id = (int) $_GET['order_id'];
133                 $order = new WC_Order( $order_id );
134 
135                 if ( ! $order->shipping_method )
136                     return false;
137 
138                 $chosen_method = $order->shipping_method;
139 
140             } elseif ( empty( $woocommerce->session->chosen_shipping_method ) ) {
141                 return false;
142             } else {
143                 $chosen_method = $woocommerce->session->chosen_shipping_method;
144             }
145 
146             $found = false;
147 
148             foreach ( $this->enable_for_methods as $method_id ) {
149                 if ( strpos( $chosen_method, $method_id ) === 0 ) {
150                     $found = true;
151                     break;
152                 }
153             }
154 
155             if ( ! $found )
156                 return false;
157         }
158 
159         return parent::is_available();
160     }

163     /**
164      * Process the payment and return the result
165      *
166      * @access public
167      * @param int $order_id
168      * @return array
169      */
170     function process_payment ($order_id) {
171         global $woocommerce;
172 
173         $order = new WC_Order( $order_id );
174 
175         // Mark as on-hold (we're awaiting the cheque)
176         $order->update_status('on-hold', __( 'Payment to be made upon delivery.', 'woocommerce' ));
177 
178         // Reduce stock levels
179         $order->reduce_order_stock();
180 
181         // Remove cart
182         $woocommerce->cart->empty_cart();
183 
184         // Return thankyou redirect
185         return array(
186             'result'    => 'success',
187             'redirect'  => add_query_arg('key', $order->order_key, add_query_arg('order', $order_id, get_permalink(woocommerce_get_page_id('thanks'))))
188         );
189     }
