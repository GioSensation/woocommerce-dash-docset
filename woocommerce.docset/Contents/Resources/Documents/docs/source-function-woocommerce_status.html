499 /**
500  * woocommerce_scan_template_files function.
501  *
502  * @access public
503  * @param mixed $template_path
504  * @return void
505  */
506 function woocommerce_scan_template_files( $template_path ) {
507     $files         = scandir( $template_path );
508     $result        = array();
509     if ( $files ) {
510         foreach ( $files as $key => $value ) {
511             if ( ! in_array( $value, array( ".",".." ) ) ) {
512                 if ( is_dir( $template_path . DIRECTORY_SEPARATOR . $value ) ) {
513                     $sub_files = woocommerce_scan_template_files( $template_path . DIRECTORY_SEPARATOR . $value );
514                     foreach ( $sub_files as $sub_file ) {
515                         $result[] = $value . DIRECTORY_SEPARATOR . $sub_file;
516                     }
517                 } else {
518                     $result[] = $value;
519                 }
520             }
521         }
522     }
523     return $result;
524 }
