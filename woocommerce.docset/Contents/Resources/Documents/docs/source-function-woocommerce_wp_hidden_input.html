108 /**
109  * Save meta boxes
110  *
111  * Runs when a post is saved and does an action which the write panel save scripts can hook into.
112  *
113  * @access public
114  * @param mixed $post_id
115  * @param mixed $post
116  * @return void
117  */
118 function woocommerce_meta_boxes_save( $post_id, $post ) {
119     if ( empty( $post_id ) || empty( $post ) ) return;
120     if ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) return;
121     if ( is_int( wp_is_post_revision( $post ) ) ) return;
122     if ( is_int( wp_is_post_autosave( $post ) ) ) return;
123     if ( empty( $_POST['woocommerce_meta_nonce'] ) || ! wp_verify_nonce( $_POST['woocommerce_meta_nonce'], 'woocommerce_save_data' ) ) return;
124     if ( !current_user_can( 'edit_post', $post_id )) return;
125     if ( $post->post_type != 'product' && $post->post_type != 'shop_order' && $post->post_type != 'shop_coupon' ) return;
126 
127     do_action( 'woocommerce_process_' . $post->post_type . '_meta', $post_id, $post );
128 
129     woocommerce_meta_boxes_save_errors();
130 }
131 
132 add_action( 'save_post', 'woocommerce_meta_boxes_save', 1, 2 );

240 /**
241  * Grant downloadable file access to any newly added files on any existing
242  * orders for this product that have previously been granted downloadable file access
243  *
244  * @access public
245  * @param int $product_id product identifier
246  * @param int $variation_id optional product variation identifier
247  * @param array $file_paths newly set file paths
248  */
249 function woocommerce_process_product_file_download_paths( $product_id, $variation_id, $file_paths ) {
250     global $wpdb;
251 
252     if ( $variation_id )
253         $product_id = $variation_id;
254 
255     // determine whether any new files have been added
256     $existing_file_paths = apply_filters( 'woocommerce_file_download_paths', get_post_meta( $product_id, '_file_paths', true ), $product_id, null, null );
257     if ( ! $existing_file_paths ) $existing_file_paths = array();
258     $new_download_ids = array_diff( array_keys( $file_paths ), array_keys( $existing_file_paths ) );
259 
260     if ( $new_download_ids ) {
261         // determine whether downloadable file access has been granted (either via the typical order completion, or via the admin ajax method)
262         $existing_permissions = $wpdb->get_results( $wpdb->prepare( "SELECT * from {$wpdb->prefix}woocommerce_downloadable_product_permissions WHERE product_id = %d GROUP BY order_id", $product_id ) );
263         foreach ( $existing_permissions as $existing_permission ) {
264             $order = new WC_Order( $existing_permission->order_id );
265 
266             if ( $order->id ) {
267                 foreach ( $new_download_ids as $new_download_id ) {
268                     // grant permission if it doesn't already exist
269                     if ( ! $wpdb->get_var( $wpdb->prepare( "SELECT true FROM {$wpdb->prefix}woocommerce_downloadable_product_permissions WHERE order_id = %d AND product_id = %d AND download_id = %s", $order->id, $product_id, $new_download_id ) ) ) {
270                         woocommerce_downloadable_file_permission( $new_download_id, $product_id, $order );
271                     }
272                 }
273             }
274         }
275     }
276 }

322 /**
323  * Output a text input box.
324  *
325  * @access public
326  * @param array $field
327  * @return void
328  */
329 function woocommerce_wp_text_input( $field ) {
330     global $thepostid, $post, $woocommerce;
331 
332     $thepostid              = empty( $thepostid ) ? $post->ID : $thepostid;
333     $field['placeholder']   = isset( $field['placeholder'] ) ? $field['placeholder'] : '';
334     $field['class']         = isset( $field['class'] ) ? $field['class'] : 'short';
335     $field['wrapper_class'] = isset( $field['wrapper_class'] ) ? $field['wrapper_class'] : '';
336     $field['value']         = isset( $field['value'] ) ? $field['value'] : get_post_meta( $thepostid, $field['id'], true );
337     $field['name']          = isset( $field['name'] ) ? $field['name'] : $field['id'];
338     $field['type']          = isset( $field['type'] ) ? $field['type'] : 'text';
339 
340     // Custom attribute handling
341     $custom_attributes = array();
342 
343     if ( ! empty( $field['custom_attributes'] ) && is_array( $field['custom_attributes'] ) )
344         foreach ( $field['custom_attributes'] as $attribute => $value )
345             $custom_attributes[] = esc_attr( $attribute ) . '="' . esc_attr( $value ) . '"';
346 
347     echo '<p class="form-field ' . esc_attr( $field['id'] ) . '_field ' . esc_attr( $field['wrapper_class'] ) . '"><label for="' . esc_attr( $field['id'] ) . '">' . wp_kses_post( $field['label'] ) . '</label><input type="' . esc_attr( $field['type'] ) . '" class="' . esc_attr( $field['class'] ) . '" name="' . esc_attr( $field['name'] ) . '" id="' . esc_attr( $field['id'] ) . '" value="' . esc_attr( $field['value'] ) . '" placeholder="' . esc_attr( $field['placeholder'] ) . '" ' . implode( ' ', $custom_attributes ) . ' /> ';
348 
349     if ( ! empty( $field['description'] ) ) {
350 
351         if ( isset( $field['desc_tip'] ) ) {
352             echo '<img class="help_tip" data-tip="' . esc_attr( $field['description'] ) . '" src="' . $woocommerce->plugin_url() . '/assets/images/help.png" height="16" width="16" />';
353         } else {
354             echo '<span class="description">' . wp_kses_post( $field['description'] ) . '</span>';
355         }
356 
357     }
358     echo '</p>';
359 }
