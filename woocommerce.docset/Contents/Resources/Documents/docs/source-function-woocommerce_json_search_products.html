 405 function woocommerce_save_attributes() {
 406     global $woocommerce;
 407 
 408     check_ajax_referer( 'save-attributes', 'security' );
 409 
 410     // Get post data
 411     parse_str( $_POST['data'], $data );
 412     $post_id = absint( $_POST['post_id'] );
 413 
 414     // Save Attributes
 415     $attributes = array();
 416 
 417     if ( isset( $data['attribute_names'] ) ) {
 418 
 419         $attribute_names  = array_map( 'stripslashes', $data['attribute_names'] );
 420         $attribute_values = $data['attribute_values'];
 421 
 422         if ( isset( $data['attribute_visibility'] ) )
 423             $attribute_visibility = $data['attribute_visibility'];
 424 
 425         if ( isset( $data['attribute_variation'] ) )
 426             $attribute_variation = $data['attribute_variation'];
 427 
 428         $attribute_is_taxonomy = $data['attribute_is_taxonomy'];
 429         $attribute_position = $data['attribute_position'];
 430 
 431         $attribute_names_count = sizeof( $attribute_names );
 432 
 433         for ( $i=0; $i < $attribute_names_count; $i++ ) {
 434             if ( ! $attribute_names[ $i ] )
 435                 continue;
 436 
 437             $is_visible     = isset( $attribute_visibility[ $i ] ) ? 1 : 0;
 438             $is_variation   = isset( $attribute_variation[ $i ] ) ? 1 : 0;
 439             $is_taxonomy    = $attribute_is_taxonomy[ $i ] ? 1 : 0;
 440 
 441             if ( $is_taxonomy ) {
 442 
 443                 if ( isset( $attribute_values[ $i ] ) ) {
 444 
 445                     // Format values (slug format)
 446                     if ( is_array( $attribute_values[ $i ] ) ) {
 447                         $values = array_map( 'sanitize_title', array_map( 'stripslashes', $attribute_values[ $i ] ) );
 448                     } else {
 449                         // Text based, separate by pipe
 450                         $values = array_map( 'sanitize_title', array_map( 'stripslashes', explode( '|', $attribute_values[ $i ] ) ) );
 451                     }
 452 
 453                     // Remove empty items in the array
 454                     $values = array_filter( $values );
 455 
 456                 } else {
 457                     $values = array();
 458                 }
 459 
 460                 // Update post terms
 461                 if ( taxonomy_exists( $attribute_names[ $i ] ) )
 462                     wp_set_object_terms( $post_id, $values, $attribute_names[ $i ] );
 463 
 464                 if ( $values ) {
 465                     // Add attribute to array, but don't set values
 466                     $attributes[ sanitize_title( $attribute_names[ $i ] ) ] = array(
 467                         'name'          => woocommerce_clean( $attribute_names[ $i ] ),
 468                         'value'         => '',
 469                         'position'      => $attribute_position[ $i ],
 470                         'is_visible'    => $is_visible,
 471                         'is_variation'  => $is_variation,
 472                         'is_taxonomy'   => $is_taxonomy
 473                     );
 474                 }
 475 
 476             } elseif ( isset( $attribute_values[ $i ] ) ) {
 477 
 478                 // Text based, separate by pipe
 479                 $values = implode( ' | ', array_map( 'woocommerce_clean', array_map( 'stripslashes', explode( '|', $attribute_values[ $i ] ) ) ) );
 480 
 481                 // Custom attribute - Add attribute to array and set the values
 482                 $attributes[ sanitize_title( $attribute_names[ $i ] ) ] = array(
 483                     'name'          => woocommerce_clean( $attribute_names[ $i ] ),
 484                     'value'         => $values,
 485                     'position'      => $attribute_position[ $i ],
 486                     'is_visible'    => $is_visible,
 487                     'is_variation'  => $is_variation,
 488                     'is_taxonomy'   => $is_taxonomy
 489                 );
 490             }
 491 
 492          }
 493     }
 494 
 495     if ( ! function_exists( 'attributes_cmp' ) ) {
 496         function attributes_cmp( $a, $b ) {
 497             if ( $a['position'] == $b['position'] ) return 0;
 498             return ( $a['position'] < $b['position'] ) ? -1 : 1;
 499         }
 500     }
 501     uasort( $attributes, 'attributes_cmp' );
 502 
 503     update_post_meta( $post_id, '_product_attributes', $attributes );
 504 
 505     die();
 506 }

 592 /**
 593  * Link all variations via ajax function
 594  *
 595  * @access public
 596  * @return void
 597  */
 598 function woocommerce_link_all_variations() {
 599     global $woocommerce;
 600 
 601     if ( ! defined( 'WC_MAX_LINKED_VARIATIONS' ) ) {
 602         define( 'WC_MAX_LINKED_VARIATIONS', 49 );
 603     }
 604 
 605     check_ajax_referer( 'link-variations', 'security' );
 606 
 607     @set_time_limit(0);
 608 
 609     $post_id = intval( $_POST['post_id'] );
 610 
 611     if ( ! $post_id ) die();
 612 
 613     $variations = array();
 614 
 615     $_product = get_product( $post_id, array( 'product_type' => 'variable' ) );
 616 
 617     // Put variation attributes into an array
 618     foreach ( $_product->get_attributes() as $attribute ) {
 619 
 620         if ( ! $attribute['is_variation'] ) continue;
 621 
 622         $attribute_field_name = 'attribute_' . sanitize_title( $attribute['name'] );
 623 
 624         if ( $attribute['is_taxonomy'] ) {
 625             $post_terms = wp_get_post_terms( $post_id, $attribute['name'] );
 626             $options = array();
 627             foreach ( $post_terms as $term ) {
 628                 $options[] = $term->slug;
 629             }
 630         } else {
 631             $options = explode( '|', $attribute['value'] );
 632         }
 633 
 634         $options = array_map( 'sanitize_title', array_map( 'trim', $options ) );
 635 
 636         $variations[ $attribute_field_name ] = $options;
 637     }
 638 
 639     // Quit out if none were found
 640     if ( sizeof( $variations ) == 0 ) die();
 641 
 642     // Get existing variations so we don't create duplicates
 643     $available_variations = array();
 644 
 645     foreach( $_product->get_children() as $child_id ) {
 646         $child = $_product->get_child( $child_id );
 647 
 648         if ( ! empty( $child->variation_id ) ) {
 649             $available_variations[] = $child->get_variation_attributes();
 650         }
 651     }
 652 
 653     // Created posts will all have the following data
 654     $variation_post_data = array(
 655         'post_title' => 'Product #' . $post_id . ' Variation',
 656         'post_content' => '',
 657         'post_status' => 'publish',
 658         'post_author' => get_current_user_id(),
 659         'post_parent' => $post_id,
 660         'post_type' => 'product_variation'
 661     );
 662 
 663     // Now find all combinations and create posts
 664     if ( ! function_exists( 'array_cartesian' ) ) {
 665         function array_cartesian( $input ) {
 666             $result = array();
 667 
 668             while ( list( $key, $values ) = each( $input ) ) {
 669                 // If a sub-array is empty, it doesn't affect the cartesian product
 670                 if ( empty( $values ) ) {
 671                     continue;
 672                 }
 673 
 674                 // Special case: seeding the product array with the values from the first sub-array
 675                 if ( empty( $result ) ) {
 676                     foreach ( $values as $value ) {
 677                         $result[] = array( $key => $value );
 678                     }
 679                 }
 680                 else {
 681                     // Second and subsequent input sub-arrays work like this:
 682                     //   1. In each existing array inside $product, add an item with
 683                     //      key == $key and value == first item in input sub-array
 684                     //   2. Then, for each remaining item in current input sub-array,
 685                     //      add a copy of each existing array inside $product with
 686                     //      key == $key and value == first item in current input sub-array
 687 
 688                     // Store all items to be added to $product here; adding them on the spot
 689                     // inside the foreach will result in an infinite loop
 690                     $append = array();
 691                     foreach( $result as &$product ) {
 692                         // Do step 1 above. array_shift is not the most efficient, but it
 693                         // allows us to iterate over the rest of the items with a simple
 694                         // foreach, making the code short and familiar.
 695                         $product[ $key ] = array_shift( $values );
 696 
 697                         // $product is by reference (that's why the key we added above
 698                         // will appear in the end result), so make a copy of it here
 699                         $copy = $product;
 700 
 701                         // Do step 2 above.
 702                         foreach( $values as $item ) {
 703                             $copy[ $key ] = $item;
 704                             $append[] = $copy;
 705                         }
 706 
 707                         // Undo the side effecst of array_shift
 708                         array_unshift( $values, $product[ $key ] );
 709                     }
 710 
 711                     // Out of the foreach, we can add to $results now
 712                     $result = array_merge( $result, $append );
 713                 }
 714             }
 715 
 716             return $result;
 717         }
 718     }
 719 
 720     $variation_ids = array();
 721     $added = 0;
 722     $possible_variations = array_cartesian( $variations );
 723 
 724     foreach ( $possible_variations as $variation ) {
 725 
 726         // Check if variation already exists
 727         if ( in_array( $variation, $available_variations ) )
 728             continue;
 729 
 730         $variation_id = wp_insert_post( $variation_post_data );
 731 
 732         $variation_ids[] = $variation_id;
 733 
 734         foreach ( $variation as $key => $value ) {
 735             update_post_meta( $variation_id, $key, $value );
 736         }
 737 
 738         $added++;
 739 
 740         do_action( 'product_variation_linked', $variation_id );
 741 
 742         if ( $added > WC_MAX_LINKED_VARIATIONS )
 743             break;
 744     }
 745 
 746     $woocommerce->clear_product_transients( $post_id );
 747 
 748     echo $added;
 749 
 750     die();
 751 }
