 213

 215                 include_once( $path . $file );

 582     /**
 583      * Load a template.
 584      *
 585      * Handles template usage so that we can use our own templates instead of the themes.
 586      *
 587      * Templates are in the 'templates' folder. woocommerce looks for theme
 588      * overrides in /theme/woocommerce/ by default
 589      *
 590      * For beginners, it also looks for a woocommerce.php template first. If the user adds
 591      * this to the theme (containing a woocommerce() inside) this will be used for all
 592      * woocommerce templates.
 593      *
 594      * @access public
 595      * @param mixed $template
 596      * @return string
 597      */
 598     public function template_loader( $template ) {
 599 
 600         $find = array( 'woocommerce.php' );
 601         $file = '';
 602 
 603         if ( is_single() && get_post_type() == 'product' ) {
 604 
 605             $file   = 'single-product.php';
 606             $find[] = $file;
 607             $find[] = $this->template_url . $file;
 608 
 609         } elseif ( is_tax( 'product_cat' ) || is_tax( 'product_tag' ) ) {
 610 
 611             $term = get_queried_object();
 612 
 613             $file       = 'taxonomy-' . $term->taxonomy . '.php';
 614             $find[]     = 'taxonomy-' . $term->taxonomy . '-' . $term->slug . '.php';
 615             $find[]     = $this->template_url . 'taxonomy-' . $term->taxonomy . '-' . $term->slug . '.php';
 616             $find[]     = $file;
 617             $find[]     = $this->template_url . $file;
 618 
 619         } elseif ( is_post_type_archive( 'product' ) || is_page( woocommerce_get_page_id( 'shop' ) ) ) {
 620 
 621             $file   = 'archive-product.php';
 622             $find[] = $file;
 623             $find[] = $this->template_url . $file;
 624 
 625         }
 626 
 627         if ( $file ) {
 628             $template = locate_template( $find );
 629             if ( ! $template ) $template = $this->plugin_path() . '/templates/' . $file;
 630         }
 631 
 632         return $template;
 633     }

 683     /**
 684      * Add Compatibility for various bits.
 685      *
 686      * @access public
 687      * @return void
 688      */
 689     public function compatibility() {
 690         // Post thumbnail support
 691         if ( ! current_theme_supports( 'post-thumbnails', 'product' ) ) {
 692             add_theme_support( 'post-thumbnails' );
 693             remove_post_type_support( 'post', 'thumbnail' );
 694             remove_post_type_support( 'page', 'thumbnail' );
 695         } else {
 696             add_post_type_support( 'product', 'thumbnail' );
 697         }
 698 
 699         // IIS
 700         if ( ! isset($_SERVER['REQUEST_URI'] ) ) {
 701             $_SERVER['REQUEST_URI'] = substr( $_SERVER['PHP_SELF'], 1 );
 702             if ( isset( $_SERVER['QUERY_STRING'] ) )
 703                 $_SERVER['REQUEST_URI'].='?'.$_SERVER['QUERY_STRING'];
 704         }
 705 
 706         // NGINX Proxy
 707         if ( ! isset( $_SERVER['REMOTE_ADDR'] ) && isset( $_SERVER['HTTP_REMOTE_ADDR'] ) )
 708             $_SERVER['REMOTE_ADDR'] = $_SERVER['HTTP_REMOTE_ADDR'];
 709 
 710         if ( ! isset( $_SERVER['HTTPS'] ) && ! empty( $_SERVER['HTTP_HTTPS'] ) )
 711             $_SERVER['HTTPS'] = $_SERVER['HTTP_HTTPS'];
 712 
 713         // Support for hosts which don't use HTTPS, and use HTTP_X_FORWARDED_PROTO
 714         if ( ! isset( $_SERVER['HTTPS'] ) && ! empty( $_SERVER['HTTP_X_FORWARDED_PROTO'] ) && $_SERVER['HTTP_X_FORWARDED_PROTO'] == 'https' )
 715             $_SERVER['HTTPS'] = '1';
 716     }
