 35     /**
 36      * Get total stock.
 37      *
 38      * This is the stock of parent and children combined.
 39      *
 40      * @access public
 41      * @return int
 42      */
 43     public function get_total_stock() {
 44 
 45         if ( empty( $this->total_stock ) ) {
 46 
 47             $transient_name = 'wc_product_total_stock_' . $this->id;
 48 
 49             if ( false === ( $this->total_stock = get_transient( $transient_name ) ) ) {
 50                 $this->total_stock = $this->stock;
 51 
 52                 if ( sizeof( $this->get_children() ) > 0 ) {
 53                     foreach ($this->get_children() as $child_id) {
 54                         $stock = get_post_meta( $child_id, '_stock', true );
 55 
 56                         if ( $stock != '' ) {
 57                             $this->total_stock += intval( $stock );
 58                         }
 59                     }
 60                 }
 61 
 62                 set_transient( $transient_name, $this->total_stock );
 63             }
 64         }
 65 
 66         return apply_filters( 'woocommerce_stock_amount', $this->total_stock );
 67     }

 69     /**
 70      * Return the products children posts.
 71      *
 72      * @access public
 73      * @return array
 74      */
 75     public function get_children() {
 76 
 77         if ( ! is_array( $this->children ) ) {
 78 
 79             $this->children = array();
 80 
 81             $transient_name = 'wc_product_children_ids_' . $this->id;
 82 
 83             if ( false === ( $this->children = get_transient( $transient_name ) ) ) {
 84 
 85                 $this->children = get_posts( 'post_parent=' . $this->id . '&post_type=product&orderby=menu_order&order=ASC&fields=ids&post_status=any&numberposts=-1' );
 86 
 87                 set_transient( $transient_name, $this->children );
 88 
 89             }
 90         }
 91 
 92         return (array) $this->children;
 93     }

119     /**
120      * Returns whether or not the product is on sale.
121      *
122      * @access public
123      * @return bool
124      */
125     public function is_on_sale() {
126         if ( $this->has_child() ) {
127 
128             foreach ( $this->get_children() as $child_id ) {
129                 $sale_price = get_post_meta( $child_id, '_sale_price', true );
130                 if ( $sale_price !== "" && $sale_price >= 0 )
131                     return true;
132             }
133 
134         } else {
135 
136             if ( $this->sale_price && $this->sale_price == $this->price )
137                 return true;
138 
139         }
140         return false;
141     }

160      * @return string

162     public function get_price_html( $price = '' ) {
163 
164         $child_prices = array();
165 
166         foreach ( $this->get_children() as $child_id )
167             $child_prices[] = get_post_meta( $child_id, '_price', true );
168 
169         $child_prices = array_unique( $child_prices );
170 
171         if ( ! empty( $child_prices ) ) {
172             $min_price = min( $child_prices );
173         } else {
174             $min_price = '';
175         }
176 
177         if ( sizeof( $child_prices ) > 1 ) $price .= $this->get_price_html_from_text();
178 
179         $price .= woocommerce_price( $min_price );
180 
181         $price = apply_filters( 'woocommerce_grouped_price_html', $price, $this );
182 
183         return apply_filters( 'woocommerce_get_price_html', $price, $this );
