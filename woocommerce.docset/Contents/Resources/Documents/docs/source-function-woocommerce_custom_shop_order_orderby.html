417 /**
418  * Search custom fields as well as content.
419  *
420  * @access public
421  * @param mixed $wp
422  * @return void
423  */
424 function woocommerce_shop_order_search_custom_fields( $wp ) {
425     global $pagenow, $wpdb;
426 
427     if ( 'edit.php' != $pagenow || empty( $wp->query_vars['s'] ) || $wp->query_vars['post_type'] != 'shop_order' )
428         return $wp;
429 
430     $search_fields = array_map( 'esc_attr', apply_filters( 'woocommerce_shop_order_search_fields', array(
431         '_order_key',
432         '_billing_first_name',
433         '_billing_last_name',
434         '_billing_company',
435         '_billing_address_1',
436         '_billing_address_2',
437         '_billing_city',
438         '_billing_postcode',
439         '_billing_country',
440         '_billing_state',
441         '_billing_email',
442         '_billing_phone'
443     ) ) );
444 
445     $search_order_id = str_replace( 'Order #', '', $_GET['s'] );
446     if ( ! is_numeric( $search_order_id ) )
447         $search_order_id = 0;
448 
449     // Search orders
450     $post_ids = array_merge(
451         $wpdb->get_col(
452             $wpdb->prepare( "
453                 SELECT post_id
454                 FROM {$wpdb->postmeta}
455                 WHERE meta_key IN ('" . implode( "','", $search_fields ) . "')
456                 AND meta_value LIKE '%%%s%%'",
457                 esc_attr( $_GET['s'] )
458             )
459         ),
460         $wpdb->get_col(
461             $wpdb->prepare( "
462                 SELECT order_id
463                 FROM {$wpdb->prefix}woocommerce_order_items as order_items
464                 WHERE order_item_name LIKE '%%%s%%'
465                 ",
466                 esc_attr( $_GET['s'] )
467             )
468         ),
469         $wpdb->get_col(
470             $wpdb->prepare( "
471                 SELECT posts.ID
472                 FROM {$wpdb->posts} as posts
473                 LEFT JOIN {$wpdb->postmeta} as postmeta ON posts.ID = postmeta.post_id
474                 LEFT JOIN {$wpdb->users} as users ON postmeta.meta_value = users.ID
475                 WHERE
476                     post_excerpt    LIKE '%%%1\$s%%' OR
477                     post_title      LIKE '%%%1\$s%%' OR
478                     (
479                         meta_key        = '_customer_user' AND
480                         (
481                             user_login      LIKE '%%%1\$s%%' OR
482                             user_nicename   LIKE '%%%1\$s%%' OR
483                             user_email      LIKE '%%%1\$s%%' OR
484                             display_name    LIKE '%%%1\$s%%'
485                         )
486                     )
487                 ",
488                 esc_attr( $_GET['s'] )
489             )
490         ),
491         array( $search_order_id )
492     );
493 
494     // Remove s - we don't want to search order name
495     unset( $wp->query_vars['s'] );
496 
497     // so we know we're doing this
498     $wp->query_vars['shop_order_search'] = true;
499 
500     // Search by found posts
501     $wp->query_vars['post__in'] = $post_ids;
502 }
