 283     /**
 284      * Get file download path identified by $download_id
 285      *
 286      * @access public
 287      * @param string $download_id file identifier
 288      * @return array
 289      */
 290     function get_file_download_path( $download_id ) {
 291 
 292         $file_paths = isset( $this->file_paths ) ? $this->file_paths : '';
 293         $file_paths = maybe_unserialize( $file_paths );
 294         $file_paths = apply_filters( 'woocommerce_file_download_paths', $file_paths, $this->id, null, null );
 295 
 296         if ( ! $download_id && count( $file_paths ) == 1 ) {
 297             // backwards compatibility for old-style download URLs and template files
 298             $file_path = array_shift( $file_paths );
 299         } elseif ( isset( $file_paths[ $download_id ] ) ) {
 300             $file_path = $file_paths[ $download_id ];
 301         } else {
 302             $file_path = '';
 303         }
 304 
 305         // allow overriding based on the particular file being requested
 306         return apply_filters( 'woocommerce_file_download_path', $file_path, $this->id, $download_id );
 307     }

 526     /**
 527      * Returns the availability of the product.
 528      *
 529      * @access public
 530      * @return string
 531      */
 532     function get_availability() {
 533 
 534         $availability = $class = "";
 535 
 536         if ( $this->managing_stock() ) {
 537             if ( $this->is_in_stock() ) {
 538 
 539                 if ( $this->get_total_stock() > 0 ) {
 540 
 541                     $format_option = get_option( 'woocommerce_stock_format' );
 542 
 543                     switch ( $format_option ) {
 544                         case 'no_amount' :
 545                             $format = __( 'In stock', 'woocommerce' );
 546                         break;
 547                         case 'low_amount' :
 548                             $low_amount = get_option( 'woocommerce_notify_low_stock_amount' );
 549 
 550                             $format = ( $this->get_total_stock() <= $low_amount ) ? __( 'Only %s left in stock', 'woocommerce' ) : __( 'In stock', 'woocommerce' );
 551                         break;
 552                         default :
 553                             $format = __( '%s in stock', 'woocommerce' );
 554                         break;
 555                     }
 556 
 557                     $availability = sprintf( $format, $this->stock );
 558 
 559                     if ( $this->backorders_allowed() && $this->backorders_require_notification() )
 560                         $availability .= ' ' . __( '(backorders allowed)', 'woocommerce' );
 561 
 562                 } else {
 563 
 564                     if ( $this->backorders_allowed() ) {
 565                         if ( $this->backorders_require_notification() ) {
 566                             $availability = __( 'Available on backorder', 'woocommerce' );
 567                             $class        = 'available-on-backorder';
 568                         } else {
 569                             $availability = __( 'In stock', 'woocommerce' );
 570                         }
 571                     } else {
 572                         $availability = __( 'Out of stock', 'woocommerce' );
 573                         $class        = 'out-of-stock';
 574                     }
 575 
 576                 }
 577 
 578             } elseif ( $this->backorders_allowed() ) {
 579                 $availability = __( 'Available on backorder', 'woocommerce' );
 580                 $class        = 'available-on-backorder';
 581             } else {
 582                 $availability = __( 'Out of stock', 'woocommerce' );
 583                 $class        = 'out-of-stock';
 584             }
 585         } elseif ( ! $this->is_in_stock() ) {
 586             $availability = __( 'Out of stock', 'woocommerce' );
 587             $class        = 'out-of-stock';
 588         }
 589 
 590         return apply_filters( 'woocommerce_get_availability', array( 'availability' => $availability, 'class' => $class ), $this );
 591     }

 605     /**
 606      * Returns whether or not the product is visible.
 607      *
 608      * @access public
 609      * @return bool
 610      */
 611     function is_visible() {
 612 
 613         $visible = true;
 614 
 615         // Out of stock visibility
 616         if ( get_option( 'woocommerce_hide_out_of_stock_items' ) == 'yes' && ! $this->is_in_stock() ) $visible = false;
 617 
 618         // visibility setting
 619         elseif ( $this->visibility == 'hidden' ) $visible = false;
 620         elseif ( $this->visibility == 'visible' ) $visible = true;
 621 
 622         // Visibility in loop
 623         elseif ( $this->visibility == 'search' && is_search() ) $visible = true;
 624         elseif ( $this->visibility == 'search' && ! is_search() ) $visible = false;
 625         elseif ( $this->visibility == 'catalog' && is_search() ) $visible = false;
 626         elseif ( $this->visibility == 'catalog' && ! is_search() ) $visible = true;
 627 
 628         return apply_filters( 'woocommerce_product_is_visible', $visible, $this->id );
 629     }

 711     /**
 712      * Returns the price (including tax). Uses customer tax rates. Can work for a specific $qty for more accurate taxes.
 713      *
 714      * @access public
 715      * @return string
 716      */
 717     function get_price_including_tax( $qty = 1 ) {
 718         global $woocommerce;
 719 
 720         $_tax  = new WC_Tax();
 721         $price = $this->get_price();
 722 
 723         if ( $this->is_taxable() ) {
 724 
 725             if ( get_option('woocommerce_prices_include_tax') == 'no' ) {
 726 
 727                 $tax_rates  = $_tax->get_rates( $this->get_tax_class() );
 728                 $taxes      = $_tax->calc_tax( $price * $qty, $tax_rates, false );
 729                 $tax_amount = $_tax->get_tax_total( $taxes );
 730                 $price      = round( $price * $qty + $tax_amount, 2 );
 731
