  3  * WooCommerce Payment Gateways class

  5  * Loads payment gateways via hooks for use in the store.
  6  *
  7  * @class       WC_Payment_Gateways
  8  * @version     1.6.4
  9  * @package     WooCommerce/Classes/Payment
 10  * @category    Class
 11  * @author      WooThemes
 12  */
 13 class WC_Payment_Gateways {
 14 
 15     /** @var array Array of payment gateway classes. */
 16     var $payment_gateways;
 17 
 18     /**
 19      * __construct function.
 20      *
 21      * @access public
 22      * @return void
 23      */
 24     public function __construct() {
 25         $this->init();

 28     /**
 29      * Load gateways and hook in functions.
 30      *
 31      * @access public
 32      * @return void
 33      */
 34     function init() {
 35 
 36         $load_gateways = apply_filters( 'woocommerce_payment_gateways', array() );
 37 
 38         // Get order option
 39         $ordering   = (array) get_option('woocommerce_gateway_order');
 40         $order_end  = 999;
 41 
 42         // Load gateways in order
 43         foreach ($load_gateways as $gateway) :
 44 
 45             $load_gateway = new $gateway();
 46 
 47             if (isset($ordering[$load_gateway->id]) && is_numeric($ordering[$load_gateway->id])) :
 48                 // Add in position
 49                 $this->payment_gateways[$ordering[$load_gateway->id]] = $load_gateway;
 50             else :
 51                 // Add to end of the array
 52                 $this->payment_gateways[$order_end] = $load_gateway;
 53                 $order_end++;
 54             endif;
 55 
 56         endforeach;
 57 
 58         ksort( $this->payment_gateways );
 59     }
 60 
 61 
 62     /**
 63      * Get gateways.
 64      *
 65      * @access public
 66      * @return array
 67      */
 68     function payment_gateways() {
 69 
 70         $_available_gateways = array();
 71 
 72         if ( sizeof( $this->payment_gateways ) > 0 )
 73             foreach ( $this->payment_gateways as $gateway )
 74                 $_available_gateways[ $gateway->id ] = $gateway;
 75 
 76         return $_available_gateways;
 77     }

 80     /**
 81      * Get available gateways.
 82      *
 83      * @access public
 84      * @return array
 85      */
 86     function get_available_payment_gateways() {
 87 
 88         $_available_gateways = array();
 89 
 90         foreach ( $this->payment_gateways as $gateway ) :
 91 
 92             if ($gateway->is_available()) $_available_gateways[$gateway->id] = $gateway;
 93 
 94         endforeach;
 95 
 96         return apply_filters( 'woocommerce_available_payment_gateways', $_available_gateways );
 97     }

100     /**
101      * Save options in admin.
102      *
103      * @access public
104      * @return void
105      */
106     function process_admin_options() {
107 
108         $default_gateway = ( isset( $_POST['default_gateway'] ) ) ? esc_attr( $_POST['default_gateway'] ) : '';
109         $gateway_order = ( isset( $_POST['gateway_order'] ) ) ? $_POST['gateway_order'] : '';
110 
111         $order = array();
112 
113         if ( is_array( $gateway_order ) && sizeof( $gateway_order ) > 0 ) {
114             $loop = 0;
115             foreach ( $gateway_order as $gateway_id ) {
116                 $order[ esc_attr( $gateway_id ) ] = $loop;
117                 $loop++;
118             }
119         }
120 
121         update_option( 'woocommerce_default_gateway', $default_gateway );
122         update_option( 'woocommerce_gateway_order', $order );
123     }
