  37 function woocommerce_template_redirect() {
  38     global $woocommerce, $wp_query;
  39 
  40     // When default permalinks are enabled, redirect shop page to post type archive url
  41     if ( ! empty( $_GET['page_id'] ) && get_option( 'permalink_structure' ) == "" && $_GET['page_id'] == woocommerce_get_page_id( 'shop' ) ) {
  42         wp_safe_redirect( get_post_type_archive_link('product') );
  43         exit;
  44     }
  45 
  46     // When on the checkout with an empty cart, redirect to cart page
  47     elseif ( is_page( woocommerce_get_page_id( 'checkout' ) ) && sizeof( $woocommerce->cart->get_cart() ) == 0 ) {
  48         wp_redirect( get_permalink( woocommerce_get_page_id( 'cart' ) ) );
  49         exit;
  50     }
  51 
  52     // When on pay page with no query string, redirect to checkout
  53     elseif ( is_page( woocommerce_get_page_id( 'pay' ) ) && ! isset( $_GET['order'] ) ) {
  54         wp_redirect( get_permalink( woocommerce_get_page_id( 'checkout' ) ) );
  55         exit;
  56     }
  57 
  58     // My account page redirects (logged out)
  59     elseif ( ! is_user_logged_in() && ( is_page( woocommerce_get_page_id( 'edit_address' ) ) || is_page( woocommerce_get_page_id( 'view_order' ) ) || is_page( woocommerce_get_page_id( 'change_password' ) ) ) ) {
  60         wp_redirect( get_permalink( woocommerce_get_page_id( 'myaccount' ) ) );
  61         exit;
  62     }
  63 
  64     // Logout
  65     elseif ( is_page( woocommerce_get_page_id( 'logout' ) ) ) {
  66         wp_redirect( str_replace( '&amp;', '&', wp_logout_url( get_permalink( woocommerce_get_page_id( 'myaccount' ) ) ) ) );
  67         exit;
  68     }
  69 
  70     // Redirect to the product page if we have a single product
  71     elseif ( is_search() && is_post_type_archive( 'product' ) && apply_filters( 'woocommerce_redirect_single_search_result', true ) && $wp_query->post_count == 1 ) {
  72         $product = get_product( $wp_query->post );
  73 
  74         if ( $product->is_visible() ) {
  75             wp_safe_redirect( get_permalink( $product->id ), 302 );
  76             exit;
  77         }
  78     }
  79 
  80     // Force SSL
  81     elseif ( get_option('woocommerce_force_ssl_checkout') == 'yes' && ! is_ssl() ) {
  82 
  83         if ( is_checkout() || is_account_page() || apply_filters( 'woocommerce_force_ssl_checkout', false ) ) {
  84             if ( 0 === strpos( $_SERVER['REQUEST_URI'], 'http' ) ) {
  85                 wp_safe_redirect( preg_replace( '|^http://|', 'https://', $_SERVER['REQUEST_URI'] ) );
  86                 exit;
  87             } else {

  89                 exit;
  90             }
  91         }
  92 
  93     }
  94

  96     elseif ( get_option('woocommerce_force_ssl_checkout') == 'yes' && get_option('woocommerce_unforce_ssl_checkout') == 'yes' && is_ssl() && $_SERVER['REQUEST_URI'] && ! is_checkout() && ! is_page( woocommerce_get_page_id('thanks') ) && ! is_ajax() && ! is_account_page() && apply_filters( 'woocommerce_unforce_ssl_checkout', true ) ) {
  97 
  98         if ( 0 === strpos( $_SERVER['REQUEST_URI'], 'http' ) ) {
  99             wp_safe_redirect( preg_replace( '|^https://|', 'http://', $_SERVER['REQUEST_URI'] ) );
 100             exit;

 102             wp_safe_redirect( 'http://' . $_SERVER['HTTP_HOST'] . $_SERVER['REQUEST_URI'] );
 103             exit;
 104         }

 110         ob_start();

 143  * Fix active class in nav for shop page.
 144  *
 145  * @access public
 146  * @param array $menu_items
 147  * @param array $args
 148  * @return array
 149  */
 150 function woocommerce_nav_menu_item_classes( $menu_items, $args ) {
 151 
 152     if ( ! is_woocommerce() ) return $menu_items;
 153 
 154     $shop_page      = (int) woocommerce_get_page_id('shop');
 155     $page_for_posts = (int) get_option( 'page_for_posts' );
 156 
 157     foreach ( (array) $menu_items as $key => $menu_item ) {
 158 
 159         $classes = (array) $menu_item->classes;
 160 
 161         // Unset active class for blog page
 162         if ( $page_for_posts == $menu_item->object_id ) {
 163             $menu_items[$key]->current = false;
 164             unset( $classes[ array_search('current_page_parent', $classes) ] );
 165             unset( $classes[ array_search('current-menu-item', $classes) ] );
 166 
 167         // Set active state if this is the shop page link
 168         } elseif ( is_shop() && $shop_page == $menu_item->object_id ) {
 169             $menu_items[$key]->current = true;
 170             $classes[] = 'current-menu-item';
 171             $classes[] = 'current_page_item';
 172 
 173         // Set parent state if this is a product page
 174         } elseif ( is_singular( 'product' ) && $shop_page == $menu_item->object_id ) {
 175             $classes[] = 'current_page_parent';
 176         }
 177 
 178         $menu_items[$key]->classes = array_unique( $classes );
 179 
 180     }
 181 
 182     return $menu_items;
 183 }

 212 /**
 213  * Remove from cart/update.
 214  *
 215  * @access public
 216  * @return void
 217  */
 218 function woocommerce_update_cart_action() {
 219     global $woocommerce;
 220 
 221     // Remove from cart
 222     if ( isset($_GET['remove_item']) && $_GET['remove_item'] && $woocommerce->verify_nonce('cart', '_GET')) {
 223 
 224         $woocommerce->cart->set_quantity( $_GET['remove_item'], 0 );
 225 
 226         $woocommerce->add_message( __( 'Cart updated.', 'woocommerce' ) );
 227 
 228         $referer = ( wp_get_referer() ) ? wp_get_referer() : $woocommerce->cart->get_cart_url();
 229         wp_safe_redirect( $referer );
 230         exit;
 231 
 232     // Update Cart
 233     } elseif ( ( ! empty( $_POST['update_cart'] ) || ! empty( $_POST['proceed'] ) ) && $woocommerce->verify_nonce('cart')) {
 234 
 235         $cart_totals = isset( $_POST['cart'] ) ? $_POST['cart'] : '';
 236 
 237         if ( sizeof( $woocommerce->cart->get_cart() ) > 0 ) {
 238             foreach ( $woocommerce->cart->get_cart() as $cart_item_key => $values ) {
 239 
 240                 $_product = $values['data'];
 241 
 242                 // Skip product if no updated quantity was posted
 243                 if ( ! isset( $cart_totals[$cart_item_key]['qty'] ) )
 244                     continue;
 245 
 246                 // Sanitize
 247                 $quantity = apply_filters( 'woocommerce_stock_amount_cart_item', apply_filters( 'woocommerce_stock_amount', preg_replace( "/[^0-9\.]/", "", $cart_totals[ $cart_item_key ]['qty'] ) ), $cart_item_key );
 248 
 249                 if ( "" === $quantity )
 250                     continue;
 251 
 252                 // Update cart validation
 253                 $passed_validation  = apply_filters( 'woocommerce_update_cart_validation', true, $cart_item_key, $values, $quantity );
 254 
 255                 // is_sold_individually
 256                 if ( $_product->is_sold_individually() && $quantity > 1 ) {
 257                     $woocommerce->add_error( sprintf( __( 'You can only have 1 %s in your cart.', 'woocommerce' ), $_product->get_title() ) );
 258                     $passed_validation = false;
 259                 }
 260 
 261                 if ( $passed_validation )
 262                     $woocommerce->cart->set_quantity( $cart_item_key, $quantity );
 263 
 264             }
 265         }
 266 
 267         if ( ! empty( $_POST['proceed'] ) ) {
 268             wp_safe_redirect( $woocommerce->cart->get_checkout_url() );
 269             exit;
 270         } else {
 271             $woocommerce->add_message( __( 'Cart updated.', 'woocommerce' ) );
 272 
 273             $referer = ( wp_get_referer() ) ? wp_get_referer() : $woocommerce->cart->get_cart_url();
 274             $referer = remove_query_arg( 'remove_discounts', $referer );
 275             wp_safe_redirect( $referer );
 276             exit;
 277         }
 278 
 279     }
 280 }

 283 /**
 284  * Add to cart action
 285  *
 286  * Checks for a valid request, does validation (via hooks) and then redirects if valid.
 287  *
 288  * @access public
 289  * @param bool $url (default: false)
 290  * @return void
 291  */
 292 function woocommerce_add_to_cart_action( $url = false ) {
 293 
 294     if ( empty( $_REQUEST['add-to-cart'] ) || ! is_numeric( $_REQUEST['add-to-cart'] ) )
 295         return;
 296 
 297     global $woocommerce;
 298 
 299     $product_id          = apply_filters( 'woocommerce_add_to_cart_product_id', absint( $_REQUEST['add-to-cart'] ) );
 300     $was_added_to_cart   = false;
 301     $added_to_cart       = array();
 302     $adding_to_cart      = get_product( $product_id );
 303     $add_to_cart_handler = apply_filters( 'woocommerce_add_to_cart_handler', $adding_to_cart->product_type, $adding_to_cart );
 304 
 305     // Variable product handling
 306     if ( 'variable' === $add_to_cart_handler ) {
 307 
 308         $variation_id       = empty( $_REQUEST['variation_id'] ) ? '' : absint( $_REQUEST['variation_id'] );
 309         $quantity           = empty( $_REQUEST['quantity'] ) ? 1 : apply_filters( 'woocommerce_stock_amount', $_REQUEST['quantity'] );
 310         $all_variations_set = true;
 311         $variations         = array();
 312 
 313         // Only allow integer variation ID - if its not set, redirect to the product page
 314         if ( empty( $variation_id ) ) {
 315             $woocommerce->add_error( __( 'Please choose product options&hellip;', 'woocommerce' ) );
 316             return;
 317         }
 318 
 319         $attributes = $adding_to_cart->get_attributes();
 320         $variation  = get_product( $variation_id );
 321 
 322         // Verify all attributes
 323         foreach ( $attributes as $attribute ) {
 324             if ( ! $attribute['is_variation'] )
 325                 continue;
 326 
 327             $taxonomy = 'attribute_' . sanitize_title( $attribute['name'] );
 328 
 329             if ( ! empty( $_REQUEST[ $taxonomy ] ) ) {
 330 
 331                 // Get value from post data
 332                 // Don't use woocommerce_clean as it destroys sanitized characters
 333                 $value = sanitize_title( trim( stripslashes( $_REQUEST[ $taxonomy ] ) ) );
 334 
 335                 // Get valid value from variation
 336                 $valid_value = $variation->variation_data[ $taxonomy ];
 337 
 338                 // Allow if valid
 339                 if ( $valid_value == '' || $valid_value == $value ) {
 340                     if ( $attribute['is_taxonomy'] )
 341                         $variations[ esc_html( $attribute['name'] ) ] = $value;
 342                     else {
 343                         // For custom attributes, get the name from the slug
 344                         $options = array_map( 'trim', explode( '|', $attribute['value'] ) );
 345                         foreach ( $options as $option ) {
 346                             if ( sanitize_title( $option ) == $value ) {
 347                                 $value = $option;
 348                                 break;
 349                             }
 350                         }
 351                          $variations[ esc_html( $attribute['name'] ) ] = $value;
 352                     }
 353                     continue;
 354                 }
 355 
 356             }
 357 
 358             $all_variations_set = false;
 359         }
 360 
 361         if ( $all_variations_set ) {
 362             // Add to cart validation
 363             $passed_validation  = apply_filters( 'woocommerce_add_to_cart_validation', true, $product_id, $quantity, $variation_id, $variations );
 364 
 365             if ( $passed_validation ) {
 366                 if ( $woocommerce->cart->add_to_cart( $product_id, $quantity, $variation_id, $variations ) ) {
 367                     woocommerce_add_to_cart_message( $product_id );
 368                     $was_added_to_cart = true;
 369                     $added_to_cart[] = $product_id;
 370                 }
 371             }
 372         } else {
 373             $woocommerce->add_error( __( 'Please choose product options&hellip;', 'woocommerce' ) );
 374             return;
 375        }
 376 
 377     // Grouped Products
 378     } elseif ( 'grouped' === $add_to_cart_handler ) {
 379 
 380         if ( ! empty( $_REQUEST['quantity'] ) && is_array( $_REQUEST['quantity'] ) ) {
 381 
 382             $quantity_set = false;
 383 
 384             foreach ( $_REQUEST['quantity'] as $item => $quantity ) {
 385                 if ( $quantity <= 0 )
 386                     continue;
 387 
 388                 $quantity_set = true;
 389 
 390                 // Add to cart validation
 391                 $passed_validation  = apply_filters( 'woocommerce_add_to_cart_validation', true, $item, $quantity );
 392 
 393                 if ( $passed_validation ) {
 394                     if ( $woocommerce->cart->add_to_cart( $item, $quantity ) ) {
 395                         $was_added_to_cart = true;
 396                         $added_to_cart[] = $item;
 397                     }
 398                 }
 399             }
 400 
 401             if ( $was_added_to_cart ) {
 402                 woocommerce_add_to_cart_message( $added_to_cart );
 403             }
 404 
 405             if ( ! $was_added_to_cart && ! $quantity_set ) {
 406                 $woocommerce->add_error( __( 'Please choose the quantity of items you wish to add to your cart&hellip;', 'woocommerce' ) );
 407                 return;
 408             }
 409 
 410         } elseif ( $product_id ) {
 411 
 412             /* Link on product archives */
 413             $woocommerce->add_error( __( 'Please choose a product to add to your cart&hellip;', 'woocommerce' ) );
 414             return;
 415 
 416         }
 417 
 418     // Simple Products
 419     } else {
 420 
 421         $quantity           = empty( $_REQUEST['quantity'] ) ? 1 : apply_filters( 'woocommerce_stock_amount', $_REQUEST['quantity'] );
 422 
 423         // Add to cart validation
 424         $passed_validation  = apply_filters( 'woocommerce_add_to_cart_validation', true, $product_id, $quantity );
 425 
 426         if ( $passed_validation ) {
 427             // Add the product to the cart
 428             if ( $woocommerce->cart->add_to_cart( $product_id, $quantity ) ) {
 429                 woocommerce_add_to_cart_message( $product_id );
 430                 $was_added_to_cart = true;
 431                 $added_to_cart[] = $product_id;
 432             }
 433         }
 434 
 435     }
 436 
 437     // If we added the product to the cart we can now do a redirect, otherwise just continue loading the page to show errors
 438     if ( $was_added_to_cart ) {
 439 
 440         $url = apply_filters( 'add_to_cart_redirect', $url );
 441 
 442         // If has custom URL redirect there
 443         if ( $url ) {
 444             wp_safe_redirect( $url );
 445             exit;
 446         }
 447 
 448         // Redirect to cart option
 449         elseif ( get_option('woocommerce_cart_redirect_after_add') == 'yes' && $woocommerce->error_count() == 0 ) {
 450             wp_safe_redirect( $woocommerce->cart->get_cart_url() );
 451             exit;
 452         }
 453 
 454         // Redirect to page without querystring args
 455         elseif ( wp_get_referer() ) {
 456             wp_safe_redirect( add_query_arg( 'added-to-cart', implode( ',', $added_to_cart ), remove_query_arg( array( 'add-to-cart', 'quantity', 'product_id' ), wp_get_referer() ) ) );
 457             exit;
 458         }
 459 
 460     }
 461 
 462 }

 511 function woocommerce_clear_cart_after_payment() {
 512     global $woocommerce;

 515

 528

 535

 537 
 538         $order = new WC_Order( $woocommerce->session->order_awaiting_payment );
 539 
 540         if ( $order->id > 0 && $order->status !== 'pending' ) {
 541             $woocommerce->cart->empty_cart();
 542         }
 543     }

1020 
1021         // Get the download URL and try to replace the url with a path
1022         $file_path = $_product->get_file_download_path( $download_id );
1023 
1024         if ( ! $file_path )
1025             wp_die( __( 'No file defined', 'woocommerce' ) . ' <a href="'.home_url().'">' . __( 'Go to homepage &rarr;', 'woocommerce' ) . '</a>' );

1034         if ( ! is_multisite() ) {
1035

1037              * Download file may be either http or https.
1038              * site_url() depends on whether the page containing the download (ie; My Account) is served via SSL because WC

1040              * So blindly doing a str_replace is incorrect because it will fail when schemes are mismatched. This code

1042              */
1043             $scheme = parse_url( $file_path, PHP_URL_SCHEME );

1068         } else {
1069             $remote_file = false;

1074

1076         }

1079         $ctype           = "application/force-download";
1080 
1081         foreach ( get_allowed_mime_types() as $mime => $type ) {
1082             $mimes = explode( '|', $mime );
1083             if ( in_array( $file_extension, $mimes ) ) {
1084                 $ctype = $type;
1085                 break;
1086             }
1087         }
1088

1090         if ( ! ini_get('safe_mode') )
1091             @set_time_limit(0);

1100         @ini_set( 'zlib.output_compression', 'Off' );

1105

1107             // IE bug prevents download via SSL when Cache Control and Pragma no-cache headers set.
1108             header( 'Expires: Wed, 11 Jan 1984 05:00:00 GMT' );
1109             header( 'Cache-Control: private' );
1110         } else {
1111             nocache_headers();
1112         }
1113

1115 
1116         if ( strstr( $file_name, '?' ) )
1117             $file_name = current( explode( '?', $file_name ) );
1118 
1119         header( "Robots: none" );
1120         header( "Content-Type: " . $ctype );
1121         header( "Content-Description: File Transfer" );
1122         header( "Content-Disposition: attachment; filename=\"" . $file_name . "\";" );
