109     public static function retrieve_password() {
110         global $woocommerce,$wpdb;
111 
112         if ( empty( $_POST['user_login'] ) ) {
113 
114             $woocommerce->add_error( __( 'Enter a username or e-mail address.', 'woocommerce' ) );
115 
116         } elseif ( strpos( $_POST['user_login'], '@' ) ) {
117 
118             $user_data = get_user_by( 'email', trim( $_POST['user_login'] ) );
119 
120             if ( empty( $user_data ) )
121                 $woocommerce->add_error( __( 'There is no user registered with that email address.', 'woocommerce' ) );
122 
123         } else {
124 
125             $login = trim( $_POST['user_login'] );
126 
127             $user_data = get_user_by('login', $login );
128         }
129 
130         do_action('lostpassword_post');
131 
132         if( $woocommerce->error_count() > 0 )
133             return false;
134 
135         if ( ! $user_data ) {
136             $woocommerce->add_error( __( 'Invalid username or e-mail.', 'woocommerce' ) );
137             return false;
138         }
139 
140         // redefining user_login ensures we return the right case in the email
141         $user_login = $user_data->user_login;
142         $user_email = $user_data->user_email;
143 
144         do_action('retrieve_password', $user_login);
145 
146         $allow = apply_filters('allow_password_reset', true, $user_data->ID);
147 
148         if ( ! $allow ) {
149 
150             $woocommerce->add_error( __( 'Password reset is not allowed for this user') );
151 
152             return false;
153 
154         } elseif ( is_wp_error( $allow ) ) {
155 
156             $woocommerce->add_error( $allow->get_error_message );
157 
158             return false;
159         }
160 
161         $key = $wpdb->get_var( $wpdb->prepare( "SELECT user_activation_key FROM $wpdb->users WHERE user_login = %s", $user_login ) );
162 
163         if ( empty( $key ) ) {

169

182     /**
183      * Retrieves a user row based on password reset key and login
184      *
185      * @uses $wpdb WordPress Database object
186      *
187      * @access public
188      * @param string $key Hash to validate sending user's password
189      * @param string $login The user login
190      * @return object|bool User's database row on success, false for invalid keys
191      */
192     public static function check_password_reset_key( $key, $login ) {
193         global $woocommerce,$wpdb;
194 
195         $key = preg_replace( '/[^a-z0-9]/i', '', $key );
196 
197         if ( empty( $key ) || ! is_string( $key ) ) {
198             $woocommerce->add_error( __( 'Invalid key', 'woocommerce' ) );
199             return false;
200         }
201 
202         if ( empty( $login ) || ! is_string( $login ) ) {
203             $woocommerce->add_error( __( 'Invalid key', 'woocommerce' ) );
204             return false;
205         }
206 
207         $user = $wpdb->get_row( $wpdb->prepare( "SELECT * FROM $wpdb->users WHERE user_activation_key = %s AND user_login = %s", $key, $login ) );
208 
209         if ( empty( $user ) ) {
210             $woocommerce->add_error( __( 'Invalid key', 'woocommerce' ) );
211             return false;
212         }
213 
214         return $user;
215     }
