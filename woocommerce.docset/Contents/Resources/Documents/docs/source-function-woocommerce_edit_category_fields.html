191 /**
192  * woocommerce_category_fields_save function.
193  *
194  * @access public
195  * @param mixed $term_id Term ID being saved
196  * @param mixed $tt_id
197  * @param mixed $taxonomy Taxonomy of the term being saved
198  * @return void
199  */
200 function woocommerce_category_fields_save( $term_id, $tt_id, $taxonomy ) {
201     if ( isset( $_POST['display_type'] ) )
202         update_woocommerce_term_meta( $term_id, 'display_type', esc_attr( $_POST['display_type'] ) );
203 
204     if ( isset( $_POST['product_cat_thumbnail_id'] ) )
205         update_woocommerce_term_meta( $term_id, 'thumbnail_id', absint( $_POST['product_cat_thumbnail_id'] ) );
206 
207     delete_transient( 'wc_term_counts' );
208 }

244 /**
245  * Fix for the per_page option
246  *
247  * Trac: http://core.trac.wordpress.org/ticket/19465
248  *
249  * @access public
250  * @param mixed $per_page
251  * @param mixed $post_type
252  * @return void
253  */
254 function woocommerce_fix_edit_posts_per_page( $per_page, $post_type ) {
255 
256     if ( $post_type !== 'product' )
257         return $per_page;
258 
259     $screen = get_current_screen();
260 
261     if ( strstr( $screen->id, '-' ) ) {
262 
263         $option = 'edit_' . str_replace( 'edit-', '', $screen->id ) . '_per_page';
264 
265         if ( isset( $_POST['wp_screen_options']['option'] ) && $_POST['wp_screen_options']['option'] == $option ) {
266 
267             update_user_meta( get_current_user_id(), $option, $_POST['wp_screen_options']['value'] );
268 
269             wp_redirect( remove_query_arg( array('pagenum', 'apage', 'paged'), wp_get_referer() ) );
270             exit;
271 
272         }
273 
274         $user_per_page = (int) get_user_meta( get_current_user_id(), $option, true );
275 
276         if ( $user_per_page )
277             $per_page = $user_per_page;
278 
279     }
280 
281     return $per_page;
282 }
