 39      * Checks for a valid postcode (UK)
 40      *
 41      * @param   string  postcode
 42      * @param   string  country
 43      * @return  bool
 44      */
 45     public function is_postcode( $postcode, $country ) {
 46         if ( strlen( trim( preg_replace( '/[\s\-A-Za-z0-9]/', '', $postcode ) ) ) > 0 )
 47             return false;
 48 
 49         switch ( $country ) {
 50             case "GB" :
 51                 return $this->is_GB_postcode( $postcode );
 52             case "US" :
 53                  if ( preg_match( "/^([0-9]{5})(-[0-9]{4})?$/i", $postcode ) )
 54                     return true;
 55                  else
 56                     return false;
 57         }
 58 
 59         return true;
 60     }

 63     /**
 64      * is_GB_postcode function.
 65      *
 66      * @author John Gardner
 67      * @access public
 68      * @param mixed $toCheck A postcode
 69      * @return bool
 70      */
 71     public function is_GB_postcode( $toCheck ) {
 72 
 73         // Permitted letters depend upon their position in the postcode.
 74         $alpha1 = "[abcdefghijklmnoprstuwyz]";                          // Character 1
 75         $alpha2 = "[abcdefghklmnopqrstuvwxy]";                          // Character 2
 76         $alpha3 = "[abcdefghjkstuw]";                                   // Character 3
 77         $alpha4 = "[abehmnprvwxy]";                                     // Character 4
 78         $alpha5 = "[abdefghjlnpqrstuwxyz]";                             // Character 5
 79 
 80         // Expression for postcodes: AN NAA, ANN NAA, AAN NAA, and AANN NAA
 81         $pcexp[0] = '^('.$alpha1.'{1}'.$alpha2.'{0,1}[0-9]{1,2})([0-9]{1}'.$alpha5.'{2})$';
 82 
 83         // Expression for postcodes: ANA NAA
 84         $pcexp[1] =  '^('.$alpha1.'{1}[0-9]{1}'.$alpha3.'{1})([0-9]{1}'.$alpha5.'{2})$';
 85 
 86         // Expression for postcodes: AANA NAA
 87         $pcexp[2] =  '^('.$alpha1.'{1}'.$alpha2.'[0-9]{1}'.$alpha4.')([0-9]{1}'.$alpha5.'{2})$';
 88 
 89         // Exception for the special postcode GIR 0AA
 90         $pcexp[3] =  '^(gir)(0aa)$';
 91 
 92         // Standard BFPO numbers
 93         $pcexp[4] = '^(bfpo)([0-9]{1,4})$';
 94 
 95         // c/o BFPO numbers
 96         $pcexp[5] = '^(bfpo)(c\/o[0-9]{1,3})$';
 97 
 98         // Load up the string to check, converting into lowercase and removing spaces
 99         $postcode = strtolower($toCheck);
100         $postcode = str_replace (' ', '', $postcode);
101 
102         // Assume we are not going to find a valid postcode
103         $valid = false;
104 
105         // Check the string against the six types of postcodes
106         foreach ($pcexp as $regexp) {
107 
108             if (ereg($regexp,$postcode, $matches)) {
109 
110                 // Load new postcode back into the form element
111                 $toCheck = strtoupper ($matches[1] . ' ' . $matches [2]);
112 
113                 // Take account of the special BFPO c/o format
114                 $toCheck = str_replace( 'C/O', 'c/o ', $toCheck );
115 
116                 // Remember that we have found that the code is valid and break from loop
117                 $valid = true;
118                 break;
119             }
120         }
121 
122         if ($valid){return true;} else {return false;};
123     }

125     /**
126      * Format the postcode according to the country and length of the postcode
127      *
128      * @param   string  postcode
129      * @param   string  country
130      * @return  string  formatted postcode
131      */
132     public function format_postcode( $postcode, $country ) {
133         $postcode = strtoupper(trim($postcode));
134         $postcode = trim(preg_replace('/[\s]/', '', $postcode));
135 
136         if ( in_array( $country, array('GB', 'CA') ) ) :
137             $postcode = trim( substr_replace( $postcode, ' ', -3, 0 ) );
138         endif;
139 
140         return $postcode;
141     }
