477 /**
478  * Get a setting from the settings API.
479  *
480  * @access public
481  * @param mixed $option
482  * @return void
483  */
484 function woocommerce_settings_get_option( $option_name, $default = '' ) {
485     // Array value
486     if ( strstr( $option_name, '[' ) ) {
487 
488         parse_str( $option_name, $option_array );
489 
490         // Option name is first key
491         $option_name = current( array_keys( $option_array ) );
492 
493         // Get value
494         $option_values = get_option( $option_name, '' );
495 
496         $key = key( $option_array[ $option_name ] );
497 
498         if ( isset( $option_values[ $key ] ) )
499             $option_value = $option_values[ $key ];
500         else
501             $option_value = null;
502 
503     // Single value
504     } else {
505         $option_value = get_option( $option_name, null );
506     }
507 
508     if ( is_array( $option_value ) )
509         $option_value = array_map( 'stripslashes', $option_value );
510     elseif ( ! is_null( $option_value ) )
511         $option_value = stripslashes( $option_value );
512 
513     return $option_value === null ? $default : $option_value;
514 }

516 /**
517  * Output admin fields.
518  *
519  * Loops though the woocommerce options array and outputs each field.
520  *
521  * @access public
522  * @param array $options Opens array to output
523  * @return void
524  */
525 function woocommerce_admin_fields( $options ) {
526     global $woocommerce;
527 
528     foreach ( $options as $value ) {
529         if ( ! isset( $value['type'] ) ) continue;
530         if ( ! isset( $value['id'] ) ) $value['id'] = '';
531         if ( ! isset( $value['title'] ) ) $value['title'] = isset( $value['name'] ) ? $value['name'] : '';
532         if ( ! isset( $value['class'] ) ) $value['class'] = '';
533         if ( ! isset( $value['css'] ) ) $value['css'] = '';
534         if ( ! isset( $value['default'] ) ) $value['default'] = '';
535         if ( ! isset( $value['desc'] ) ) $value['desc'] = '';
536         if ( ! isset( $value['desc_tip'] ) ) $value['desc_tip'] = false;
537 
538         // Custom attribute handling
539         $custom_attributes = array();
540 
541         if ( ! empty( $value['custom_attributes'] ) && is_array( $value['custom_attributes'] ) )
542             foreach ( $value['custom_attributes'] as $attribute => $attribute_value )
543                 $custom_attributes[] = esc_attr( $attribute ) . '="' . esc_attr( $attribute_value ) . '"';
544 
545         // Description handling
546         if ( $value['desc_tip'] === true ) {
547             $description = '';
548             $tip = $value['desc'];
