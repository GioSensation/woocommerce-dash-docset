  5 /**
  6  * ShareThis Integration
  7  *
  8  * Enables ShareThis integration.
  9  *
 10  * @class       WC_ShareThis
 11  * @extends     WC_Integration
 12  * @version     1.6.4
 13  * @package     WooCommerce/Classes/Integrations
 14  * @author      WooThemes
 15  */
 16 class WC_ShareThis extends WC_Integration {
 17 
 18     /** @var string Default code for share this */
 19     var $default_code;
 20 
 21     /**
 22      * Init and hook in the integration.
 23      *
 24      * @access public
 25      * @return void
 26      */
 27     public function __construct() {
 28         $this->id                   = 'sharethis';
 29         $this->method_title         = __( 'ShareThis', 'woocommerce' );
 30         $this->method_description   = __( 'ShareThis offers a sharing widget which will allow customers to share links to products with their friends.', 'woocommerce' );
 31 
 32         $this->default_code = '<div class="social">
 33     <iframe src="https://www.facebook.com/plugins/like.php?href={permalink}&layout=button_count&show_faces=false&width=100&action=like&colorscheme=light&height=21" style="border:none; overflow:hidden; width:100px; height:21px;"></iframe>
 34     <span class="st_twitter"></span><span class="st_email"></span><span class="st_sharethis" st_image="{image}"></span><span class="st_plusone_button"></span>
 35 </div>';
 36 
 37         // Load the settings.
 38         $this->init_form_fields();
 39         $this->init_settings();
 40 
 41         // Define user set variables
 42         $this->publisher_id     = $this->get_option( 'publisher_id' );
 43         $this->sharethis_code   = $this->get_option( 'sharethis_code', $this->default_code );
 44 
 45         // Actions
 46         add_action( 'woocommerce_update_options_integration_sharethis', array( $this, 'process_admin_options' ) );
 47 
 48         // Share widget
 49         add_action( 'woocommerce_share', array( $this, 'sharethis_code' ) );
 50     }

 53     /**
 54      * Initialise Settings Form Fields
 55      *
 56      * @access public
 57      * @return void
 58      */
 59     function init_form_fields() {
 60 
 61         $this->form_fields = array(
 62             'publisher_id' => array(
 63                 'title'         => __( 'ShareThis Publisher ID', 'woocommerce' ),
 64                 'description'   => sprintf( __( 'Enter your %1$sShareThis publisher ID%2$s to show social sharing buttons on product pages.', 'woocommerce' ), '<a href="http://sharethis.com/account/">', '</a>' ),
 65                 'type'          => 'text',
 66                 'default'       => get_option('woocommerce_sharethis')
 67             ),
 68             'sharethis_code' => array(
 69                 'title'         => __( 'ShareThis Code', 'woocommerce' ),
 70                 'description'   => __( 'You can tweak the ShareThis code by editing this option.', 'woocommerce' ),
 71                 'type'          => 'textarea',
 72                 'default'       => $this->default_code
 73             )
 74         );
 75 
 76     }

 79     /**
 80      * Output share code.
 81      *
 82      * @access public
 83      * @return void
 84      */
 85     function sharethis_code() {
 86         global $post;
 87 
 88         if ( $this->publisher_id ) {
 89 
 90             $thumbnail = ( $thumbnail_id = get_post_thumbnail_id( $post->ID ) ) ? current(wp_get_attachment_image_src( $thumbnail_id, 'large' )) : '';
 91 
 92             $sharethis = ( is_ssl() ) ? 'https://ws.sharethis.com/button/buttons.js' : 'http://w.sharethis.com/button/buttons.js';
 93 
 94             $sharethis_code = str_replace( '{permalink}', urlencode( get_permalink( $post->ID ) ), $this->sharethis_code );
 95             if ( isset( $thumbnail ) ) $sharethis_code = str_replace( '{image}', urlencode( $thumbnail ), $sharethis_code );
 96 
 97             echo str_replace( '&', '&amp;', $sharethis_code );
 98 
 99             echo '<script type="text/javascript">var switchTo5x=true;</script><script type="text/javascript" src="' . $sharethis . '"></script>';
100             echo '<script type="text/javascript">stLight.options({publisher:"' . $this->publisher_id . '"});</script>';
101 
102         }
103     }
