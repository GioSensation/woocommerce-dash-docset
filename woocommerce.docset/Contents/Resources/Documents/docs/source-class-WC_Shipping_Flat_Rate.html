170     /**
171      * calculate_shipping function.
172      *
173      * @access public
174      * @param array $package (default: array())
175      * @return void
176      */
177     function calculate_shipping( $package = array() ) {
178         global $woocommerce;
179 
180         $this->rates        = array();
181         $cost_per_order     = ( isset( $this->cost_per_order ) && ! empty( $this->cost_per_order ) ) ? $this->cost_per_order : 0;
182 
183         if ( $this->type == 'order' ) {
184 
185             $shipping_total = $this->order_shipping( $package );
186 
187             if ( ! is_null( $shipping_total ) || $cost_per_order > 0 )
188                 $rate = array(
189                     'id'    => $this->id,
190                     'label' => $this->title,
191                     'cost'  => $shipping_total + $cost_per_order,
192                 );
193 
194         } elseif ( $this->type == 'class' ) {
195 
196             $shipping_total = $this->class_shipping( $package );
197 
198             if ( ! is_null( $shipping_total ) || $cost_per_order > 0 )
199                 $rate = array(
200                     'id'    => $this->id,
201                     'label' => $this->title,
202                     'cost'  => $shipping_total + $cost_per_order,
203                 );
204 
205         } elseif ( $this->type == 'item' ) {
206 
207             $costs = $this->item_shipping( $package );
208 
209             if ( ! is_null( $costs ) || $cost_per_order > 0 ) {
210 
211                 if ( ! is_array( $costs ) )
212                     $costs = array();
213 
214                 $costs['order'] = $cost_per_order;
215 
216                 $rate = array(
217                     'id'        => $this->id,
218                     'label'     => $this->title,
219                     'cost'      => $costs,
220                     'calc_tax'  => 'per_item',
221                 );
222 
223             }
224         }
225 
226         if ( isset( $rate ) )
227             $this->add_rate( $rate );
228 
229         // Add any extra rates
230         if ( sizeof( $this->options ) > 0) {
231 
232             if ( ! isset( $rate ) )
233                 $rate = array(
234                     'id'    => $this->id,
235                     'label' => $this->title,
236                     'cost'  => 0,
237                 );
238 
239             // Get item qty
240             $total_quantity = 0;
241 
242             foreach ( $package['contents'] as $item_id => $values )
243                 if ( $values['quantity'] > 0 && $values['data']->needs_shipping() )
244                     $total_quantity += $values['quantity'];
245 
246             // Loop options
247             foreach ( $this->options as $option ) {
248 
249                 $this_option = array_map( 'trim', explode( '|', $option ) );
250 
251                 if ( sizeof( $this_option ) !== 3 ) continue;
252 
253                 $extra_rate = $rate;
254 
255                 $extra_rate['id']    = $this->id . ':' . sanitize_title( $this_option[0] );
256                 $extra_rate['label'] = $this_option[0];
257                 $this_cost           = $this_option[1];
258 
259                 // Backwards compat with yes and no
260                 if ( $this_option[2] == 'yes' ) {
261                     $this_type = 'order';
262                 } elseif ( $this_option[2] == 'no' ) {
263                     $this_type = $this->type;
264                 } else {
265                     $this_type = $this_option[2];
266                 }
267 
268                 switch ( $this_type ) {
269                     case 'class' :
270                         $this_cost = $this_cost * $found_shipping_classes;
271                     break;
272                     case 'item' :
273                         $this_cost = $this_cost * $total_quantity;
274                     break;
275                 }
276 
277                 // Per item rates
278                 if ( is_array( $extra_rate['cost'] ) ) $extra_rate['cost']['order'] = $extra_rate['cost']['order'] + $this_cost;
279 
280                 // Per order or class rates
281                 else $extra_rate['cost'] = $extra_rate['cost'] + $this_cost;
282 
283                 $this->add_rate( $extra_rate );
284             }
285         }
286     }

289     /**
290      * order_shipping function.
291      *
292      * @access public
293      * @param array $package
294      * @return float
295      */
296     function order_shipping( $package ) {
297         $cost   = null;
298         $fee    = null;
299 
300         if ( sizeof( $this->flat_rates ) > 0 ) {
301 
302             $found_shipping_classes = array();
303 
304             // Find shipping classes for products in the cart
305             if ( sizeof( $package['contents'] ) > 0 ) {
306                 foreach ( $package['contents'] as $item_id => $values ) {
307                     if ( $values['data']->needs_shipping() )
308                         $found_shipping_classes[] = $values['data']->get_shipping_class();
309                 }
310             }
311 
312             $found_shipping_classes = array_unique( $found_shipping_classes );
313 
314             // Find most expensive class (if found)
315             foreach ( $found_shipping_classes as $shipping_class ) {
316                 if ( isset( $this->flat_rates[ $shipping_class ] ) ) {
317                     if ( $this->flat_rates[ $shipping_class ]['cost'] > $cost ) {
318                         $cost   = $this->flat_rates[ $shipping_class ]['cost'];
319                         $fee    = $this->flat_rates[ $shipping_class ]['fee'];
320                     }
321                 } else {
322                     // No matching classes so use defaults
323                     if ( ! empty( $this->cost ) && $this->cost > $cost ) {
324                         $cost   = $this->cost;
325                         $fee    = $this->fee;
326                     }
327                 }
328             }
329 
330         }
331 
332         // Default rates if set
333         if ( is_null( $cost ) && $this->cost !== '' ) {
334             $cost   = $this->cost;
335             $fee    = $this->fee;
336         } elseif ( is_null( $cost ) ) {
337             // No match
338             return null;
339         }
340 
341         // Shipping for whole order
342         return $cost + $this->get_fee( $fee, $package['contents_cost'] );
343     }

346     /**
347      * class_shipping function.
348      *
349      * @access public
350      * @param array $package
351      * @return float
352      */
353     function class_shipping( $package ) {
354         $cost   = null;
355         $fee    = null;
356 
357         if ( sizeof( $this->flat_rates ) > 0 ) {
358             $found_shipping_classes = array();
359 
360             // Find shipping classes for products in the cart. Store prices too, so we can calc a fee for the class.
361             if ( sizeof( $package['contents'] ) > 0 ) {
362                 foreach ( $package['contents'] as $item_id => $values ) {
363                     if ( $values['data']->needs_shipping() ) {
364                         if ( isset( $found_shipping_classes[ $values['data']->get_shipping_class() ] ) ) {
365                             $found_shipping_classes[ $values['data']->get_shipping_class() ] = ( $values['data']->get_price() * $values['quantity'] ) + $found_shipping_classes[ $values['data']->get_shipping_class() ];
366                         } else {
367                             $found_shipping_classes[ $values['data']->get_shipping_class() ] = ( $values['data']->get_price() * $values['quantity'] );
368                         }
369                     }
370                 }
371             }
372 
373             $found_shipping_classes = array_unique( $found_shipping_classes );
374 
375             $matched = false;
376 
377             // For each found class, add up the costs and fees
378             foreach ( $found_shipping_classes as $shipping_class => $class_price ) {
379                 if ( isset( $this->flat_rates[ $shipping_class ] ) ) {
380                     $cost   += $this->flat_rates[ $shipping_class ]['cost'];
381                     $fee    += $this->get_fee( $this->flat_rates[ $shipping_class ]['fee'], $class_price );
382                     $matched = true;
383                 } elseif ( $this->cost !== '' ) {
384                     // Class not set so we use default rate if its set
385                     $cost   += $this->cost;
386                     $fee    += $this->get_fee( $this->fee, $class_price );
387                     $matched = true;
388                 }
389             }
390         }
391 
392         // Total
393         if ( $matched )
394             return $cost + $fee;
395         else
396             return null;
397     }

400     /**
401      * item_shipping function.
402      *
403      * @access public
404      * @param array $package
405      * @return array
406      */
407     function item_shipping( $package ) {
408         // Per item shipping so we pass an array of costs (per item) instead of a single value
409         $costs = array();
410 
411         $matched = false;
412 
413         // Shipping per item
414         foreach ( $package['contents'] as $item_id => $values ) {
415             $_product = $values['data'];
416 
417             if ( $values['quantity'] > 0 && $_product->needs_shipping() ) {
418                 $shipping_class = $_product->get_shipping_class();
419 
420                 $fee = $cost = 0;
421 
422                 if ( isset( $this->flat_rates[ $shipping_class ] ) ) {
423                     $cost   = $this->flat_rates[ $shipping_class ]['cost'];
424                     $fee    = $this->get_fee( $this->flat_rates[ $shipping_class ]['fee'], $_product->get_price() );
425                     $matched = true;
426                 } elseif ( $this->cost !== '' ) {
427                     $cost   = $this->cost;
428                     $fee    = $this->get_fee( $this->fee, $_product->get_price() );
429                     $matched = true;
430                 }
431 
432                 $costs[ $item_id ] = ( ( $cost + $fee ) * $values['quantity'] );
433             }
434         }
435 
436         if ( $matched )
437             return $costs;
438         else
439             return null;
440     }

571         // Save the rates
572         $flat_rate_class = array();
573         $flat_rate_cost = array();
574         $flat_rate_fee = array();
