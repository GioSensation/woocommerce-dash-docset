140     /**
141      * is_available function.
142      *
143      * @access public
144      * @param array $package
145      * @return bool
146      */
147     function is_available( $package ) {
148         global $woocommerce;
149 
150         $is_available = true;
151 
152         if ( $this->enabled == "no" ) {
153 
154             $is_available = false;
155 
156         } else {
157 
158             // If post codes are listed, let's use them.
159             $codes = '';
160             if ( $this->codes != '' ) {
161                 foreach( explode( ',', $this->codes ) as $code ) {
162                     $codes[] = $this->clean( $code );
163                 }
164             }
165 
166             if ( is_array( $codes ) ) {
167 
168                 $found_match = false;
169 
170                 if ( in_array( $this->clean( $package['destination']['postcode'] ), $codes ) )
171                     $found_match = true;
172 
173                 // Wildcard search
174                 if ( ! $found_match ) {
175 
176                     $customer_postcode = $this->clean( $package['destination']['postcode'] );
177                     $customer_postcode_length = strlen( $customer_postcode );
178 
179                     for ( $i = 0; $i <= $customer_postcode_length; $i++ ) {
180 
181                         if ( in_array( $customer_postcode, $codes ) )
182                             $found_match = true;
183 
184                         $customer_postcode = substr( $customer_postcode, 0, -2 ) . '*';
185                     }
186                 }
187 
188                 if ( ! $found_match ) {
189 
190                     $is_available = false;
191 
192                 } else {
193 
194                     $ship_to_countries = '';
195 
196                     if ( $this->availability == 'specific' ) {
197                         $ship_to_countries = $this->countries;
198                     } elseif ( get_option( 'woocommerce_allowed_countries' ) == 'specific' ) {
199                         $ship_to_countries = get_option( 'woocommerce_specific_allowed_countries' );
200                     }
201 
202                     if ( is_array( $ship_to_countries ) && ! in_array( $package['destination']['country'], $ship_to_countries ) ) {
203                         $is_available = false;
204                     }
205 
206                 }
207             }
208 
209         }
210 
211         return apply_filters( 'woocommerce_shipping_' . $this->id . '_is_available', $is_available, $package );
212     }

222     function taxable_address( $address ) {

226             if ( $this->get_option( 'apply_base_tax' ) == 'yes' ) {

229                 $state      = $woocommerce->countries->get_base_state();
