171      * Is customer outside base country (for tax purposes)?
172      *
173      * @access public
174      * @return bool
175      */
176     public function is_customer_outside_base() {
177         list( $country, $state, $postcode, $city ) = $this->get_taxable_address();
178 
179         if ( $country ) {
180 
181             $default = get_option('woocommerce_default_country');
182 
183             if ( strstr( $default, ':' ) ) {
184                 list( $default_country, $default_state ) = explode( ':', $default );
185             } else {
186                 $default_country = $default;
187                 $default_state = '';
188             }
189 
190             if ( $default_country !== $country ) return true;
191             if ( $default_state && $default_state !== $state ) return true;
192 
193         }
194         return false;
195     }

343      *
344      * @access public
345      * @return void
346      */
347     public function get_taxable_address() {
348         $tax_based_on = get_option( 'woocommerce_tax_based_on' );
349 
350         if ( $tax_based_on == 'base' ) {
351 
352             $default = get_option( 'woocommerce_default_country' );
353             if ( strstr( $default, ':' ) ) {
354                 list( $country, $state ) = explode( ':', $default );
355             } else {
356                 $country = $default;
357                 $state = '';
358             }
359 
360             $postcode   = '';
361             $city       = '';
362 
363         } elseif ( $tax_based_on == 'billing' ) {
364 
365             $country    = $this->get_country();
366             $state      = $this->get_state();
367             $postcode   = $this->get_postcode();
368             $city       = $this->get_city();
369 
370         } else {
371 
372             $country    = $this->get_shipping_country();
373             $state      = $this->get_shipping_state();
374             $postcode   = $this->get_shipping_postcode();
375             $city       = $this->get_shipping_city();
376 
377         }
378 
379         return apply_filters( 'woocommerce_customer_taxable_address', array( $country, $state, $postcode, $city ) );
380     }

582     /**
583      * Gets a user's downloadable products if they are logged in.
584      *
585      * @access public
586      * @return array Array of downloadable products
587      */
588     public function get_downloadable_products() {
589         global $wpdb, $woocommerce;
590 
591         $downloads = array();
592 
593         if ( is_user_logged_in() ) :
594 
595             $user_info = get_userdata( get_current_user_id() );
596 
597             $results = $wpdb->get_results( $wpdb->prepare("SELECT * FROM ".$wpdb->prefix."woocommerce_downloadable_product_permissions WHERE user_id = '%s' ORDER BY order_id, product_id, download_id", get_current_user_id()) );
598 
599             $_product = null;
600             $order = null;
601             $file_number = 0;
602             if ($results) foreach ($results as $result) :
603 
604                 if ($result->order_id>0) :
605 
606                     if ( ! $order || $order->id != $result->order_id ) {
607                         // new order
608                         $order = new WC_Order( $result->order_id );
609                         $_product = null;
610                     }
611 
612                     // order exists and downloads permitted?
613                     if ( ! $order->id || ! $order->is_download_permitted() || $order->post_status != 'publish' ) continue;
614 
615                     if ( ! $_product || $_product->id != $result->product_id ) :
616                         // new product
617                         $file_number = 0;
618                         $_product = get_product( $result->product_id );
619                     endif;
620 
621                     if ( ! $_product || ! $_product->exists() ) continue;
622 
623                     if ( ! $_product->has_file( $result->download_id ) ) continue;
624 
625                     // Download name will be 'Product Name' for products with a single downloadable file, and 'Product Name - File X' for products with multiple files
626                     $download_name = apply_filters(
627                         'woocommerce_downloadable_product_name',
628                         $_product->get_title() . ( $file_number > 0 ? ' &mdash; ' . sprintf( __( 'File %d', 'woocommerce' ), $file_number + 1 ) : '' ),
629                         $_product,
630                         $result->download_id,
631                         $file_number
632                     );
633 
634                     // Rename previous download with file number if there are multiple files only
635                     if ( $file_number == 1 ) {
636                         $downloads[ count( $downloads ) - 1 ]['download_name'] = apply_filters(
637                             'woocommerce_downloadable_product_name',
638                             $downloads[ count( $downloads ) - 1 ]['download_name'] . ' &mdash; ' . sprintf( __( 'File %d', 'woocommerce' ), $file_number ),
639                             $_product,
640                             $result->download_id,
641                             0
642                         );
643                     }
644 
645                     $downloads[] = array(
646                         'download_url' => add_query_arg( array( 'download_file' => $result->product_id, 'order' => $result->order_key, 'email' => $result->user_email, 'key' => $result->download_id ), trailingslashit( home_url( '', 'http' ) ) ),
647                         'download_id' => $result->download_id,
648                         'product_id' => $result->product_id,
649                         'download_name' => $download_name,
650                         'order_id' => $order->id,
651                         'order_key' => $order->order_key,
652                         'downloads_remaining' => $result->downloads_remaining
653                     );
654 
655                     $file_number++;
656 
657                 endif;
658 
659             endforeach;
660 
661         endif;
662 
663         return apply_filters('woocommerce_customer_get_downloadable_products', $downloads);
664     }
