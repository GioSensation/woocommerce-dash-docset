 322     /**
 323      * Get the base country for the store.
 324      *
 325      * @access public
 326      * @return string
 327      */
 328     public function get_base_country() {
 329         $default = esc_attr( get_option('woocommerce_default_country') );
 330         if ( ( $pos = strpos( $default, ':' ) ) === false )
 331             return $default;
 332         return substr( $default, 0, $pos );
 333     }

 350     /**
 351      * Get the allowed countries for the store.
 352      *
 353      * @access public
 354      * @return array
 355      */
 356     public function get_allowed_countries() {
 357 
 358         if ( apply_filters('woocommerce_sort_countries', true ) )
 359             asort( $this->countries );
 360 
 361         if ( get_option('woocommerce_allowed_countries') !== 'specific' )
 362             return $this->countries;
 363 
 364         $allowed_countries = array();
 365 
 366         $allowed_countries_raw = get_option( 'woocommerce_specific_allowed_countries' );
 367 
 368         foreach ( $allowed_countries_raw as $country )
 369             $allowed_countries[ $country ] = $this->countries[ $country ];
 370 
 371         return $allowed_countries;
 372     }

 375     /**
 376      * get_allowed_country_states function.
 377      *
 378      * @access public
 379      * @return array
 380      */
 381     public function get_allowed_country_states() {
 382 
 383         if ( get_option('woocommerce_allowed_countries') !== 'specific' )
 384             return $this->states;
 385 
 386         $allowed_states = array();
 387 
 388         $allowed_countries_raw = get_option( 'woocommerce_specific_allowed_countries' );
 389 
 390         foreach ( $allowed_countries_raw as $country )
 391             if ( ! empty( $this->states[ $country ] ) )
 392                 $allowed_states[ $country ] = $this->states[ $country ];
 393 
 394         return $allowed_states;
 395     }

 599     /**
 600      * Get country address format
 601      *
 602      * @access public
 603      * @param array $args (default: array())
 604      * @return string address
 605      */
 606     public function get_formatted_address( $args = array() ) {
 607 
 608         $args = array_map( 'trim', $args );
 609 
 610         extract( $args );
 611 
 612         // Get all formats
 613         $formats        = $this->get_address_formats();
 614 
 615         // Get format for the address' country
 616         $format         = ( $country && isset( $formats[ $country ] ) ) ? $formats[ $country ] : $formats['default'];
 617 
 618         // Handle full country name
 619         $full_country   = ( isset( $this->countries[ $country ] ) ) ? $this->countries[ $country ] : $country;
 620 
 621         // Country is not needed if the same as base
 622         if ( $country == $this->get_base_country() )
 623             $format = str_replace( '{country}', '', $format );
 624 
 625         // Handle full state name
 626         $full_state     = ( $country && $state && isset( $this->states[ $country ][ $state ] ) ) ? $this->states[ $country ][ $state ] : $state;
 627 
 628         // Substitute address parts into the string
 629         $replace = apply_filters( 'woocommerce_formatted_address_replacements', array(
 630             '{first_name}'       => $first_name,
 631             '{last_name}'        => $last_name,
 632             '{name}'             => $first_name . ' ' . $last_name,
 633             '{company}'          => $company,
 634             '{address_1}'        => $address_1,
 635             '{address_2}'        => $address_2,
 636             '{city}'             => $city,
 637             '{state}'            => $full_state,
 638             '{postcode}'         => $postcode,
 639             '{country}'          => $full_country,
 640             '{first_name_upper}' => strtoupper( $first_name ),
 641             '{last_name_upper}'  => strtoupper( $last_name ),
 642             '{name_upper}'       => strtoupper( $first_name . ' ' . $last_name ),
 643             '{company_upper}'    => strtoupper( $company ),
 644             '{address_1_upper}'  => strtoupper( $address_1 ),
 645             '{address_2_upper}'  => strtoupper( $address_2 ),
 646             '{city_upper}'       => strtoupper( $city ),
 647             '{state_upper}'      => strtoupper( $full_state ),
 648             '{postcode_upper}'   => strtoupper( $postcode ),
 649             '{country_upper}'    => strtoupper( $full_country ),
 650         ), $args ) ;
 651 
 652         $replace = array_map( 'esc_html', $replace );
 653 
 654         $formatted_address = str_replace( array_keys( $replace ), $replace, $format );
 655 
 656         // Clean up white space
 657         $formatted_address = preg_replace( '/  +/', ' ', trim( $formatted_address ) );
 658         $formatted_address = preg_replace( '/\n\n+/', "\n", $formatted_address );
 659 
 660         // Add html breaks
 661         $formatted_address = nl2br( $formatted_address );
 662 
 663         // We're done!
 664         return $formatted_address;
 665     }
