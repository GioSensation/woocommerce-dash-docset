 44     /**
 45      * trigger function.
 46      *
 47      * @access public
 48      * @return void
 49      */
 50     function trigger( $user_id, $user_pass = '' ) {
 51         global $woocommerce;
 52 
 53         if ( $user_id ) {
 54             $this->object       = new WP_User( $user_id );
 55 
 56             $this->user_pass    = $user_pass;
 57             $this->user_login   = stripslashes( $this->object->user_login );
 58             $this->user_email   = stripslashes( $this->object->user_email );
 59             $this->recipient    = $this->user_email;
 60         }
 61 
 62         if ( ! $this->is_enabled() || ! $this->get_recipient() )
 63             return;
 64 
 65         $this->send( $this->get_recipient(), $this->get_subject(), $this->get_content(), $this->get_headers(), $this->get_attachments() );
 66     }
