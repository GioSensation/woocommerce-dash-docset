173 
174         $tax_class = sanitize_title( $tax_class );
175 
176         /* Checkout uses customer location for the tax rates. Also, if shipping has been calculated, use the customers address. */
177         if ( ( defined('WOOCOMMERCE_CHECKOUT') && WOOCOMMERCE_CHECKOUT ) || $woocommerce->customer->has_calculated_shipping() ) {
178 
179             list( $country, $state, $postcode, $city ) = $woocommerce->customer->get_taxable_address();
180 
181             $matched_tax_rates = $this->find_rates( array(
182                 'country'   => $country,
183                 'state'     => $state,
184                 'postcode'  => $postcode,
185                 'city'      => $city,
186                 'tax_class' => $tax_class
187             ) );
188 
189         } else {
190 
191             // Prices which include tax should always use the base rate if we don't know where the user is located
192             // Prices excluding tax however should just not add any taxes, as they will be added during checkout.
193             // The woocommerce_default_customer_address option (when set to base) is also used here.
194             $matched_tax_rates = $woocommerce->cart->prices_include_tax || get_option( 'woocommerce_default_customer_address' ) == 'base'
195                 ? $this->get_shop_base_rate( $tax_class )
196                 : array();
197 
198         }
199 
200         return apply_filters('woocommerce_matched_rates', $matched_tax_rates, $tax_class);
201 
202     }

366     /**
367      * Calculate the tax using a passed array of rates.
368      *
369      * @param   float   price
370      * @param   array   rates
371      * @param   bool    passed price includes tax
372      * @return  array   array of rates/amounts
373      */
374     public function calc_tax( $price, $rates, $price_includes_tax = true, $suppress_rounding = false ) {
375 
376         $price = $price * 100;  // To avoid float rounding errors, work with integers (pence)
377 
378         // Taxes array
379         $taxes = array();
380 
381         // Price inc tax
382         if ( $price_includes_tax ) {
383 
384             $regular_tax_rates = $compound_tax_rates = 0;
385 
386             foreach ( $rates as $key => $rate )
387                 if ($rate['compound']=='yes')
388                     $compound_tax_rates = $compound_tax_rates + $rate['rate'];
389                 else
390                     $regular_tax_rates = $regular_tax_rates + $rate['rate'];
391 
392             $regular_tax_rate   = 1 + ( $regular_tax_rates / 100 );
393             $compound_tax_rate  = 1 + ( $compound_tax_rates / 100 );
394             $tax_rate           = $regular_tax_rate * $compound_tax_rate;
395             $non_compound_price = ( $price / $compound_tax_rate );
396 
397             foreach ( $rates as $key => $rate ) {
398 
399                 $the_rate = $rate['rate'] / 100;
400 
401                 if ( $rate['compound'] == 'yes' )
402                     $tax_amount = ( $the_rate / $compound_tax_rate ) * $price;
403                 else
404                     $tax_amount = ( $the_rate / $regular_tax_rate ) * $non_compound_price;
405 
406                 // ADVANCED: Allow third parties to modify this rate
407                 $tax_amount = apply_filters( 'woocommerce_price_inc_tax_amount', $tax_amount, $key, $rate, $price, $non_compound_price );
408 
409                 // Back to pounds
410                 $tax_amount = ( $tax_amount / 100 );
411 
412                 // Rounding
413                 if ( get_option( 'woocommerce_tax_round_at_subtotal' ) == 'no' && ! $suppress_rounding ) {
414                     $tax_amount = $this->round( $tax_amount );
415                 }
416 
417                 // Add rate
418                 if ( ! isset( $taxes[$key] ) )
419                     $taxes[ $key ] = $tax_amount;
420                 else
421                     $taxes[ $key ] += $tax_amount;
422 
423             }
424 
425         // Prices ex tax
426         } else {
427 
428             // Multiple taxes
429             foreach ( $rates as $key => $rate ) {
430 
431                 if ( $rate['compound'] == 'yes' )
432                     continue;
433 
434                 $tax_amount = $price * ( $rate['rate'] / 100 );
435 
436                 // ADVANCED: Allow third parties to modify this rate
437                 $tax_amount = apply_filters( 'woocommerce_price_ex_tax_amount', $tax_amount, $key, $rate, $price );
438 
439                 // Back to pounds
440                 $tax_amount = ( $tax_amount / 100 );
441 
442                 // Rounding
443                 if ( get_option( 'woocommerce_tax_round_at_subtotal' ) == 'no' && ! $suppress_rounding ) {
444                     $tax_amount = $this->round( $tax_amount );
445                 }
446 
447                 // Add rate
448                 if ( ! isset( $taxes[ $key ] ) )
449                     $taxes[ $key ] = $tax_amount;
450                 else
451                     $taxes[ $key ] += $tax_amount;
452 
453             }
454 
455             $pre_compound_total = array_sum( $taxes );
456 
457             // Compound taxes
458             if ( $rates ) {
459                 foreach ( $rates as $key => $rate ) {
460 
461                     if ( $rate['compound'] == 'no' )
462                         continue;
463 
464                     $the_price_inc_tax = $price + ( $pre_compound_total * 100 );
465 
466                     $tax_amount = $the_price_inc_tax * ( $rate['rate'] / 100 );
467 
468                     // ADVANCED: Allow third parties to modifiy this rate
469                     $tax_amount = apply_filters( 'woocommerce_price_ex_tax_amount', $tax_amount, $key, $rate, $price, $the_price_inc_tax, $pre_compound_total );
470 
471                     // Back to pounds
472                     $tax_amount = ( $tax_amount / 100 );
473 
474                     // Rounding
475                     if ( get_option( 'woocommerce_tax_round_at_subtotal' ) == 'no' && ! $suppress_rounding ) {
476                         $tax_amount = $this->round( $tax_amount );
477                     }
478 
479                     // Add rate
480                     if ( ! isset( $taxes[ $key ] ) )
481                         $taxes[ $key ] = $tax_amount;
482                     else
483                         $taxes[ $key ] += $tax_amount;
484 
485                 }
486             }
487         }
488 
489         return apply_filters( 'woocommerce_calc_tax', $taxes, $price, $rates, $price_includes_tax, $suppress_rounding );
490     }

493      * Calculate the shipping tax using a passed array of rates.
494      *
495      * @param   int     Price
496      * @param   int     Taxation Rate
497      * @return  int
498      */
499     public function calc_shipping_tax( $price, $rates ) {
500 
501         // Taxes array
502         $taxes = array();
503 
504         // Multiple taxes
505         foreach ($rates as $key => $rate) :
506             if ($rate['compound']=='yes') continue;
507 
508             $tax_amount = $price * ($rate['rate']/100);
509 
510             // Add rate
511             if (!isset($taxes[$key])) $taxes[$key] = $tax_amount; else $taxes[$key] += $tax_amount;
512 
513         endforeach;
514 
515         $pre_compound_total = array_sum($taxes);
516 
517         // Compound taxes
518         foreach ($rates as $key => $rate) :
519             if ($rate['compound']=='no') continue;
520 
521             $the_price_inc_tax = $price + $pre_compound_total;
522 
523             $tax_amount = $the_price_inc_tax * ($rate['rate']/100);
524 
525             // Add rate
526             if (!isset($taxes[$key])) $taxes[$key] = $tax_amount; else $taxes[$key] += $tax_amount;
527 
528         endforeach;
529 
530         return $taxes;
