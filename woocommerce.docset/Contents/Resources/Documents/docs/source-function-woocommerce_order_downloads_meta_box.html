197  * @param mixed $post

199  */
200 function woocommerce_order_downloads_save( $post_id, $post ) {
201     global $wpdb, $woocommerce;
202 
203     if ( isset( $_POST['download_id'] ) ) {
204 
205         // Download data
206         $download_ids           = $_POST['download_id'];
207         $product_ids            = $_POST['product_id'];
208         $downloads_remaining    = $_POST['downloads_remaining'];
209         $access_expires         = $_POST['access_expires'];
210 
211         // Order data
212         $order_key              = get_post_meta( $post->ID, '_order_key', true );
213         $customer_email         = get_post_meta( $post->ID, '_billing_email', true );
214         $customer_user          = get_post_meta( $post->ID, '_customer_user', true );
215         $product_ids_count      = sizeof( $product_ids );
216 
217         for ( $i = 0; $i < $product_ids_count; $i ++ ) {
218 
219             $data = array(
220                 'user_id'               => absint( $customer_user ),
221                 'user_email'            => woocommerce_clean( $customer_email ),
222                 'downloads_remaining'   => woocommerce_clean( $downloads_remaining[$i] )
223             );
224 
225             $format = array( '%d', '%s', '%s' );
226 
227             $expiry  = ( array_key_exists( $i, $access_expires ) && $access_expires[ $i ] != '' ) ? date_i18n( 'Y-m-d', strtotime( $access_expires[ $i ] ) ) : null;
228 
229             if ( ! is_null( $expiry ) ) {
230                 $data['access_expires'] = $expiry;
231                 $format[] = '%s';
232             }
233 
234             $wpdb->update( $wpdb->prefix . "woocommerce_downloadable_product_permissions",
235                 $data,
236                 array(
237                     'order_id'      => $post_id,
238                     'product_id'    => absint( $product_ids[$i] ),
239                     'download_id'   => woocommerce_clean( $download_ids[$i] )
240                     ),
241                 $format, array( '%d', '%d', '%s' )
242             );
243 
244         }
245     }
246 }
