 52      * @access public
 53      * @param string $order (default: '')
 54      * @return string
 55      */
 56     function get_return_url( $order = '' ) {
 57 
 58         $thanks_page_id = woocommerce_get_page_id('thanks');
 59         if ( $thanks_page_id ) :
 60             $return_url = get_permalink($thanks_page_id);
 61         else :
 62             $return_url = home_url();
 63         endif;
 64 
 65         if ( $order ) :
 66             $return_url = add_query_arg( 'key', $order->order_key, add_query_arg( 'order', $order->id, $return_url ) );
 67         endif;
 68 
 69         if ( is_ssl() || get_option('woocommerce_force_ssl_checkout') == 'yes' )
 70             $return_url = str_replace( 'http:', 'https:', $return_url );
 71 
 72         return apply_filters( 'woocommerce_get_return_url', $return_url );
 73     }
