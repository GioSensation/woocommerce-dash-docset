172     /**
173      * Send the email.
174      *
175      * @access public
176      * @param mixed $to
177      * @param mixed $subject
178      * @param mixed $message
179      * @param string $headers (default: "Content-Type: text/html\r\n")
180      * @param string $attachments (default: "")
181      * @param string $content_type (default: "text/html")
182      * @return void
183      */
184     function send( $to, $subject, $message, $headers = "Content-Type: text/html\r\n", $attachments = "", $content_type = 'text/html' ) {
185 
186         // Set content type
187         $this->_content_type = $content_type;
188 
189         // Filters for the email
190         add_filter( 'wp_mail_from', array( $this, 'get_from_address' ) );
191         add_filter( 'wp_mail_from_name', array( $this, 'get_from_name' ) );
192         add_filter( 'wp_mail_content_type', array( $this, 'get_content_type' ) );
193 
194         // Send
195         wp_mail( $to, $subject, $message, $headers, $attachments );
196 
197         // Unhook filters
198         remove_filter( 'wp_mail_from', array( $this, 'get_from_address' ) );
199         remove_filter( 'wp_mail_from_name', array( $this, 'get_from_name' ) );
200         remove_filter( 'wp_mail_content_type', array( $this, 'get_content_type' ) );
201     }

232      * Add order meta to email templates.
233      *
234      * @access public
235      * @param mixed $order
236      * @param bool $sent_to_admin (default: false)
237      * @param bool $plain_text (default: false)
238      * @return void
239      */
240     function order_meta( $order, $sent_to_admin = false, $plain_text = false ) {
241 
242         $meta = array();
243         $show_fields = apply_filters( 'woocommerce_email_order_meta_keys', array(), $sent_to_admin );
244 
245         if ( $order->customer_note )
246             $meta[ __( 'Note', 'woocommerce' ) ] = wptexturize( $order->customer_note );
247 
248         if ( $show_fields )
249             foreach ( $show_fields as $field ) {
250                 $value = get_post_meta( $order->id, $field, true );
251                 if ( $value )
252                     $meta[ ucwords( esc_attr( $field ) ) ] = wptexturize( $value );
253             }
254 
255         if ( sizeof( $meta ) > 0 ) {
256 
257             if ( $plain_text ) {
258 
259                 foreach ( $meta as $key => $value )
260                     echo $key . ': ' . $value . "\n";
261 
262             } else {
263 
264                 foreach ( $meta as $key => $value )
265                     echo '<p><strong>' . $key . ':</strong> ' . $value . '</p>';
266             }
267         }
268     }

282 
283         $sku = ($product->sku) ? '(' . $product->sku . ') ' : '';
284 
285         if ( ! empty( $product->variation_id ) )
286             $title = sprintf(__( 'Variation #%s of %s', 'woocommerce' ), $product->variation_id, get_the_title($product->id)) . ' ' . $sku;
287         else
288             $title = sprintf(__( 'Product #%s - %s', 'woocommerce' ), $product->id, get_the_title($product->id)) . ' ' . $sku;
289 
290         $message = $title . __( 'is low in stock.', 'woocommerce' );
291 
292         //  CC, BCC, additional headers
293         $headers = apply_filters('woocommerce_email_headers', '', 'low_stock', $product);
294 
295         // Attachments
296         $attachments = apply_filters('woocommerce_email_attachments', '', 'low_stock', $product);
297 
298         // Send the mail
299         wp_mail( get_option('woocommerce_stock_email_recipient'), $subject, $message, $headers, $attachments );
