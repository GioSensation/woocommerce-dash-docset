  35 /**
  36  * Function that returns an array containing the IDs of the products that are on sale.
  37  *
  38  * @since 2.0
  39  * @access public
  40  * @return array
  41  */
  42 function woocommerce_get_product_ids_on_sale() {
  43     // Load from cache
  44     $product_ids_on_sale = get_transient( 'wc_products_onsale' );
  45 
  46     // Valid cache found
  47     if ( false !== $product_ids_on_sale )
  48         return $product_ids_on_sale;
  49 
  50     $on_sale = get_posts( array(
  51         'post_type'      => array( 'product', 'product_variation' ),
  52         'posts_per_page' => -1,
  53         'post_status'    => 'publish',
  54         'meta_query'     => array(
  55             array(
  56                 'key'        => '_sale_price',
  57                 'value'      => 0,
  58                 'compare'    => '>=',
  59                 'type'       => 'DECIMAL',
  60             ),
  61             array(
  62                 'key'        => '_sale_price',
  63                 'value'      => '',
  64                 'compare'    => '!=',
  65                 'type'       => '',
  66             )
  67         ),
  68         'fields'         => 'id=>parent',
  69     ) );
  70 
  71     $product_ids = array_keys( $on_sale );
  72     $parent_ids  = array_values( $on_sale );
  73 
  74     // Check for scheduled sales which have not started
  75     foreach ( $product_ids as $key => $id ) {
  76         if ( get_post_meta( $id, '_sale_price_dates_from', true ) > current_time( 'timestamp' ) ) {
  77             unset( $product_ids[ $key ] );
  78         }
  79     }
  80 
  81     $product_ids_on_sale = array_unique( array_merge( $product_ids, $parent_ids ) );
  82 
  83     set_transient( 'wc_products_onsale', $product_ids_on_sale );
  84 
  85     return $product_ids_on_sale;
  86 }

  89  * Sanitize taxonomy names. Slug format (no spaces, lowercase).
  90  *
  91  * Doesn't use sanitize_title as this destroys utf chars.
  92  *
  93  * @access public
  94  * @param mixed $taxonomy
  95  * @return void
  96  */
  97 function woocommerce_sanitize_taxonomy_name( $taxonomy ) {
  98     $taxonomy = strtolower( stripslashes( strip_tags( $taxonomy ) ) );
  99     $taxonomy = preg_replace( '/&.+?;/', '', $taxonomy ); // Kill entities
 100     $taxonomy = str_replace( array( '.', '\'', '"' ), '', $taxonomy ); // Kill quotes and full stops.

 145 /**
 146  * woocommerce_get_dimension function.
 147  *
 148  * Normalise dimensions, unify to cm then convert to wanted unit value
 149  *
 150  * Usage: woocommerce_get_dimension(55, 'in');
 151  *
 152  * @access public
 153  * @param mixed $dim
 154  * @param mixed $to_unit 'in', 'm', 'cm', 'm'
 155  * @return float
 156  */
 157 function woocommerce_get_dimension( $dim, $to_unit ) {
 158 
 159     $from_unit  = strtolower( get_option( 'woocommerce_dimension_unit' ) );
 160     $to_unit    = strtolower( $to_unit );
 161 
 162     // Unify all units to cm first
 163     if ( $from_unit !== $to_unit ) {
 164 
 165         switch ( $from_unit ) {
 166             case 'in':
 167                 $dim *= 2.54;
 168             break;
 169             case 'm':
 170                 $dim *= 100;
 171             break;
 172             case 'mm':
 173                 $dim *= 0.1;
 174             break;
 175             case 'yd':
 176                 $dim *= 0.010936133;
 177             break;
 178         }
 179 
 180         // Output desired unit
 181         switch ( $to_unit ) {
 182             case 'in':
 183                 $dim *= 0.3937;
 184             break;
 185             case 'm':
 186                 $dim *= 0.01;
 187             break;
 188             case 'mm':
 189                 $dim *= 10;
 190             break;
 191             case 'yd':
 192                 $dim *= 91.44;
 193             break;
 194         }
 195     }
 196     return ( $dim < 0 ) ? 0 : $dim;
 197 }

 250  * Get product name with extra details such as SKU price and attributes. Used within admin.
 251  *
 252  * @access public
 253  * @param mixed $product
 254  * @return void
 255  */
 256 function woocommerce_get_formatted_product_name( $product ) {
 257     if ( ! $product || ! is_object( $product ) )
 258         return;
 259 
 260     if ( $product->get_sku() )
 261         $identifier = $product->get_sku();
 262     elseif ( $product->is_type( 'variation' ) )
 263         $identifier = '#' . $product->variation_id;
 264     else
 265         $identifier = '#' . $product->id;
 266 
 267     if ( $product->is_type( 'variation' ) ) {
 268         $attributes = $product->get_variation_attributes();
 269         $extra_data = ' &ndash; ' . implode( ', ', $attributes ) . ' &ndash; ' . woocommerce_price( $product->get_price() );
 270     } else {
 271         $extra_data = '';
 272     }
 273 
 274     return sprintf( __( '%s &ndash; %s%s', 'woocommerce' ), $identifier, $product->get_title(), $extra_data );
 275 }

 626  *

 632  * @param mixed $template_name

 638     global $woocommerce;
 639 
 640     if ( ! $template_path ) $template_path = $woocommerce->template_url;
 641     if ( ! $default_path ) $default_path = $woocommerce->plugin_path() . '/templates/';

 643     // Look within passed path within the theme - this is priority

1638 /**
1639  * Move a term before the a given element of its hierarchy level
1640  *
1641  * @access public
1642  * @param int $the_term
1643  * @param int $next_id the id of the next sibling element in save hierarchy level
1644  * @param string $taxonomy
1645  * @param int $index (default: 0)
1646  * @param mixed $terms (default: null)
1647  * @return int
1648  */
1649 function woocommerce_order_terms( $the_term, $next_id, $taxonomy, $index = 0, $terms = null ) {
1650 
1651     if( ! $terms ) $terms = get_terms($taxonomy, 'menu_order=ASC&hide_empty=0&parent=0' );
1652     if( empty( $terms ) ) return $index;
1653 
1654     $id = $the_term->term_id;
1655 
1656     $term_in_level = false; // flag: is our term to order in this level of terms
1657 
1658     foreach ($terms as $term) {
1659 
1660         if( $term->term_id == $id ) { // our term to order, we skip
1661             $term_in_level = true;
1662             continue; // our term to order, we skip
1663         }
1664         // the nextid of our term to order, lets move our term here
1665         if(null !== $next_id && $term->term_id == $next_id) {
1666             $index++;
1667             $index = woocommerce_set_term_order($id, $index, $taxonomy, true);
1668         }
1669 
1670         // set order
1671         $index++;
1672         $index = woocommerce_set_term_order($term->term_id, $index, $taxonomy);
1673 
1674         // if that term has children we walk through them
1675         $children = get_terms($taxonomy, "parent={$term->term_id}&menu_order=ASC&hide_empty=0");
1676         if( !empty($children) ) {
1677             $index = woocommerce_order_terms( $the_term, $next_id, $taxonomy, $index, $children );
1678         }
1679     }
1680 
1681     // no nextid meaning our term is in last position
1682     if( $term_in_level && null === $next_id )
1683         $index = woocommerce_set_term_order($id, $index+1, $taxonomy, true);
1684 
1685     return $index;
