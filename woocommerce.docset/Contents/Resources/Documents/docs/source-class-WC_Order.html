 190     /**
 191      * Populates an order from the loaded post data.
 192      *
 193      * @access public
 194      * @param mixed $result
 195      * @return void
 196      */
 197     public function populate( $result ) {
 198         // Standard post data
 199         $this->id = $result->ID;
 200         $this->order_date = $result->post_date;
 201         $this->modified_date = $result->post_modified;
 202         $this->customer_note = $result->post_excerpt;
 203         $this->post_status = $result->post_status;
 204         $this->order_custom_fields = get_post_meta( $this->id );
 205 
 206         // Define the data we're going to load: Key => Default value
 207         $load_data = apply_filters( 'woocommerce_load_order_data', array(
 208             'order_key'             => '',
 209             'billing_first_name'    => '',
 210             'billing_last_name'     => '',
 211             'billing_company'       => '',
 212             'billing_address_1'     => '',
 213             'billing_address_2'     => '',
 214             'billing_city'          => '',
 215             'billing_postcode'      => '',
 216             'billing_country'       => '',
 217             'billing_state'         => '',
 218             'billing_email'         => '',
 219             'billing_phone'         => '',
 220             'shipping_first_name'   => '',
 221             'shipping_last_name'    => '',
 222             'shipping_company'      => '',
 223             'shipping_address_1'    => '',
 224             'shipping_address_2'    => '',
 225             'shipping_city'         => '',
 226             'shipping_postcode'     => '',
 227             'shipping_country'      => '',
 228             'shipping_state'        => '',
 229             'shipping_method'       => '',
 230             'shipping_method_title' => '',
 231             'payment_method'        => '',
 232             'payment_method_title'  => '',
 233             'order_discount'        => '',
 234             'cart_discount'         => '',
 235             'order_tax'             => '',
 236             'order_shipping'        => '',
 237             'order_shipping_tax'    => '',
 238             'order_total'           => '',
 239             'customer_user'         => '',
 240             'completed_date'        => $this->modified_date
 241         ) );
 242 
 243         // Load the data from the custom fields
 244         foreach ( $load_data as $key => $default ) {
 245             if ( isset( $this->order_custom_fields[ '_' . $key ][0] ) && $this->order_custom_fields[ '_' . $key ][0] !== '' ) {
 246                 $this->$key = $this->order_custom_fields[ '_' . $key ][0];
 247             } else {
 248                 $this->$key = $default;
 249             }
 250         }
 251 
 252         // Aliases
 253         $this->user_id = (int) $this->customer_user;
 254 
 255         // Get status
 256         $terms = wp_get_object_terms( $this->id, 'shop_order_status', array('fields' => 'slugs') );
 257         $this->status = (isset($terms[0])) ? $terms[0] : 'pending';
 258     }

 395     /**
 396      * Return an array of items/products within this order.
 397      *
 398      * @access public
 399      * @param string $type Types of line items to get (array or string)
 400      * @return void
 401      */
 402     public function get_items( $type = '' ) {
 403         global $wpdb, $woocommerce;
 404 
 405         if ( empty( $type ) )
 406             $type = array( 'line_item' );
 407 
 408         if ( ! is_array( $type ) )
 409             $type = array( $type );
 410 
 411         $type = array_map( 'esc_attr', $type );
 412 
 413         $line_items = $wpdb->get_results( $wpdb->prepare( "
 414             SELECT      order_item_id, order_item_name, order_item_type
 415             FROM        {$wpdb->prefix}woocommerce_order_items
 416             WHERE       order_id = %d
 417             AND         order_item_type IN ( '" . implode( "','", $type ) . "' )
 418             ORDER BY    order_item_id
 419         ", $this->id ) );
 420 
 421         $items = array();
 422 
 423         foreach ( $line_items as $item ) {
 424 
 425             // Place line item into array to return
 426             $items[ $item->order_item_id ]['name'] = $item->order_item_name;
 427             $items[ $item->order_item_id ]['type'] = $item->order_item_type;
 428             $items[ $item->order_item_id ]['item_meta'] = $this->get_item_meta( $item->order_item_id );
 429 
 430             // Put meta into item array
 431             foreach( $items[ $item->order_item_id ]['item_meta'] as $name => $value ) {
 432                 $key = substr( $name, 0, 1 ) == '_' ? substr( $name, 1 ) : $name;
 433                 $items[ $item->order_item_id ][ $key ] = $value[0];
 434             }
 435         }
 436 
 437         return apply_filters( 'woocommerce_order_get_items', $items, $this );
 438     }

 446     public function get_item_count( $type = '' ) {

 489     /**
 490      * Get taxes, merged by code, formatted ready for output.
 491      *
 492      * @access public
 493      * @return void
 494      */
 495     public function get_tax_totals() {
 496         $taxes      = $this->get_items( 'tax' );
 497         $tax_totals = array();
 498 
 499         foreach ( $taxes as $key => $tax ) {
 500 
 501             $code = $tax[ 'name' ];
 502 
 503             if ( ! isset( $tax_totals[ $code ] ) ) {
 504                 $tax_totals[ $code ] = new stdClass();
 505                 $tax_totals[ $code ]->amount = 0;
 506             }
 507 
 508             $tax_totals[ $code ]->is_compound       = $tax[ 'compound' ];
 509             $tax_totals[ $code ]->label             = isset( $tax[ 'label' ] ) ? $tax[ 'label' ] : $tax[ 'name' ];
 510             $tax_totals[ $code ]->amount           += $tax[ 'tax_amount' ] + $tax[ 'shipping_tax_amount' ];
 511             $tax_totals[ $code ]->formatted_amount  = woocommerce_price( $tax_totals[ $code ]->amount );
 512         }
 513 
 514         return apply_filters( 'woocommerce_order_tax_totals', $tax_totals, $this );
 515     }

 652      * @param bool $round (default: true)
 653      * @return float
 654      */
 655     public function get_item_subtotal( $item, $inc_tax = false, $round = true ) {
 656         if ( $inc_tax )

 660         return apply_filters( 'woocommerce_order_amount_item_subtotal', ($round) ? number_format( $price, 2, '.', '') : $price );

1267     /**

1274      * Finally, record the date of payment
1275      *

1277      * @return void
1278      */

1280         global $woocommerce;
1281

1283             unset( $woocommerce->session->order_awaiting_payment );

1286 
1287             $order_needs_processing = true;
1288 
1289             if ( sizeof( $this->get_items() ) > 0 ) {
1290 
1291                 foreach( $this->get_items() as $item ) {
1292 
1293                     if ( $item['product_id'] > 0 ) {
1294
