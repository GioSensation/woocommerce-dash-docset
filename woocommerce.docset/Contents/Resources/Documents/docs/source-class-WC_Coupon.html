112             $this->usage_limit                = absint( $coupon_data['usage_limit'] );

115             $this->apply_before_tax           = esc_html( $coupon_data['apply_before_tax'] );
116             $this->free_shipping              = esc_html( $coupon_data['free_shipping'] );
117             $this->product_categories         = is_array( $coupon_data['product_categories'] ) ? $coupon_data['product_categories'] : array();
118             $this->exclude_product_categories = is_array( $coupon_data['exclude_product_categories'] ) ? $coupon_data['exclude_product_categories'] : array();
119             $this->exclude_sale_items         = esc_html( $coupon_data['exclude_sale_items'] );

126

244     /**
245      * is_valid function.
246      *
247      * Check if a coupon is valid. Return a reason code if invalid. Reason codes:
248      *
249      * @access public
250      * @return bool|WP_Error validity or a WP_Error if not valid
251      */
252     public function is_valid() {
253         global $woocommerce;
254 
255         $error_code = null;
256         $valid      = true;
257         $error      = false;
258 
259         if ( $this->id ) {
260 
261             // Usage Limit
262             if ( $this->usage_limit > 0 ) {
263                 if ( $this->usage_count >= $this->usage_limit ) {
264                     $valid = false;
265                     $error_code = self::E_WC_COUPON_USAGE_LIMIT_REACHED;
266                 }
267             }
268 
269             // Expired
270             if ( $this->expiry_date ) {
271                 if ( current_time( 'timestamp' ) > $this->expiry_date ) {
272                     $valid = false;
273                     $error_code = self::E_WC_COUPON_EXPIRED;
274                 }
275             }
276 
277             // Minimum spend
278             if ( $this->minimum_amount > 0 ) {
279                 if ( $this->minimum_amount > $woocommerce->cart->subtotal ) {
280                     $valid = false;
281                     $error_code = self::E_WC_COUPON_MIN_SPEND_LIMIT_NOT_MET;
282                 }
283             }
284 
285             // Product ids - If a product included is found in the cart then its valid
286             if ( sizeof( $this->product_ids ) > 0 ) {
287                 $valid_for_cart = false;
288                 if ( sizeof( $woocommerce->cart->get_cart() ) > 0 ) {
289                     foreach( $woocommerce->cart->get_cart() as $cart_item_key => $cart_item ) {
290 
291                         if ( in_array( $cart_item['product_id'], $this->product_ids ) || in_array( $cart_item['variation_id'], $this->product_ids ) || in_array( $cart_item['data']->get_parent(), $this->product_ids ) )
292                             $valid_for_cart = true;
293                     }
294                 }
295                 if ( ! $valid_for_cart ) {
296                     $valid = false;
297                     $error_code = self::E_WC_COUPON_NOT_APPLICABLE;
298                 }
299             }
300 
301             // Category ids - If a product included is found in the cart then its valid
302             if ( sizeof( $this->product_categories ) > 0 ) {
303                 $valid_for_cart = false;
304                 if ( sizeof( $woocommerce->cart->get_cart() ) > 0 ) {
305                     foreach( $woocommerce->cart->get_cart() as $cart_item_key => $cart_item ) {
306 
307                         $product_cats = wp_get_post_terms($cart_item['product_id'], 'product_cat', array("fields" => "ids"));
308 
309                         if ( sizeof( array_intersect( $product_cats, $this->product_categories ) ) > 0 )
310                             $valid_for_cart = true;
311                     }
312                 }
313                 if ( ! $valid_for_cart ) {
314                     $valid = false;
315                     $error_code = self::E_WC_COUPON_NOT_APPLICABLE;
316                 }
317             }
318 
319             // Cart discounts cannot be added if non-eligble product is found in cart
320             if ( $this->type != 'fixed_product' && $this->type != 'percent_product' ) {
321 
322                 // Exclude Products
323                 if ( sizeof( $this->exclude_product_ids ) > 0 ) {
324                     $valid_for_cart = true;
325                     if ( sizeof( $woocommerce->cart->get_cart() ) > 0 ) {
326                         foreach( $woocommerce->cart->get_cart() as $cart_item_key => $cart_item ) {
327                             if ( in_array( $cart_item['product_id'], $this->exclude_product_ids ) || in_array( $cart_item['variation_id'], $this->exclude_product_ids ) || in_array( $cart_item['data']->get_parent(), $this->exclude_product_ids ) ) {
328                                 $valid_for_cart = false;
329                             }
330                         }
331                     }
332                     if ( ! $valid_for_cart ) {
333                         $valid = false;
334                         $error_code = self::E_WC_COUPON_NOT_APPLICABLE;
335                     }
336                 }
337 
338                 // Exclude Sale Items
339                 if ( $this->exclude_sale_items == 'yes' ) {
340                     $valid_for_cart = true;
341                     $product_ids_on_sale = woocommerce_get_product_ids_on_sale();
342                     if ( sizeof( $woocommerce->cart->get_cart() ) > 0 ) {
343                         foreach( $woocommerce->cart->get_cart() as $cart_item_key => $cart_item ) {
344                             if ( in_array( $cart_item['product_id'], $product_ids_on_sale, true ) || in_array( $cart_item['variation_id'], $product_ids_on_sale, true ) || in_array( $cart_item['data']->get_parent(), $product_ids_on_sale, true ) ) {
345                                 $valid_for_cart = false;
346                             }
347                         }
348                     }
349                     if ( ! $valid_for_cart ) {
350                         $valid = false;
351                         $error_code = self::E_WC_COUPON_NOT_VALID_SALE_ITEMS;
352                     }
353                 }
354 
355                 // Exclude Categories
356                 if ( sizeof( $this->exclude_product_categories ) > 0 ) {
357                     $valid_for_cart = true;
358                     if ( sizeof( $woocommerce->cart->get_cart() ) > 0 ) {
359                         foreach( $woocommerce->cart->get_cart() as $cart_item_key => $cart_item ) {
360 
361                             $product_cats = wp_get_post_terms( $cart_item['product_id'], 'product_cat', array( "fields" => "ids" ) );
362 
363                             if ( sizeof( array_intersect( $product_cats, $this->exclude_product_categories ) ) > 0 )
364                                 $valid_for_cart = false;
365                         }
366                     }
367                     if ( ! $valid_for_cart ) {
368                         $valid = false;
369                         $error_code = self::E_WC_COUPON_NOT_APPLICABLE;
370                     }
371                 }
372             }
373 
374             $valid = apply_filters( 'woocommerce_coupon_is_valid', $valid, $this );
375 
376             if ( $valid ) {
377                 return true;
378             } else {
379                 if ( is_null( $error_code ) )
380                     $error_code = self::E_WC_COUPON_INVALID_FILTERED;
381             }
382 
383         } else {
384             $error_code = self::E_WC_COUPON_NOT_EXIST;
385         }
386 
387         if ( $error_code )
388             $this->error_message = $this->get_coupon_error( $error_code );
389 
390         return false;
391     }

431     /**
432      * Map one of the WC_Coupon error codes to a message string
433      *
434      * @access public
435      * @param int $err_code Message/error code.
436      * @return string| Message/error string
437      */
438     public function get_coupon_error( $err_code ) {
439 
440         switch ( $err_code ) {
441             case self::E_WC_COUPON_INVALID_FILTERED:
442                 $err = __( 'Coupon is not valid.', 'woocommerce' );
443             break;
444             case self::E_WC_COUPON_NOT_EXIST:
445                 $err = __( 'Coupon does not exist!', 'woocommerce' );
446             break;
447             case self::E_WC_COUPON_INVALID_REMOVED:
448                 $err = sprintf( __( 'Sorry, it seems the coupon "%s" is invalid - it has now been removed from your order.', 'woocommerce' ), $this->code );
449             break;
450             case self::E_WC_COUPON_NOT_YOURS_REMOVED:
451                 $err = sprintf( __( 'Sorry, it seems the coupon "%s" is not yours - it has now been removed from your order.', 'woocommerce' ), $this->code );
452             break;
453             case self::E_WC_COUPON_ALREADY_APPLIED:
454                 $err = __( 'Coupon code already applied!', 'woocommerce' );
455             break;
456             case self::E_WC_COUPON_ALREADY_APPLIED_INDIV_USE_ONLY:
457                 $err = sprintf( __( 'Sorry, coupon "%s" has already been applied and cannot be used in conjunction with other coupons.', 'woocommerce' ), $this->code );
458             break;
459             case self::E_WC_COUPON_USAGE_LIMIT_REACHED:
460                 $err = __( 'Coupon usage limit has been reached.', 'woocommerce' );
461             break;
462             case self::E_WC_COUPON_EXPIRED:
463                 $err = __( 'This coupon has expired.', 'woocommerce' );
464             break;
465             case self::E_WC_COUPON_MIN_SPEND_LIMIT_NOT_MET:
466                 $err = sprintf( __( 'The minimum spend for this coupon is %s.', 'woocommerce' ), woocommerce_price( $this->minimum_amount ) );
467             break;
468             case self::E_WC_COUPON_NOT_APPLICABLE:
469                 $err = __( 'Sorry, this coupon is not applicable to your cart contents.', 'woocommerce' );
470             break;
471             case self::E_WC_COUPON_NOT_VALID_SALE_ITEMS:
472                 $err = __( 'Sorry, this coupon is not valid for sale items.', 'woocommerce' );
473             break;
474             default:
475                 $err = '';
476             break;
477         }
478 
479         return apply_filters( 'woocommerce_coupon_error', $err, $err_code, $this );
480     }
