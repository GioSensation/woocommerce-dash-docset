 35     /**
 36      * Get total stock.
 37      *
 38      * This is the stock of parent and children combined.
 39      *
 40      * @access public
 41      * @return int
 42      */
 43     public function get_total_stock() {
 44 
 45         if ( empty( $this->total_stock ) ) {
 46 
 47             $transient_name = 'wc_product_total_stock_' . $this->id;
 48 
 49             if ( false === ( $this->total_stock = get_transient( $transient_name ) ) ) {
 50                 $this->total_stock = $this->stock;
 51 
 52                 if ( sizeof( $this->get_children() ) > 0 ) {
 53                     foreach ($this->get_children() as $child_id) {
 54                         $stock = get_post_meta( $child_id, '_stock', true );
 55 
 56                         if ( $stock != '' ) {
 57                             $this->total_stock += intval( $stock );
 58                         }
 59                     }
 60                 }
 61 
 62                 set_transient( $transient_name, $this->total_stock );
 63             }
 64         }
 65         return apply_filters( 'woocommerce_stock_amount', $this->total_stock );
 66     }

 68     /**
 69      * Set stock level of the product.
 70      *
 71      * @access public
 72      * @param mixed $amount (default: null)
 73      * @return int Stock
 74      */
 75     function set_stock( $amount = null ) {
 76         global $woocommerce;
 77 
 78         if ( $this->managing_stock() && ! is_null( $amount ) ) {
 79 
 80             $this->stock = intval( $amount );
 81             $this->total_stock = intval( $amount );
 82             update_post_meta( $this->id, '_stock', $this->stock );
 83 
 84             // Check parents out of stock attribute
 85             if ( ! $this->backorders_allowed() && $this->get_total_stock() <= 0 )
 86                 $this->set_stock_status( 'outofstock' );
 87             elseif ( $this->backorders_allowed() || $this->get_total_stock() > 0 )
 88                 $this->set_stock_status( 'instock' );
 89 
 90             $woocommerce->clear_product_transients( $this->id ); // Clear transient
 91 
 92             return apply_filters( 'woocommerce_stock_amount', $this->stock );
 93         }

232      * @return string

234     public function get_price_html( $price = '' ) {
235 
236         // Ensure variation prices are synced with variations
237         if ( $this->min_variation_price === '' || $this->min_variation_regular_price === '' || $this->price === '' ) {
238             $this->variable_product_sync();
239             $this->price = $this->min_variation_price;

242         // Get the price
243         if ( $this->price > 0 ) {
244             if ( $this->is_on_sale() && isset( $this->min_variation_price ) && $this->min_variation_regular_price !== $this->get_price() ) {

246                 if ( ! $this->min_variation_price || $this->min_variation_price !== $this->max_variation_price )
247                     $price .= $this->get_price_html_from_text();

253             } else {

256                     $price .= $this->get_price_html_from_text();

262             }

264

266 
267         } elseif ( $this->price == 0 ) {
268 
269             if ( $this->is_on_sale() && isset( $this->min_variation_regular_price ) && $this->min_variation_regular_price !== $this->get_price() ) {
270 
271                 if ( $this->min_variation_price !== $this->max_variation_price )
272                     $price .= $this->get_price_html_from_text();
273 
274                 $price .= $this->get_price_html_from_to( $this->min_variation_regular_price, __( 'Free!', 'woocommerce' ) );

276                 $price = apply_filters( 'woocommerce_variable_free_sale_price_html', $price, $this );

278             } else {

281                     $price .= $this->get_price_html_from_text();

295     /**
296      * Return an array of attributes used for variations, as well as their possible values.
297      *
298      * @access public
299      * @return array of attributes and their available values
300      */
301     public function get_variation_attributes() {
302 
303         $variation_attributes = array();
304 
305         if ( ! $this->has_child() )
306             return $variation_attributes;
307 
308         $attributes = $this->get_attributes();
309 
310         foreach ( $attributes as $attribute ) {
311             if ( ! $attribute['is_variation'] )
312                 continue;
313 
314             $values = array();
315             $attribute_field_name = 'attribute_' . sanitize_title( $attribute['name'] );
316 
317             foreach ( $this->get_children() as $child_id ) {
318 
319                 if ( get_post_status( $child_id ) != 'publish' )
320                     continue; // Disabled
321 
322                 $child = $this->get_child( $child_id );
323 
324                 $child_variation_attributes = $child->get_variation_attributes();
325 
326                 foreach ( $child_variation_attributes as $name => $value )
327                     if ( $name == $attribute_field_name )
328                         $values[] = sanitize_title( $value );
329             }
330 
331             // empty value indicates that all options for given attribute are available
332             if ( in_array( '', $values ) ) {
333 
334                 $values = array();
335 
336                 // Get all options
337                 if ( $attribute['is_taxonomy'] ) {
338                     $post_terms = wp_get_post_terms( $this->id, $attribute['name'] );
339                     foreach ( $post_terms as $term )
340                         $values[] = $term->slug;
341                 } else {
342                     $values = array_map( 'trim', explode( '|', $attribute['value'] ) );
343                 }
344 
345                 $values = array_unique( $values );
346 
347             // Order custom attributes (non taxonomy) as defined
348             } elseif ( ! $attribute['is_taxonomy'] ) {
349 
350                 $option_names = array_map( 'trim', explode( '|', $attribute['value'] ) );
351                 $option_slugs = $values;
352                 $values       = array();
353 
354                 foreach ( $option_names as $option_name ) {
355                     if ( in_array( sanitize_title( $option_name ), $option_slugs ) )
356                         $values[] = $option_name;
357                 }
358             }
359 
360             $variation_attributes[ $attribute['name'] ] = array_unique( $values );
361         }
362 
363         return $variation_attributes;
364     }

442     /**
443      * Sync variable product prices with the children lowest/highest prices.
444      *
445      * @access public
446      * @return void
447      */
448     public function variable_product_sync() {
449         global $woocommerce;
450 
451         $children = get_posts( array(
452             'post_parent'   => $this->id,
453             'posts_per_page'=> -1,
454             'post_type'     => 'product_variation',
455             'fields'        => 'ids',
456             'post_status'   => 'publish'
457         ));
458 
459         $this->min_variation_price = $this->min_variation_regular_price = $this->min_variation_sale_price = $this->max_variation_price = $this->max_variation_regular_price = $this->max_variation_sale_price = '';
460 
461         if ( $children ) {
462             foreach ( $children as $child ) {
463 
464                 $child_price            = get_post_meta( $child, '_price', true );
465                 $child_regular_price    = get_post_meta( $child, '_regular_price', true );
466                 $child_sale_price       = get_post_meta( $child, '_sale_price', true );
467 
468                 if ( $child_price === '' && $child_regular_price === '' )
469                     continue;
470 
471                 // Regular prices
472                 if ( $child_regular_price !== '' ) {
473                     if ( ! is_numeric( $this->min_variation_regular_price ) || $child_regular_price < $this->min_variation_regular_price )
474                         $this->min_variation_regular_price = $child_regular_price;
475 
476                     if ( ! is_numeric( $this->max_variation_regular_price ) || $child_regular_price > $this->max_variation_regular_price )
477                         $this->max_variation_regular_price = $child_regular_price;
478                 }
479 
480                 // Sale prices
481                 if ( $child_sale_price !== '' ) {
482                     if ( $child_price == $child_sale_price ) {
483                         if ( ! is_numeric( $this->min_variation_sale_price ) || $child_sale_price < $this->min_variation_sale_price )
484                             $this->min_variation_sale_price = $child_sale_price;
485 
486                         if ( ! is_numeric( $this->max_variation_sale_price ) || $child_sale_price > $this->max_variation_sale_price )
487                             $this->max_variation_sale_price = $child_sale_price;
488                     }
489                 }
490 
491                 // Actual prices
492                 if ( $child_price !== '' ) {
493                     if ( $child_price > $this->max_variation_price )
494                         $this->max_variation_price = $child_price;
495 
496                     if ( $this->min_variation_price === '' || $child_price < $this->min_variation_price )
497                         $this->min_variation_price = $child_price;
498                 }
499             }
500 
501             update_post_meta( $this->id, '_price', $this->min_variation_price );
502             update_post_meta( $this->id, '_min_variation_price', $this->min_variation_price );
503             update_post_meta( $this->id, '_max_variation_price', $this->max_variation_price );
504             update_post_meta( $this->id, '_min_variation_regular_price', $this->min_variation_regular_price );
505             update_post_meta( $this->id, '_max_variation_regular_price', $this->max_variation_regular_price );
506             update_post_meta( $this->id, '_min_variation_sale_price', $this->min_variation_sale_price );
507             update_post_meta( $this->id, '_max_variation_sale_price', $this->max_variation_sale_price );
508 
509             $woocommerce->clear_product_transients( $this->id );
510         }
511     }
