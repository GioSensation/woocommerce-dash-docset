 211      * woocommerce_page_title function.
 212      *
 213      * @access public
 214      * @return void
 215      */
 216     function woocommerce_page_title() {
 217 
 218         if ( is_search() ) {
 219             $page_title = sprintf( __( 'Search Results: &ldquo;%s&rdquo;', 'woocommerce' ), get_search_query() );
 220 
 221             if ( get_query_var( 'paged' ) )
 222                 $page_title .= sprintf( __( '&nbsp;&ndash; Page %s', 'woocommerce' ), get_query_var( 'paged' ) );
 223 
 224         } elseif ( is_tax() ) {
 225 
 226             $page_title = single_term_title( "", false );
 227 
 228         } else {
 229 
 230             $shop_page_id = woocommerce_get_page_id( 'shop' );
 231             $page_title   = get_the_title( $shop_page_id );
 232 
 233         }
 234 
 235         echo apply_filters( 'woocommerce_page_title', $page_title );
 236     }
