22 function woocommerce_import_start() {
23     global $wpdb;
24 
25     if (!isset($_POST['import_id'])) return;
26     if (!class_exists('WXR_Parser')) return;
27 
28     $id = (int) $_POST['import_id'];
29     $file = get_attached_file( $id );
30 
31     $parser = new WXR_Parser();
32     $import_data = $parser->parse( $file );
33 
34     if (isset($import_data['posts'])) :
35         $posts = $import_data['posts'];
36 
37         if ($posts && sizeof($posts)>0) foreach ($posts as $post) :
38 
39             if ($post['post_type']=='product') :
40 
41                 if ($post['terms'] && sizeof($post['terms'])>0) :
42 
43                     foreach ($post['terms'] as $term) :
44 
45                         $domain = $term['domain'];
46 
47                         if (strstr($domain, 'pa_')) :
48 
49                             // Make sure it exists!
50                             if (!taxonomy_exists( $domain )) :
51 
52                                 $nicename = strtolower(sanitize_title(str_replace('pa_', '', $domain)));
53 
54                                 $exists_in_db = $wpdb->get_var( $wpdb->prepare( "SELECT attribute_id FROM " . $wpdb->prefix . "woocommerce_attribute_taxonomies WHERE attribute_name = %s;", $nicename ) );
55 
56                                 if (!$exists_in_db) :
57 
58                                     // Create the taxonomy
59                                     $wpdb->insert( $wpdb->prefix . "woocommerce_attribute_taxonomies", array( 'attribute_name' => $nicename, 'attribute_type' => 'select', 'attribute_orderby' => 'menu_order' ), array( '%s', '%s', '%s' ) );
60 
61                                 endif;
62 
63                                 // Register the taxonomy now so that the import works!
64                                 register_taxonomy( $domain,
65                                     apply_filters( 'woocommerce_taxonomy_objects_' . $domain, array('product') ),
66                                     apply_filters( 'woocommerce_taxonomy_args_' . $domain, array(
67                                         'hierarchical' => true,
68                                         'show_ui' => false,
69                                         'query_var' => true,
70                                         'rewrite' => false,
71                                     ) )
72                                 );
