 62      *
 63      * Loads all shipping methods which are hooked in. If a $package is passed some methods may add themselves conditionally.
 64      *
 65      * Methods are sorted into their user-defined order after being loaded.
 66      *
 67      * @access public
 68      * @return array
 69      */
 70     function load_shipping_methods( $package = false ) {
 71 
 72         $this->unregister_shipping_methods();
 73 
 74         // Methods can register themselves through this hook
 75         do_action( 'woocommerce_load_shipping_methods', $package );
 76 
 77         // Register methods through a filter
 78         $shipping_methods_to_load = apply_filters( 'woocommerce_shipping_methods', array() );
 79 
 80         foreach ( $shipping_methods_to_load as $method )
 81             $this->register_shipping_method( $method );
 82 
 83         $this->sort_shipping_methods();
 84 
 85         return $this->shipping_methods;

117      * Sorts shipping methods into the user defined order.

122     function sort_shipping_methods() {
123 
124         $sorted_shipping_methods = array();
125 
126         // Get order option
127         $ordering   = (array) get_option('woocommerce_shipping_method_order');
128         $order_end  = 999;
129 
130         // Load shipping methods in order
131         foreach ( $this->shipping_methods as $method ) {
132 
133             if ( isset( $ordering[ $method->id ] ) && is_numeric( $ordering[ $method->id ] ) ) {
134                 // Add in position
135                 $sorted_shipping_methods[ $ordering[ $method->id ] ][] = $method;
136             } else {
137                 // Add to end of the array
138                 $sorted_shipping_methods[ $order_end ][] = $method;
139             }
140         }
141 
142         ksort( $sorted_shipping_methods );
143 
144         $this->shipping_methods = array();
145 
146         foreach ( $sorted_shipping_methods as $methods )
147             foreach ( $methods as $method ) {
148                 $id = empty( $method->instance_id ) ? $method->id : $method->instance_id;
149                 $this->shipping_methods[ $id ] = $method;
150             }
151 
152         return $this->shipping_methods;

183     /**
184      * calculate_shipping function.
185      *
186      * Calculate shipping for (multiple) packages of cart items.
187      *
188      * @access public
189      * @param array $packages multi-dimensional array of cart items to calc shipping for
190      */
191     function calculate_shipping( $packages = array() ) {
192         global $woocommerce;
193 
194         if ( ! $this->enabled || empty( $packages ) )
195             return;
196 
197         $this->shipping_total   = 0;
198         $this->shipping_taxes   = array();
199         $this->shipping_label   = null;
200         $this->packages         = array();
201         $_cheapest_cost = $_cheapest_method = $chosen_method = '';
202 
203         // Calculate costs for passed packages
204         $package_keys       = array_keys( $packages );
205         $package_keys_size  = sizeof( $package_keys );
206 
207         for ( $i = 0; $i < $package_keys_size; $i ++ )
208             $this->packages[ $package_keys[ $i ] ] = $this->calculate_shipping_for_package( $packages[ $package_keys[ $i ] ] );
209 
210         // Get available methods (in this case methods for all packages)
211         $_available_methods = $this->get_available_shipping_methods();
212 
213         // Get chosen method
214         if ( ! empty( $woocommerce->session->chosen_shipping_method ) )
215             $chosen_method = $woocommerce->session->chosen_shipping_method;
216 
217         $woocommerce->session->available_methods_count = sizeof( $_available_methods );
218 
219         if ( sizeof( $_available_methods ) > 0 ) {
220 
221             // If not set, set a default
222             if ( empty( $chosen_method ) || ! isset( $_available_methods[ $chosen_method ] ) ) {
223 
224                 $chosen_method = apply_filters( 'woocommerce_shipping_chosen_method', get_option('woocommerce_default_shipping_method'), $_available_methods );
225 
226                 // Loops methods and find a match
227                 if ( ! empty( $chosen_method ) && ! isset( $_available_methods[ $chosen_method ] ) ) {
228                     foreach ( $_available_methods as $method_id => $method ) {
229                         if ( strpos( $method->id, $chosen_method ) === 0 ) {
230                             $chosen_method = $method->id;
231                             break;
232                         }
233                     }
234                 }
235 
236                 if ( empty( $chosen_method ) || ! isset( $_available_methods[$chosen_method] ) ) {
237 
238                     // Default to cheapest
239                     foreach ( $_available_methods as $method_id => $method ) {
240                         if ( $method->cost < $_cheapest_cost || ! is_numeric( $_cheapest_cost ) ) {
241                             $_cheapest_cost     = $method->cost;
242                             $_cheapest_method   = $method_id;
243                         }
244                     }
245                     $chosen_method = $_cheapest_method;
246                 }
247 
248                 // Store chosen method
249                 $woocommerce->session->chosen_shipping_method = $chosen_method;
250 
251                 // Do action for this chosen method
252                 do_action( 'woocommerce_shipping_method_chosen', $chosen_method );
253             }
254 
255             if ( $chosen_method ) {
256                 $this->shipping_total   = $_available_methods[ $chosen_method ]->cost;
257                 $this->shipping_taxes   = $_available_methods[ $chosen_method ]->taxes;
258                 $this->shipping_label   = $_available_methods[ $chosen_method ]->label;
259             }
260         }
261     }

263     /**
264      * calculate_shipping_for_package function.
265      *
266      * Calculates each shipping methods cost. Rates are cached based on the package to speed up calculations.
267      *
268      * @access public
269      * @param array $package cart items
270      */
271     function calculate_shipping_for_package( $package = array() ) {
272         if ( ! $this->enabled ) return false;
273         if ( ! $package ) return false;
274 
275         // Check if we need to recalculate shipping for this package
276         $package_hash = 'wc_ship_' . md5( json_encode( $package ) );
277 
278         if ( false === ( $stored_rates = get_transient( $package_hash ) ) ) {
279 
280             // Calculate shipping method rates
281             $package['rates'] = array();
282 
283             foreach ( $this->load_shipping_methods( $package ) as $shipping_method ) {
284 
285                 if ( $shipping_method->is_available( $package ) ) {
286 
287                     // Reset Rates
288                     $shipping_method->rates = array();
289 
290                     // Calculate Shipping for package
291                     $shipping_method->calculate_shipping( $package );
292 
293                     // Place rates in package array
294                     if ( ! empty( $shipping_method->rates ) && is_array( $shipping_method->rates ) )
295                         foreach ( $shipping_method->rates as $rate )
296                             $package['rates'][$rate->id] = $rate;
297                 }
298 
299             }
300 
301             // Filter the calculated rates
302             $package['rates'] = apply_filters( 'woocommerce_package_rates', $package['rates'], $package );
303 
304             // Store
305             set_transient( $package_hash, $package['rates'], 60 * 60 ); // Cached for an hour
306 
307         } else {
308 
309             $package['rates'] = $stored_rates;
310 
311         }
312 
313         return $package;
314     }

316     /**
317      * get_available_shipping_methods function.
318      *
319      * Gets all available shipping methods which have rates.
320      *
321      * @todo Currently we support 1 shipping method per order so this function merges rates - in the future we should offer
322      * 1 rate per package and list them accordingly for user selection
323      *
324      * @access public
325      * @return array
326      */
327     function get_available_shipping_methods() {
328         if ( ! $this->enabled ) return;
329         if ( empty( $this->packages ) ) return;
330 
331         // Loop packages and merge rates to get a total for each shipping method
332         $available_methods = array();
333 
334         foreach ( $this->packages as $package ) {
335             if ( ! $package['rates'] ) continue;
336 
337             foreach ( $package['rates'] as $id => $rate ) {
338 
339                 if ( isset( $available_methods[$id] ) ) {
340                     // Merge cost and taxes - label and ID will be the same
341                     $available_methods[$id]->cost += $rate->cost;
342 
343                     foreach ( array_keys( $available_methods[$id]->taxes + $rate->taxes ) as $key ) {
344                         $available_methods[$id]->taxes[$key] = ( isset( $rate->taxes[$key] ) ? $rate->taxes[$key] : 0 ) + ( isset( $available_methods[$id]->taxes[$key] ) ? $available_methods[$id]->taxes[$key] : 0 );
345                     }
346                 } else {
347                     $available_methods[$id] = $rate;
348                 }
349 
350             }
351 
352         }
353 
354         return apply_filters( 'woocommerce_available_shipping_methods', $available_methods );
355     }

384     function process_admin_options() {
385 
386         $default_shipping_method = ( isset( $_POST['default_shipping_method'] ) ) ? esc_attr( $_POST['default_shipping_method'] ) : '';
387         $method_order = ( isset( $_POST['method_order'] ) ) ? $_POST['method_order'] : '';
388

390

392             $loop = 0;

394                 $order[$method_id] = $loop;
395                 $loop++;
396             }
397         }
398

400         update_option( 'woocommerce_shipping_method_order', $order );
