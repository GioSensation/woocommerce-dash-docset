315     /**
316      * Apply inline styles to dynamic content.
317      *
318      * @access public
319      * @param mixed $content
320      * @return void
321      */
322     function style_inline( $content ) {
323 
324         if ( ! class_exists( 'DOMDocument' ) )
325             return $content;
326 
327         $dom = new DOMDocument();
328         @$dom->loadHTML( $content );
329 
330         $nodes = $dom->getElementsByTagName('img');
331 
332         foreach( $nodes as $node )
333             if ( ! $node->hasAttribute( 'style' ) )
334                 $node->setAttribute( "style", "display:inline; border:none; font-size:14px; font-weight:bold; height:auto; line-height:100%; outline:none; text-decoration:none; text-transform:capitalize;" );
335 
336         $nodes_h1 = $dom->getElementsByTagName('h1');
337         $nodes_h2 = $dom->getElementsByTagName('h2');
338         $nodes_h3 = $dom->getElementsByTagName('h3');
339 
340         foreach( $nodes_h1 as $node )
341             if ( ! $node->hasAttribute( 'style' ) )
342                 $node->setAttribute( "style", "color: " . get_option( 'woocommerce_email_text_color' ) . "; display:block; font-family:Arial; font-size:34px; font-weight:bold; margin-top: 10px; margin-right:0; margin-bottom:10px; margin-left:0; text-align:left; line-height: 150%;" );
343 
344         foreach( $nodes_h2 as $node )
345             if ( ! $node->hasAttribute( 'style' ) )
346                 $node->setAttribute( "style", "color: " . get_option( 'woocommerce_email_text_color' ) . "; display:block; font-family:Arial; font-size:30px; font-weight:bold; margin-top: 10px; margin-right:0; margin-bottom:10px; margin-left:0; text-align:left; line-height: 150%;" );
347 
348         foreach( $nodes_h3 as $node )
349             if ( ! $node->hasAttribute( 'style' ) )
350                 $node->setAttribute( "style", "color: " . get_option( 'woocommerce_email_text_color' ) . "; display:block; font-family:Arial; font-size:26px; font-weight:bold; margin-top: 10px; margin-right:0; margin-bottom:10px; margin-left:0; text-align:left; line-height: 150%;" );
351 
352         $nodes = $dom->getElementsByTagName('a');
353 
354         foreach( $nodes as $node )
355             if ( ! $node->hasAttribute( 'style' ) )
356                 $node->setAttribute( "style", "color: " . get_option( 'woocommerce_email_text_color' ) . "; font-weight:normal; text-decoration:underline;" );
357 
358         $content = $dom->saveHTML();
359 
360         return $content;
361     }

465     /**
466      * Admin Panel Options Processing
467      * - Saves the options to the DB
468      *
469      * @since 1.0.0
470      * @access public
471      * @return bool
472      */
473     public function process_admin_options() {
474 
475         // Save regular options
476         parent::process_admin_options();
477 
478         // Save templates
479         if ( ! empty( $_POST['template_html_code'] ) && ! empty( $this->template_html ) ) {
480 
481             $saved  = false;
482             $file   = get_stylesheet_directory() . '/woocommerce/' . $this->template_html;
483             $code   = stripslashes( $_POST['template_html_code'] );
484 
485             if ( is_writeable( $file ) ) {
486                 $f = fopen( $file, 'w+' );
487                 if ( $f !== FALSE ) {
488                     fwrite( $f, $code );
489                     fclose( $f );
490                     $saved = true;
491                 }
492             }
493 
494             if ( ! $saved ) {
495                 $redirect = add_query_arg( 'wc_error', urlencode( __( 'Could not write to template file.', 'woocommerce' ) ) );
496                 wp_redirect( $redirect );
497                 exit;
498             }
499         }
500         if ( ! empty( $_POST['template_plain_code'] ) && ! empty( $this->template_plain ) ) {
501 
502             $saved  = false;
503             $file   = get_stylesheet_directory() . '/woocommerce/' . $this->template_plain;
504             $code   = stripslashes( $_POST['template_plain_code'] );
505 
506             if ( is_writeable( $file ) ) {
507                 $f = fopen( $file, 'w+' );
508                 if ( $f !== FALSE ) {
509                     fwrite( $f, $code );
510                     fclose( $f );
511                     $saved = true;
512                 }
513             }
514 
515             if ( ! $saved ) {
516                 $redirect = add_query_arg( 'wc_error', __( 'Could not write to template file.', 'woocommerce' ) );
517                 wp_redirect( $redirect );
518                 exit;
519             }
520         }
521     }
