 15 /**
 16  * Attributes admin panel
 17  *
 18  * Shows the created attributes and lets you add new ones or edit existing ones.
 19  * The added attributes are stored in the database and can be used for layered navigation.
 20  *
 21  * @access public
 22  * @return void
 23  */
 24 function woocommerce_attributes() {
 25     global $wpdb, $woocommerce;
 26 
 27     // Action to perform: add, edit, delete or none
 28     $action = '';
 29     if ( ! empty( $_POST['add_new_attribute'] ) ) {
 30         $action = 'add';
 31     } elseif ( ! empty( $_POST['save_attribute'] ) && ! empty( $_GET['edit'] ) ) {
 32         $action = 'edit';
 33     } elseif ( ! empty( $_GET['delete'] ) ) {
 34         $action = 'delete';
 35     }
 36 
 37     // Add or edit an attribute
 38     if ( 'add' === $action || 'edit' === $action ) {
 39 
 40         // Security check
 41         if ( 'add' === $action ) {
 42             check_admin_referer( 'woocommerce-add-new_attribute' );
 43         }
 44         if ( 'edit' === $action ) {
 45             $attribute_id = absint( $_GET['edit'] );
 46             check_admin_referer( 'woocommerce-save-attribute_' . $attribute_id );
 47         }
 48 
 49         // Grab the submitted data
 50         $attribute_label   = ( isset( $_POST['attribute_label'] ) )   ? (string) stripslashes( $_POST['attribute_label'] ) : '';
 51         $attribute_name    = ( isset( $_POST['attribute_name'] ) )    ? woocommerce_sanitize_taxonomy_name( stripslashes( (string) $_POST['attribute_name'] ) ) : '';
 52         $attribute_type    = ( isset( $_POST['attribute_type'] ) )    ? (string) stripslashes( $_POST['attribute_type'] ) : '';
 53         $attribute_orderby = ( isset( $_POST['attribute_orderby'] ) ) ? (string) stripslashes( $_POST['attribute_orderby'] ) : '';
 54 
 55         // Auto-generate the label or slug if only one of both was provided
 56         if ( ! $attribute_label ) {
 57             $attribute_label = ucwords( $attribute_name );
 58         } elseif ( ! $attribute_name ) {
 59             $attribute_name = woocommerce_sanitize_taxonomy_name( stripslashes( $attribute_label ) );
 60         }
 61 
 62         // Forbidden attribute names
 63         // http://codex.wordpress.org/Function_Reference/register_taxonomy#Reserved_Terms
 64         $reserved_terms = array(
 65             'attachment', 'attachment_id', 'author', 'author_name', 'calendar', 'cat', 'category', 'category__and',
 66             'category__in', 'category__not_in', 'category_name', 'comments_per_page', 'comments_popup', 'cpage', 'day',
 67             'debug', 'error', 'exact', 'feed', 'hour', 'link_category', 'm', 'minute', 'monthnum', 'more', 'name',
 68             'nav_menu', 'nopaging', 'offset', 'order', 'orderby', 'p', 'page', 'page_id', 'paged', 'pagename', 'pb', 'perm',
 69             'post', 'post__in', 'post__not_in', 'post_format', 'post_mime_type', 'post_status', 'post_tag', 'post_type',
 70             'posts', 'posts_per_archive_page', 'posts_per_page', 'preview', 'robots', 's', 'search', 'second', 'sentence',
 71             'showposts', 'static', 'subpost', 'subpost_id', 'tag', 'tag__and', 'tag__in', 'tag__not_in', 'tag_id',
 72             'tag_slug__and', 'tag_slug__in', 'taxonomy', 'tb', 'term', 'type', 'w', 'withcomments', 'withoutcomments', 'year',
 73         );
 74 
 75         // Error checking
 76         if ( ! $attribute_name || ! $attribute_name || ! $attribute_type ) {
 77             $error = __( 'Please, provide an attribute name, slug and type.', 'woocommerce' );
 78         } elseif ( strlen( $attribute_name ) >= 28 ) {
 79             $error = sprintf( __( 'Slug “%s” is too long (28 characters max). Shorten it, please.', 'woocommerce' ), sanitize_title( $attribute_name ) );
 80         } elseif ( in_array( $attribute_name, $reserved_terms ) ) {
 81             $error = sprintf( __( 'Slug “%s” is not allowed because it is a reserved term. Change it, please.', 'woocommerce' ), sanitize_title( $attribute_name ) );
 82         } else {
 83             $taxonomy_exists = taxonomy_exists( $woocommerce->attribute_taxonomy_name( $attribute_name ) );
 84 
 85             if ( 'add' === $action && $taxonomy_exists ) {
 86                 $error = sprintf( __( 'Slug “%s” is already in use. Change it, please.', 'woocommerce' ), sanitize_title( $attribute_name ) );
 87             }
 88             if ( 'edit' === $action ) {
 89                 $old_attribute_name = $wpdb->get_var( "SELECT attribute_name FROM {$wpdb->prefix}woocommerce_attribute_taxonomies WHERE attribute_id = $attribute_id" );
 90                 if ( $old_attribute_name != $attribute_name && woocommerce_sanitize_taxonomy_name( $old_attribute_name ) != $attribute_name && $taxonomy_exists ) {
 91                     $error = sprintf( __( 'Slug “%s” is already in use. Change it, please.', 'woocommerce' ), sanitize_title( $attribute_name ) );
 92                 }
 93             }
 94         }
 95 
 96         // Show the error message if any
 97         if ( ! empty( $error ) ) {
 98             echo '<div id="woocommerce_errors" class="error fade"><p>' . $error . '</p></div>';
 99         } else {
100 
101             // Add new attribute
102             if ( 'add' === $action ) {
103                 $wpdb->insert(
104                     $wpdb->prefix . 'woocommerce_attribute_taxonomies',
105                     array(
106                         'attribute_label'   => $attribute_label,
107                         'attribute_name'    => $attribute_name,
108                         'attribute_type'    => $attribute_type,
109                         'attribute_orderby' => $attribute_orderby,
110                     )
111                 );
112 
113                 $action_completed = true;
114             }
115 
116             // Edit existing attribute
117             if ( 'edit' === $action ) {
118                 $wpdb->update(
119                     $wpdb->prefix . 'woocommerce_attribute_taxonomies',
120                     array(
121                         'attribute_label'   => $attribute_label,
122                         'attribute_name'    => $attribute_name,
123                         'attribute_type'    => $attribute_type,
124                         'attribute_orderby' => $attribute_orderby,
125                     ),
126                     array( 'attribute_id' => $attribute_id )
127                 );
128 
129                 if ( $old_attribute_name != $attribute_name && ! empty( $old_attribute_name ) ) {
130                     // Update taxonomies in the wp term taxonomy table
131                     $wpdb->update(
132                         $wpdb->term_taxonomy,
133                         array( 'taxonomy' => $woocommerce->attribute_taxonomy_name( $attribute_name ) ),
134                         array( 'taxonomy' => 'pa_' . $old_attribute_name )
135                     );
136 
137                     // Update taxonomy ordering term meta
138                     $wpdb->update(
139                         $wpdb->prefix . 'woocommerce_termmeta',
140                         array( 'meta_key' => 'order_pa_' . sanitize_title( $attribute_name ) ),
141                         array( 'meta_key' => 'order_pa_' . sanitize_title( $old_attribute_name ) )
142                     );
143 
144                     // Update product attributes which use this taxonomy
145                     $old_attribute_name_length = strlen( $old_attribute_name ) + 3;
146                     $attribute_name_length = strlen( $attribute_name ) + 3;
147 
148                     $wpdb->query( "
149                         UPDATE {$wpdb->postmeta}
150                         SET meta_value = REPLACE( meta_value, 's:{$old_attribute_name_length}:\"pa_{$old_attribute_name}\"', 's:{$attribute_name_length}:\"pa_{$attribute_name}\"' )
151                         WHERE meta_key = '_product_attributes'"
152                     );
153 
154                     // Update variations which use this taxonomy
155                     $wpdb->update(
156                         $wpdb->postmeta,
157                         array( 'meta_key' => 'attribute_pa_' . sanitize_title( $attribute_name ) ),
158                         array( 'meta_key' => 'attribute_pa_' . sanitize_title( $old_attribute_name ) )
159                     );
160                 }
161 
162                 $action_completed = true;
163             }
164         }
165     }
166 
167     // Delete an attribute
168     if ( 'delete' === $action ) {
169         // Security check
170         $attribute_id = absint( $_GET['delete'] );
171         check_admin_referer( 'woocommerce-delete-attribute_' . $attribute_id );
172 
173         $attribute_name = $wpdb->get_var( "SELECT attribute_name FROM {$wpdb->prefix}woocommerce_attribute_taxonomies WHERE attribute_id = $attribute_id" );
174 
175         if ( $attribute_name && $wpdb->query( "DELETE FROM {$wpdb->prefix}woocommerce_attribute_taxonomies WHERE attribute_id = $attribute_id" ) ) {
176 
177             $taxonomy = $woocommerce->attribute_taxonomy_name( $attribute_name );
178 
179             if ( taxonomy_exists( $taxonomy ) ) {
180                 $terms = get_terms( $taxonomy, 'orderby=name&hide_empty=0' );
181                 foreach ( $terms as $term ) {
182                     wp_delete_term( $term->term_id, $taxonomy );
183                 }
184             }
185 
186             $action_completed = true;
187         }
188     }
189 
190     // If an attribute was added, edited or deleted: clear cache and redirect
191     if ( ! empty( $action_completed ) ) {
192         delete_transient( 'wc_attribute_taxonomies' );
193         wp_safe_redirect( get_admin_url() . 'edit.php?post_type=product&page=woocommerce_attributes' );
194         exit;
195     }
196 
197     // Show admin interface
198     if ( ! empty( $_GET['edit'] ) )
199         woocommerce_edit_attribute();
200     else
201         woocommerce_add_attribute();
202 }
